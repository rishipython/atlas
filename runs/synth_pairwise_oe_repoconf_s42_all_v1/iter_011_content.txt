import numpy as np
import scipy.linalg.blas as blas
from numba import njit, prange

@njit(parallel=True)
def _compute_sq_dist(X):
    n = X.shape[0]
    # compute squared norms
    sq_norms = np.empty(n, dtype=np.float64)
    for i in prange(n):
        sq_norms[i] = np.dot(X[i], X[i])
    # compute dot product matrix
    # Use blas dgemm for speed
    prod = blas.dgemm(alpha=1.0, a=X, b=X.T, trans_a=False, trans_b=False)
    # combine
    out = -2 * prod
    out += sq_norms[:, None]
    out += sq_norms[None, :]
    return out

def pairwise_sq_dist(X):
    return _compute_sq_dist(X)

Thus final code:

import numpy as np
import scipy.linalg.blas as blas
from numba import njit

@njit
def _pairwise_sq_dist_numba(X):
    n = X.shape[0]
    sq_norms = np.empty(n, dtype=np.float64)
    for i in range(n):
        sq_norms[i] = X[i].dot(X[i])
    prod = blas.dgemm(alpha=1.0, a=X, b=X.T)
    out = -2*prod + sq_norms[:,None] + sq_norms[None,:]
    return out

def pairwise_sq_dist(X):
    return _pairwise_sq_dist_numba(X)
