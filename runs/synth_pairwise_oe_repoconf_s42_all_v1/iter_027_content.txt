```python
import numpy as np
import scipy.linalg as la
import numba as nb


@nb.njit(fastmath=True)
def _sq_norms(x):
    n = x.shape[0]
    out = np.empty(n, dtype=np.float64)
    for i in range(n):
        s = 0.0
        for j in range(x.shape[1]):
            v = x[i, j]
            s += v * v
        out[i] = s
    return out


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.
    
    Parameters
    ----------
    X : np.ndarray of shape (N, D), dtype float64.
    
    Returns
    -------
    np.ndarray of shape (N, N), dtype float64.
    """
    if X.size == 0:
        return np.empty((0, 0), dtype=np.float64)

    # Compute squared norms using a small numba loop.
    sq_norms = _sq_norms(X)

    # Compute Gram matrix via SciPy BLAS routine.
    gram = la.blas.dgemm(alpha=1.0,
                         a=X,
                         b=X.T,
                         trans_a=False,
                         trans_b=False)

    # Build distance matrix from the identity:
    #   ||x_i||^2 + ||x_j||^2 - 2 x_i·x_j
    dist_sq = sq_norms[:, None] + sq_norms[None, :] - 2 * gram

    # Numerical errors may produce tiny negative values;
    # clip them to zero.
    dist_sq[dist_sq < 0] = 0.0

    return dist_sq
```
