```
import numpy as np
import scipy.linalg as la
import numba

@numba.njit
def pairwise_sq_dist(X):
    sq = np.sum(X**2, axis=1)
    cross = la.blas.dgemm(alpha=1.0,
                          a=X,
                          b=X.T,
                          trans_a=False,
                          trans_b=False)
    return sq[:, None] + sq[None, :] - 2 * cross
```
