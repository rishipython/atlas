```python
import numpy as np
from scipy.linalg import blas
import numba


@numba.njit
def _row_squares(X):
    N, D = X.shape
    res = np.empty(N, dtype=np.float64)
    for i in range(N):
        s = 0.0
        for k in range(D):
            v = X[i, k]
            s += v * v
        res[i] = s
    return res


def pairwise_sq_dist(X):
    """
    Return the (N,N) pairwise squared Euclidean distance matrix.
    
    Parameters:
        X : ndarray of shape (N,D), dtype float64
    
    Returns:
        ndarray of shape (N,N), dtype float64 containing squared distances.
    """
    X = np.asarray(X, dtype=np.float64)
    N, D = X.shape

    if N == 0:
        return np.empty((0, 0), dtype=np.float64)

    # Compute squared norms per point using a small JIT‑compiled loop.
    sq_norms = _row_squares(X)

    # Gram matrix via BLAS.
    G = blas.dgemm(1.0, X, X.T)

    # Broadcasted formula: ||x_i||^2 + ||x_j||^2 - 2 x_i·x_j.
    dist = sq_norms[:, None] + sq_norms[None, :] - 2 * G

    # Numerical safety: clip tiny negatives caused by rounding.
    dist[dist < 0] = 0.0

    return dist

```
