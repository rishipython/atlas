import numpy as np
import scipy.linalg.blas as blas
from numba import njit

@njit(parallel=True)
def _assemble(norms,G):
    N = G.shape[0]
    D = np.empty((N,N), dtype=np.float64)
    for i in range(N):
        ni = normals[i]
        rowi = D[i]
        for j in range(N):
            rowi[j] = ni + normals[j] - 2*G[i,j]
    return D

def pairwise_sq_dist(X):
    Xf = np.asarray(X, dtype=np.float64)
    G = blas.dgemm(alpha=1.0,
                   a=Xf,
                   b=Xf.T,
                   trans_a=False,
                   trans_b=True,
                   overwrite_c=False)
    # compute squared norms
    sq_norms = np.sum(Xf * Xf, axis=1)
    # assemble distances
    D = _assemble(sq_norms,G)
    # clip tiny negatives
    D[D < 0.] = 0.
    return D
