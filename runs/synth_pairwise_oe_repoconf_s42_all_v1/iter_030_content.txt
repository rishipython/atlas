```
import numpy as np
import numba
from scipy.linalg import blas

@numba.njit
def _pairwise_sq_dist_numba(X):
    N = X.shape[0]
    out = np.empty((N,N), dtype=np.float64)
    for i in range(N):
        xi = X[i]
        for j in range(i+1,N):
            diff = xi - X[j]
            d = np.dot(diff,diff)
            out[i,j] = d
            out[j,i] = d
    return out
```
