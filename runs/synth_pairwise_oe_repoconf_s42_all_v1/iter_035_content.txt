```
def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    N,D = X.shape
    # compute squared norms using SciPy
    norms_sq = scipy.linalg.norm(X, axis=1)**2
    # compute Gram matrix via BLAS
    G = X @ X.T
    out = np.empty((N,N), dtype=np.float64)
    _compute(out,norms_sq,G)
    return out
```

Define _compute:

```
@numba.njit
def _compute(out,norms_sq,G):
    N = out.shape[0]
    for i in range(N):
        ni = norms_sq[i]
        for j in range(N):
            out[i,j] = ni + norms_sq[j] - 2*G[i,j]
```
