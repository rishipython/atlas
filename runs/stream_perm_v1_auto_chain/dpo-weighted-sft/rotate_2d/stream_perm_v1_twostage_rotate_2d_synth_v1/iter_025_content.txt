```
def run_solver(problem):
    logger = logging.getLogger(__name__)
    arr = problem['array']   # or problem['image']
    angle = problem['angle']
    # detect rad vs deg
    if isinstance(angle, float) and abs(angle) > 360:
        # assume rad
        angle_deg = np.degrees(angle)
    else:
        angle_deg = angle
    # options
    reshape = problem.get('reshape', False)
    order = problem.get('order', 3)
    mode = problem.get('mode', 'constant')
    cval = problem.get('cval', 0.0)
    prefilter = problem.get('prefilter', True)

    # handle trivial
    if angle_deg % 360 == 0:
        return arr.copy()

    # perform rotation
    rotated = scipy.ndimage.rotate(arr, angle_deg, reshape=reshape,
                                   order=order, mode=mode,
                                   cval=cval, prefilter=prefilter)

    # cast back if integer dtype
    if np.issubdtype(arr.dtype, np.integer):
        info = np.iinfo(arr.dtype)
        rotated = np.clip(rotated, info.min, info.max)
        rotated = rotated.astype(arr.dtype)

    return rotated
```
