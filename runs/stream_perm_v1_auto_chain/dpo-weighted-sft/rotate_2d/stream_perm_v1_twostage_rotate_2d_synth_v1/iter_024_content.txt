**Final design sketch**

```python
import logging
from typing import Any, Dict

import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    img = problem.get("data")
    if img is None or img.size == 0:
        return np.array([], dtype=np.float64)

    angle = float(problem.get("angle", 0))
    # Normalise angle
    angle %= 360

    order = int(problem.get("order", 1))
    reshape = bool(problem.get("reshape", False))
    mode = str(problem.get("mode", "constant"))
    cval = float(problem.get("cval", 0))

    logging.debug("Rotating shape=%s angle=%s reshape=%s order=%s",
                  img.shape, angle, reshape, order)

    rotated = scipy.ndimage.rotate(img,
                                   angle,
                                   reshape=reshape,
                                   order=order,
                                   mode=mode,
                                   cval=cval)

    # Preserve original dtype if integer
    if np.isfinite(rotated).all() and np.isinteger(img.dtype):
        rotated = rotated.astype(img.dtype)

    logging.debug("Rotation finished")
    return rotated
```

All heavy lifting happens inside `scipy.ndimage.rotate`, which is fully vectorised and written in C/Cython – satisfying “no Python-level loops” while staying within standard imports (`numpy`, `scipy`).  

Edge cases are handled gracefully and type preservation ensures compatibility with downstream consumers expecting the same dtype as input.
