def run_solver(problem):
    # parse
    if 'points' in problem:
        pts = np.asarray(problem['points'], dtype=float)
        angle = float(problem.get('angle',0))
        center = np.asarray(problem.get('center',[0.,0.]),dtype=float)
        # normalize angle
        theta = np.deg2rad(angle % 360)
        cos_t = np.cos(theta); sin_t = np.sin(theta)
        R = np.array([[cos_t,-sin_t],[sin_t,cos_t]],dtype=float)
        # shift
        shifted = pts - center
        rotated = shifted @ R.T
        result = rotated + center
        return {'points': result}
    elif 'image' in problem:
        img = np.asarray(problem['image'])
        angle = float(problem.get('angle',0))
        # use ndimage.rotate
        rotated_img = scipy.ndimage.rotate(img, angle=angle, reshape=False)
        return {'image': rotated_img}
    else:
        raise ValueError("Unsupported problem format")
