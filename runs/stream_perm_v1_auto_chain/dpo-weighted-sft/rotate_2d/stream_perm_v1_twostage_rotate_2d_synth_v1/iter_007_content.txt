```python
import logging
import random
from typing import Any, Dict

import numpy as np
import scipy.ndimage

# Configure basic logger once at import time
logging.basicConfig(level=logging.INFO)
log = logging.getLogger(__name__)

class Rotate2D:
    """
    Fast 2‑D rotation using scipy's ndimage.rotate.
    Expects problem dict with keys:
      - 'image': np.ndarray (H x W) or (H x W x C) – only HxW supported cleanly
      - 'angle': float – degrees counter‑clockwise
      Optional keys:
      - 'order': int – interpolation order (default 1)
      - 'reshape': bool – whether output should be reshaped to fit full rotated image (default False)
    """
    def solve(self, problem: Dict[str, Any]) -> np.ndarray:
        img = problem.get('image')
        if img is None:
            raise ValueError("Problem dict must contain 'image'")
        if not isinstance(img, np.ndarray):
            raise TypeError("'image' must be a numpy array")

        angle = problem.get('angle')
        if angle is None:
            raise ValueError("Problem dict must contain 'angle'")
        try:
            angle = float(angle)
        except Exception as exc:
            raise TypeError("'angle' must be convertible to float") from exc

        order = int(problem.get('order', 1))
        reshape = bool(problem.get('reshape', False))

        log.debug("Rotating image of shape %s by %s degrees (order=%s, reshape=%s)",
                  img.shape, angle, order, reshape)

        # Ensure we are dealing with a 2‑D array; if more dims are present,
        # we warn but proceed – ndimage will treat all trailing dims as spatial.
        if img.ndim != 2:
            log.warning("Input image has %d dimensions; expected 2-D.", img.ndim)

        rotated = scipy.ndimage.rotate(
            img,
            angle,
            reshape=reshape,
            order=order,
            mode='nearest',
            prefilter=False
        )

        log.debug("Rotation complete; output shape %s", rotated.shape)
        return rotated

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    """
    Entry point expected by AlgoTune harness.
    Delegates to Rotate2D.solve.
    """
    return Rotate2D().solve(problem)
```
