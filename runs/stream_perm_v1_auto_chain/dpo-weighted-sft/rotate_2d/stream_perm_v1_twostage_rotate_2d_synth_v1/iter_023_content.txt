import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    img = problem['image']
    angle = problem.get('angle', 0)
    reshape = problem.get('reshape', True)
    order = problem.get('order', 1)
    mode = problem.get('mode', 'nearest')
    
    logging.debug(f"Running rotate_2d on shape {img.shape} with angle {angle}")
    
    if isinstance(angle, (int, float)):
        # check near multiple of 90
        if np.isclose(angle % 90, 0):
            k = int(round(angle / 90)) % 4
            rotated = np.rot90(img, k=k)
            return rotated.astype(img.dtype)
    
    rotated = scipy.ndimage.rotate(img,
                                   angle=angle,
                                   reshape=reshape,
                                   order=order,
                                   mode=mode,
                                   prefilter=(order > 1))
    return rotated.astype(img.dtype)
