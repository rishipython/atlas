```
def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    arr = problem.get('array')
    angle = problem.get('angle')
    reshape = problem.get('reshape', True)
    order = problem.get('order', 3)
    mode = problem.get('mode', 'constant')
    cval = problem.get('cval', 0)
    if arr is None or angle is None:
        raise ValueError
    if not isinstance(arr, np.ndarray):
        arr = np.asarray(arr)
    if not np.isscalar(angle):
        raise ValueError
    # convert to float
    angle = float(angle)
    # rotate
    rotated = scipy.ndimage.rotate(arr, angle, reshape=reshape,
                                   order=order, mode=mode, cval=cval)
    return rotated
```
