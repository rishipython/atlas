```python
import logging
from typing import Any, Dict

import numpy as np
import scipy.ndimage


def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    """
    Rotate a 2‑D array (or multi‑channel image) by a given angle using
    SciPy's ndimage.rotate function. The implementation is fully vectorised
    and contains no Python‑level loops.
    
    Parameters
    ----------
    problem : dict
        Dictionary containing at least:
        - 'image' : np.ndarray
            The array to be rotated. Expected shape (H, W) or (H, W, C).
        - 'angle' : float
            Rotation angle in degrees.
    
    Returns
    -------
    np.ndarray
        The rotated array with the same dtype as the input.
    """
    # Extract inputs with minimal validation
    img = problem.get("image")
    angle = problem.get("angle", 0.0)

    if img is None:
        raise ValueError("Problem dictionary must contain an 'image' key.")

    # Ensure we are working with a NumPy array
    img = np.asarray(img)
    
    # Log basic info for debugging purposes
    logging.debug(
        "Rotating image of shape %s by %s degrees (reshape=False).",
        img.shape,
        angle,
    )
    
    # Perform rotation using SciPy's ndimage.rotate.
    # We set reshape=False to keep the original output size.
    # order=1 gives bilinear interpolation which is fast and reasonably accurate.
    rotated = scipy.ndimage.rotate(
        img,
        angle=angle,
        reshape=False,
        order=1,
        mode="constant",
        cval=0,
        prefilter=False,
    )
    
    # Ensure dtype consistency with input
    if rotated.dtype != img.dtype:
        rotated = rotated.astype(img.dtype)
    
    return rotated
```
