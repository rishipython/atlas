```python
import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    logger = logging.getLogger(__name__)

    # Point cloud case -------------------------------------------------------
    if "points" in problem:
        pts = np.asarray(problem["points"])
        angle = problem.get("angle", 0)
        rad_flag = bool(problem.get("angle_rad", False))

        theta = angle if rad_flag else np.deg2rad(angle)
        c, s = np.cos(theta), np.sin(theta)
        rot_mat = np.array([[c, -s], [s, c]], dtype=pts.dtype)

        try:
            result = pts @ rot_mat.T
        except Exception as exc:
            logger.exception("Failed to rotate points")
            raise exc

    # Image case -------------------------------------------------------------
    elif "image" in problem:
        img = np.asarray(problem["image"])
        angle = problem.get("angle", 0)
        rad_flag = bool(problem.get("angle_rad", False))

        ang_deg = float(angle) * (180 / np.pi) if rad_flag else float(angle)

        # guard against empty arrays – SciPy may complain otherwise
        if img.size == 0:
            result = img
        else:
            try:
                result = scipy.ndimage.rotate(
                    img,
                    ang_deg,
                    reshape=False,
                    order=1,
                    mode="constant",
                    cval=0.0,
                    prefilter=False,
                )
            except Exception as exc:
                logger.exception("Failed to rotate image")
                raise exc

    # Unknown format ---------------------------------------------------------
    else:
        logger.warning("Problem dict missing expected keys")
        result = None

    return {"solution": result}
```
