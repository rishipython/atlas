```
import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    logging.debug("Starting run_solver")
    seed = problem.get('seed')
    if seed is not None:
        np.random.seed(seed)
        random.seed(seed)
    if 'points' in problem:
        pts = np.asarray(problem['points'])
        if pts.size == 0:
            return {'points': pts}
        angle_deg = random.uniform(0, 360)
        theta = np.deg2rad(angle_deg)
        R = np.array([[np.cos(theta), -np.sin(theta)],
                      [np.sin(theta),  np.cos(theta)]], dtype=pts.dtype)
        rotated = pts @ R.T
        return {'points': rotated}
    elif 'array' in problem:
        arr = np.asarray(problem['array'])
        angle_deg = random.uniform(0, 360)
        rotated = scipy.ndimage.rotate(arr, angle_deg, reshape=False, order=1)
        return {'array': rotated}
    else:
        raise ValueError("Unsupported problem format")
```
