import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

class Rotate2D:
    def solve(self, problem: Dict[str, Any]) -> np.ndarray:
        arr = problem.get('array')
        angle = problem.get('angle', 0)
        if arr is None:
            raise ValueError("Problem must contain 'array'")
        if angle == 0:
            return arr.copy()
        h,w = arr.shape[:2]
        # compute center
        cy,cx = (h-1)/2.0,(w-1)/2.0
        # meshgrid of output coordinates
        y,x = np.indices((h,w))
        y_flat = y.ravel() - cy
        x_flat = x.ravel() - cx
        coords = np.vstack((x_flat,y_flat))  # shape (2,N)
        rad = np.deg2rad(angle)
        cos_sin = np.array([[np.cos(rad), -np.sin(rad)],
                            [np.sin(rad),  np.cos(rad)]])
        inv_coords = cos_sin.T @ coords  # inverse transform
        x_src = inv_coords[0,:] + cx
        y_src = inv_coords[1,:] + cy
        # stack for map_coordinates expects shape (ndim,N)
        src_coords = np.vstack((y_src,x_src))
        # For each channel if 3D array
        if arr.ndim == 3:
            channels = []
            for c in range(arr.shape[2]):
                ch = scipy.ndimage.map_coordinates(arr[:,:,c], src_coords, order=1, mode='constant', cval=0.0)
                channels.append(ch.reshape((h,w)))
            return np.stack(channels, axis=2)
        else:
            out = scipy.ndimage.map_coordinates(arr, src_coords, order=1, mode='constant', cval=0.0)
            return out.reshape((h,w))

def run_solver(problem):
    return Rotate2D().solve(problem)
