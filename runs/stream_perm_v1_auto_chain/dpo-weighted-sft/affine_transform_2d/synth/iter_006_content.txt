class AffineTransform2D:
    def __init__(self):
        self.order = 3
        self.mode = 'constant'
        self.cval = 0.0

    def solve(self, problem):
        try:
            image = np.asarray(problem["image"], dtype=np.float32)
            matrix = np.asarray(problem["matrix"], dtype=np.float32)
            # compute inverse
            A = matrix[:,:2]
            t = matrix[:,2]
            try:
                invA = np.linalg.inv(A)
            except np.linalg.LinAlgError:
                # fallback
                return {"transformed_image": []}
            invt = -invA @ t
            # grid
            n_rows, n_cols = image.shape
            # assume square
            Y, X = np.meshgrid(np.arange(n_rows), np.arange(n_cols), indexing='ij')
            # flatten
            X_flat = X.ravel()
            Y_flat = Y.ravel()
            coords_in_flat = invA @ np.vstack((X_flat, Y_flat)) + invt[:,None]
            # map_coordinates expects shape (ndim, npoints)
            transformed_flat = scipy.ndimage.map_coordinates(
                image,
                coords_in_flat,
                order=self.order,
                mode=self.mode,
                cval=self.cval,
                prefilter=True
            )
            transformed_image = transformed_flat.reshape(image.shape)
            return {"transformed_image": transformed_image}
        except Exception as e:
            logging.error(...)
            return {"transformed_image": []}
