def solve(problem):
    image = np.asarray(problem["image"], dtype=np.float32)
    M = np.asarray(problem["matrix"], dtype=np.float32)
    invM = np.linalg.inv(np.vstack([M, [0,0,1]]))[:2,:] # 2x3 inverse
    n = image.shape[0]
    xs, ys = np.meshgrid(np.arange(n), np.arange(n), indexing='ij')
    ones = np.ones_like(xs)
    coords = np.stack([xs.ravel(), ys.ravel(), ones.ravel()], axis=0)  # 3 x N
    src_coords = invM @ coords  # 2 x N
    src_coords = src_coords.reshape(2, n, n)
    # map_coordinates expects coordinates in order (row, col)
    transformed = scipy.ndimage.map_coordinates(image, [src_coords[0], src_coords[1]], order=3, mode='constant', cval=0.0)
    return {"transformed_image": transformed.reshape(n,n)}
