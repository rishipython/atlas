import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

class AffineTransform2D:
    def __init__(self):
        self.order = 1
        self.mode = 'constant'

    def solve(self, problem):
        try:
            img = np.asarray(problem["image"], dtype=np.float32)
            mat = np.asarray(problem["matrix"], dtype=np.float32)
            # Build homogeneous
            hom = np.vstack([mat, [0.,0.,1.]])
            try:
                inv_hom = np.linalg.inv(hom)
            except np.linalg.LinAlgError:
                logging.error("Singular transformation matrix.")
                return {"transformed_image": []}
            mat_inv = inv_hom[:2,:2]
            offset = inv_hom[:2,2]
            transformed = scipy.ndimage.affine_transform(
                img,
                mat_inv,
                offset=offset,
                order=self.order,
                mode=self.mode,
                cval=0.0
            )
            return {"transformed_image": transformed}
        except Exception as e:
            logging.error(f"Error in solve method: {e}")
            raise e

    def is_solution(self,...): # unchanged

def run_solver(problem):
    solver = AffineTransform2D()
    return solver.solve(problem)
