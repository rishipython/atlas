```
def solve(self, problem):
    image = problem["image"]
    matrix = problem["matrix"]
    img_arr = np.asarray(image, dtype=np.float64)
    mat_arr = np.asarray(matrix, dtype=np.float64)
    A = mat_arr[:,:2]
    t = mat_arr[:,2]
    invA = np.linalg.inv(A)
    offset = -invA @ t
    n_rows,n_cols = img_arr.shape
    # assume square
    y,x = np.mgrid[0:n_rows, 0:n_cols]
    # flatten
    xy_flat = np.vstack((y.ravel(), x.ravel()))   # shape (2,N)
    inp_coords = invA @ xy_flat + offset[:,None]
    # map coordinates
    out_flat = scipy.ndimage.map_coordinates(img_arr,
                                             inp_coords,
                                             order=self.order,
                                             mode=self.mode,
                                             cval=0.)
    out_img = out_flat.reshape(img_arr.shape)
    return {"transformed_image": out_img}
```
