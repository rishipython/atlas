```
def solve(self, problem):
    image = np.asarray(problem["image"], dtype=np.float32)
    mat = np.asarray(problem["matrix"], dtype=np.float32)
    # mat shape (2,3)
    # Compute inverse
    A = mat[:, :2]
    b = mat[:, 2]
    try:
        invA = np.linalg.inv(A)
    except np.linalg.LinAlgError:
        return {"transformed_image": []}
    invb = -invA @ b
    # Build coordinate grid
    h,w = image.shape
    y,x = np.mgrid[0:h,0:w]  # shape (h,w)
    coords_out = np.vstack((y.ravel(), x.ravel()))  # shape (2,N)
    coords_in = invA @ coords_out + invb[:,None]
    # Interpolate
    transformed_flat = scipy.ndimage.map_coordinates(image, coords_in,
                                                     order=self.order,
                                                     mode='constant',
                                                     cval=0.0,
                                                     prefilter=True)
    transformed_image = transformed_flat.reshape(h,w)
    return {"transformed_image": transformed_image}
```
