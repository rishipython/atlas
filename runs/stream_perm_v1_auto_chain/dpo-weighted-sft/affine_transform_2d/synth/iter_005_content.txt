def solve(problem):
    image = np.asarray(problem["image"], dtype=np.float64)
    M = np.asarray(problem["matrix"], dtype=np.float64)
    h,w = image.shape
    # create grid
    y_coords, x_coords = np.meshgrid(np.arange(h), np.arange(w), indexing='ij')
    ones = np.ones_like(x_coords)
    # stack
    out_coords = np.vstack((x_coords.ravel(), y_coords.ravel(), ones.ravel()))
    # transform
    in_coords = M @ out_coords
    # sample
    transformed_flat = scipy.ndimage.map_coordinates(image, in_coords, order=3, mode='constant', cval=0.0)
    transformed_image = transformed_flat.reshape(h,w)
    return {"transformed_image": transformed_image}

class AffineTransform2D:
   def __init__(self): self.order=3 self.mode="constant"

   def solve(self, problem):
        try:
            img = np.asarray(problem["image"], dtype=np.float64)
            M = np.asarray(problem["matrix"], dtype=np.float64)
            h,w = img.shape
            # generate grid
            y,x = np.meshgrid(np.arange(h), np.arange(w), indexing='ij')
            ones = np.ones_like(x)
            out_coords = np.vstack((x.ravel(), y.ravel(), ones.ravel()))
            in_coords = M @ out_coords
            flat_res = scipy.ndimage.map_coordinates(img, in_coords,
                                                     order=self.order,
                                                     mode=self.mode,
                                                     cval=0.0)
            res_img = flat_res.reshape(h,w)
            return {"transformed_image": res_img}
        except Exception:
            logging.error("Failed during transform")
            return {"transformed_image": []}
