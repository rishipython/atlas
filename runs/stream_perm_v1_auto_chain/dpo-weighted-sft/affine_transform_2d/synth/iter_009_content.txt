@numba.njit
def compute_inv_coords(A00,A01,A10,A11,t0,t1,n):
    # compute inverse matrix elements
    det = A00*A11 - A01*A10
    invA00 = A11/det
    invA01 = -A01/det
    invA10 = -A10/det
    invA11 = A00/det
    # allocate array shape (2,n*n)
    out_coords = np.empty((2,n*n), dtype=np.float64)
    idx=0
    for i in range(n):
        for j in range(n):
            # shift by t
            u = i - t0
            v = j - t1
            out_coords[0,idx] = invA00*u + invA01*v
            out_coords[1,idx] = invA10*u + invA11*v
            idx+=1
    return out_coords
