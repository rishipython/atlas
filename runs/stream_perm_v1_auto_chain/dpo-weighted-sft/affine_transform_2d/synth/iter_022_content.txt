def solve(problem):
    img = np.asarray(problem["image"], dtype=np.float64)
    mat = np.asarray(problem["matrix"], dtype=np.float64)
    n = img.shape[0]
    # Build homogeneous transform
    H = np.eye(3)
    H[:2,:] = mat
    invH = np.linalg.inv(H)
    # Generate output grid
    y,x = np.mgrid[0:n, 0:n]   # shape (n,n)
    coords = np.vstack((x.ravel(), y.ravel(), np.ones(n*n)))  # shape (3,N)
    src_coords = invH @ coords
    src_coords = src_coords[:2,:]  # ignore homogeneous row
    transformed_flat = scipy.ndimage.map_coordinates(img,
                                                     src_coords,
                                                     order=3,
                                                     mode='constant',
                                                     cval=0,
                                                     prefilter=True)
    transformed_img = transformed_flat.reshape((n,n))
