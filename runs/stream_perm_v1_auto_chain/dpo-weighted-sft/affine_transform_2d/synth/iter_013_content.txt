def solve(problem):
    img = np.asarray(problem["image"], dtype=np.float64)
    H,W = img.shape
    M = np.asarray(problem["matrix"], dtype=np.float64)
    A = M[:,:2]
    t = M[:,2]
    # generate grid
    y_grid,x_grid = np.meshgrid(np.arange(H), np.arange(W), indexing='ij')
    # flatten
    y_flat = y_grid.ravel()
    x_flat = x_grid.ravel()
    # compute input coords
    inp_coords = A @ np.vstack((y_flat,x_flat)) + t[:,None]
    # Map coordinates
    out_flat = scipy.ndimage.map_coordinates(img.astype(np.float32),
                                             [inp_coords[0], inp_coords[1]],
                                             order=3,
                                             mode='constant',
                                             cval=0,
                                             prefilter=True)
    out_img = out_flat.reshape(H,W)
    return {"transformed_image": out_img}
