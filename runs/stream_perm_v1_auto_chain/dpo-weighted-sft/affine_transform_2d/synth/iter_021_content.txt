```python
import logging
from typing import Dict, Any
import numpy as np
import scipy.ndimage


class AffineTransform2D:
    """
    Optimized affine transform solver.
    """

    def __init__(self):
        self.order = 3
        self.mode = "constant"
        self.cval = 0.0

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Apply a 2D affine transformation using SciPy's fast implementation.
        """
        try:
            # Extract inputs
            image = np.asarray(problem["image"], dtype=np.float32)
            mat = np.asarray(problem["matrix"], dtype=np.float32)

            # Separate linear part and offset
            linear = mat[:, :2]
            offset = mat[:, 2]

            # Perform transformation
            transformed = scipy.ndimage.affine_transform(
                image,
                linear,
                offset=offset,
                output_shape=image.shape,
                order=self.order,
                mode=self.mode,
                cval=self.cval,
            )

            # Return as float64 for consistency with reference
            return {"transformed_image": transformed.astype(np.float64)}

        except Exception as exc:
            logging.error(f"AffineTransform2D.solve failed: {exc}")
            return {"transformed_image": []}


def run_solver(problem):
    """
    Entry point used by the evaluation harness.
    """
    solver = AffineTransform2D()
    return solver.solve(problem)
```
