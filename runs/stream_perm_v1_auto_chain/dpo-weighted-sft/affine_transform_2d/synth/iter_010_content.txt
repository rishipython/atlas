def solve(self, problem):
    try:
        image = np.asarray(problem["image"], dtype=np.float64)
        n_rows,n_cols = image.shape
        # assume square
        n=n_rows
        matrix=np.asarray(problem["matrix"],dtype=np.float64)
        # extend
        aff_full=np.vstack([matrix,np.array([0.,0.,1.]])]
        inv_aff=np.linalg.inv(aff_full)
        # create grid
        Y,X=np.mgrid[0:n,0:n]
        # compute input coords
        y_in=inv_aff[0,0]*Y + inv_aff[0,1]*X + inv_aff[0,2]
        x_in=inv_aff[1,0]*Y + inv_aff[1,1]*X + inv_aff[1,2]
        coords=np.array([y_in,x_in])
        transformed_image=scipy.ndimage.map_coordinates(image,
            coords,
            order=self.order,
            mode=self.mode,
            cval=0)
        return {"transformed_image": transformed_image}
    except Exception as e:
         logging.error(...)
         return {"transformed_image": []}
