```
def solve(problem):
    a = np.asarray(problem['a'], dtype=np.float64)
    b = np.asarray(problem['b'], dtype=np.float64)
    # optional smoothing
    sigma = problem.get('sigma', 0)
    if sigma > 0:
        a = scipy.ndimage.gaussian_filter(a, sigma)
        b = scipy.ndimage.gaussian_filter(b, sigma)
    corr = scipy.signal.correlate2d(a, b, mode='full')
    # find peak
    idx = np.unravel_index(np.argmax(corr), corr.shape)
    H,W = a.shape
    dx = idx[0] - (H - 1)
    dy = idx[1] - (W - 1)
    return {'shift': [int(dx), int(dy)]}
```
