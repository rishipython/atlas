```python
import logging
from typing import Any, Dict

import numpy as np
import scipy.ndimage

class Shift2D:
    def __init__(self):
        self.logger = logging.getLogger(__name__)

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        # Extract images; they must have identical shapes.
        ref = problem["reference"]
        tgt = problem["target"]
        if ref.shape != tgt.shape:
            raise ValueError("Reference and target must be the same shape")

        # Phase‑correlation via FFTs.
        f_ref = np.fft.rfft2(ref)
        f_tgt = np.fft.rfft2(tgt)

        # Cross‑power spectrum – normalise magnitude.
        product = f_ref * np.conj(f_tgt)
        eps = 1e-12
        product /= np.abs(product) + eps

        # Inverse transform back to spatial domain.
        corr = np.fft.irfft2(product, s=ref.shape)

        # Optional smoothing – keeps SciPy in play.
        corr_smooth = scipy.ndimage.gaussian_filter(corr, sigma=1)

        # Locate maximum of smoothed correlation.
        peak_idx = np.unravel_index(np.argmax(corr_smooth), corr_smooth.shape)

        # Compute shift relative to image centre.
        center_y, center_x = (s // 2 for s in ref.shape)
        dy = peak_idx[0] - center_y
        dx = peak_idx[1] - center_x

        return {"shift": (dy, dx)}

def run_solver(problem):
    return Shift2D().solve(problem)
```

* The small `eps` prevents division by zero when two spectra have identical magnitude everywhere (unlikely in practice).
