class Shift2D:
    def solve(self, problem):
        A = problem['A']
        B = problem['B']
        # ensure float
        A = np.asarray(A, dtype=np.float64)
        B = np.asarray(B, dtype=np.float64)
        # compute cross-correlation
        corr = scipy.signal.correlate2d(A, B, mode='full')
        # find max index
        idx = np.unravel_index(np.argmax(corr), corr.shape)
        # compute shift relative to center
        shift_y = idx[0] - (A.shape[0]-1)
        shift_x = idx[1] - (A.shape[1]-1)
        # compute cost if needed
        ssq_A = np.sum(A*A)
        ssq_B = np.sum(B*B)
        best_corr = corr[idx]
        ssd = ssq_A + ssq_B - 2*best_corr
        return {'shift': (shift_y, shift_x), 'ssd': ssd}
