def run_solver(problem):
    # determine type
    if 'array' in problem and 'shift' in problem:
        arr = problem['array']
        dy, dx = problem['shift']
        return scipy.ndimage.shift(arr, shift=(dy, dx), order=1, mode='nearest')
    elif 'reference' in problem and 'target' in problem:
        ref = problem['reference']
        tgt = problem['target']
        # compute cross-correlation
        corr = scipy.signal.correlate2d(ref, tgt, mode='full', boundary='fill', fillvalue=0)
        # find peak
        yx = np.unravel_index(np.argmax(corr), corr.shape)
