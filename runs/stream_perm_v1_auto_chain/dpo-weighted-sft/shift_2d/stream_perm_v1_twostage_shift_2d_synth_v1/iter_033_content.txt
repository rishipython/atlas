```
from __future__ import annotations
import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

class Shift2D:
    def __init__(self):
        self.logger = logging.getLogger(__name__)

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        data = np.asarray(problem.get('data'))
        # extract shifts
        sx = problem.get('shift_x', 0)
        sy = problem.get('shift_y', 0)
        # ensure numeric
        sx = float(sx)
        sy = float(sy)
        self.logger.debug(f"Shifting array shape {data.shape} by ({sy}, {sx})")
        # perform shift
        shifted = scipy.ndimage.shift(
            data,
            shift=(sy, sx),
            order=0,
            mode='constant',
            cval=0,
            prefilter=False
        )
        return {'data': shifted}
```
