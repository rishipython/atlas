```
def solve(problem):
    X = np.asarray(problem['X'])
    Y = np.asarray(problem['Y'])
    # Compute cross-correlation
    corr = scipy.signal.correlate2d(X, Y, mode='full', boundary='fill', fillvalue=0)
    idx = np.unravel_index(np.argmax(corr), corr.shape)
    H,W = X.shape
    dx = idx[0] - H + 1
    dy = idx[1] - W + 1
    return {'shift': (dx, dy)}
```
