class Shift2D:
    def solve(self, problem: Dict[str, Any]) -> Tuple[int,int]:
        # extract arrays
        arrs=[v for v in problem.values() if isinstance(v,np.ndarray)]
        if len(arrs)<2:
            raise ValueError("Problem must contain two numpy arrays")
        a,b=arrs[0],arrs[1]
        # ensure float dtype
        a=a.astype(np.float64, copy=False)
        b=b.astype(np.float64, copy=False)
        # compute full correlation
        corr=scipy.signal.correlate(a,b,'full')
        # find argmax
        idx=np.unravel_index(np.argmax(corr), corr.shape)
        # compute offsets
        off_y=idx[0]-(a.shape[0]-1)
        off_x=idx[1]-(a.shape[1]-1)
        return off_y, off_x
