def compute_best_shift(src,tgt):
    # pad both to next power of two size sum dims
    shape = tuple(np.array(src.shape)+np.array(tgt.shape)-1)
    f_src = np.fft.rfftn(src, shape)
    f_tgt = np.fft.rfftn(np.flipud(np.fliplr(tgt)), shape)
    corr = np.fft.irfftn(f_src * f_tgt.conj(), shape)
    idx = np.unravel_index(np.argmax(corr), corr.shape)
    # compute shift
    dy = idx[0] - src.shape[0] + 1
    dx = idx[1] - src.shape[1] + 1
    return dy,dx

def run_solver(problem):
    if 'shift' in problem:
        img = problem.get('image')
        return scipy.ndimage.shift(img, problem['shift'], order=3)
    elif 'source' in problem and 'target' in problem:
        src=problem['source']; tgt=problem['target']
        dy,dx=compute_best_shift(src,tgt)
        return scipy.ndimage.shift(src,(dy,dx),order=3)
    else:
        raise ValueError("Unsupported problem format")
