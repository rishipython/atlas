```python
from __future__ import annotations
import logging
from typing import Any, Dict, Iterable, Tuple, Union
import numpy as np
import scipy.ndimage as ndi

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    # pull data out of the dict – accept two common names
    arr = problem.get('array') or problem.get('data')
    if arr is None:
        raise ValueError("Problem dict must contain 'array' or 'data'")
    
    # get the shift vector
    shift = problem.get('shift')
    if shift is None:
        raise ValueError("Problem dict must contain 'shift'")
    
    # normalise the shift into a tuple of floats
    if not isinstance(shift, (tuple,list)):
        raise TypeError("'shift' must be a tuple or list")
    shift = tuple(float(s) for s in shift)
    
    # perform the actual operation – fully vectorised inside SciPy
    out = ndi.shift(arr,
                    shift=shift,
                    order=1,
                    mode='nearest',
                    prefilter=False)
    
    # preserve original dtype if it was integer‑based
    if np.issubdtype(arr.dtype, np.integer):
        out = out.astype(arr.dtype)
    
    return out
```
