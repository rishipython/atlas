import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage
from scipy.signal import fftconvolve

def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    logger = logging.getLogger(__name__)
    logger.debug("Problem keys: %s", list(problem.keys()))
    
    # extract source & target
    src_key_candidates=['src','source','a','x']
    tgt_key_candidates=['tgt','target','b','y']
    
    src=None;tgt=None
    for k in src_key_candidates:
        if k in problem:
            src=problem[k]; break
    for k in tgt_key_candidates:
        if k in problem:
            tgt=problem[k]; break
    
    if src is None or tgt is None:
        raise ValueError("Problem must contain source and target images.")
    
    src_arr=np.asarray(src,dtype=np.float64)
    tgt_arr=np.asarray(tgt,dtype=np.float64)
    
    # Ensure same shape
    if src_arr.shape != tgt_arr.shape:
        raise ValueError(f"Source and target shapes differ: {src_arr.shape} vs {tgt_arr.shape}")
    
    # If multi-channel (>2 dims), average channels
    if src_arr.ndim > 2:
        src_arr=src_arr.mean(axis=-1)
        tgt_arr=tgt_arr.mean(axis=-1)
    
    # Optional smoothing
    sigma=problem.get('sigma',0)  # default no smoothing
    if sigma>0:
        src_arr=scipy.ndimage.gaussian_filter(src_arr,sigma=sigma)
        tgt_arr=scipy.ndimage.gaussian_filter(tgt_arr,sigma=sigma)
    
    # Cross-correlation via FFT
    corr=fftconvolve(src_arr,tgt_arr[::-1,:],mode='same')
    
    peak_idx=np.unravel_index(np.argmax(corr),corr.shape)
    
    center=(src_arr.shape[0]//2,src_arr.shape[1]//2)
    
    shift=(peak_idx[0]-center[0],peak_idx[1]-center[1])
    
    logger.debug("Computed shift %s",shift)
    
    return {'shift':shift}
