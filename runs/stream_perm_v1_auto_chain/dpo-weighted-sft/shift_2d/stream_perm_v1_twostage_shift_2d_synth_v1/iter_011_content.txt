def run_solver(problem):
    logger = logging.getLogger(__name__)
    arr = np.asarray(problem['data'])
    shift = problem.get('shift', (0,0))
    if isinstance(shift, (int,float)):
        shift = (shift, shift)
    # Ensure tuple length 2
    if len(shift)!=2:
        raise ValueError
    # Check if both integers
    if all(abs(s - round(s)) < 1e-8 for s in shift):
        s0,s1 = map(int, map(round, shift))
        res = np.roll(arr, shift=s0, axis=0)
        res = np.roll(res, shift=s1, axis=1)
    else:
        order = problem.get('order',1)
        mode = problem.get('mode','nearest')
        res = scipy.ndimage.shift(arr, shift=shift[::-1], order=order, mode=mode) # note axis order
    return {'result': res}
