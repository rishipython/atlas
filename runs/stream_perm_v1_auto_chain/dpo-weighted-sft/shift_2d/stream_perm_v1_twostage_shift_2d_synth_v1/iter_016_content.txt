```
import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem):
    # extract data
    a = np.asarray(problem['a'], dtype=np.float64)
    b = np.asarray(problem['b'], dtype=np.float64)
    # compute fft-based cross-correlation
    A = np.fft.fftn(a)
    B = np.fft.fftn(b)
    corr = np.real(np.fft.ifftn(np.conj(A)*B))
    # optional SciPy usage
    corr = scipy.ndimage.shift(corr, shift=(0,0), mode='nearest')
    # find peak
    idx = np.unravel_index(np.argmax(corr), corr.shape)
    # compute center
    cy, cx = (np.array(corr.shape)//2)
    dy = idx[0] - cy
    dx = idx[1] - cx
    return {'shift': [int(dy), int(dx)]}
```
