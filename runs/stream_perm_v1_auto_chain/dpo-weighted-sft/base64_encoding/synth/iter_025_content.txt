```python
class Base64Encoding:
    def solve(self, problem: Dict[str, Any]) -> str:
        msg = problem['message']
        if isinstance(msg, str):
            msg_bytes = msg.encode()
        else:
            msg_bytes = msg

        key = problem.get('key')
        if key is not None:
            if isinstance(key, str):
                key_bytes = key.encode()
            else:
                key_bytes = key
            digest = hmac.new(key_bytes, msg_bytes, hashlib.sha256).digest()
            out_bytes = base64.b64encode(digest)
        else:
            out_bytes = base64.b64encode(msg_bytes)

        return out_bytes.decode()
```
