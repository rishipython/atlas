import base64
import numpy as np
from typing import Any

BASE64_ALPHABET = np.frombuffer(b'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/', dtype=np.uint8)

def encode_base64_vectorized(data_bytes: bytes) -> str:
    if not data_bytes:
        return ''
    arr = np.frombuffer(data_bytes, dtype=np.uint8)
    n = arr.size
    pad_len = (3 - n % 3) % 3
    if pad_len:
        arr_padded = np.pad(arr, (0,pad_len), mode='constant')
    else:
        arr_padded = arr
    # reshape
    groups = arr_padded.reshape(-1,3)
    # compute int24
    int24 = ((groups[:,0].astype(np.uint32)<<16) |
             (groups[:,1].astype(np.uint32)<<8) |
             groups[:,2].astype(np.uint32))
    # extract indices
    idx0 = (int24 >>18)&63
    idx1 = (int24 >>12)&63
    idx2 = (int24 >>6)&63
    idx3 = int24 &63
    # stack indices
    indices = np.stack([idx0,idx1,idx2,idx3], axis=1).reshape(-1)
    # map to chars
    encoded_arr = BASE64_ALPHABET[indices]
    # apply padding if needed
    if pad_len:
        encoded_arr[-pad_len:] = ord('=')
    return encoded_arr.tobytes().decode('ascii')

def run_solver(problem: Any):
   data = problem.get('data')
   if isinstance(data,str):
       data_bytes=data.encode()
   else:
       data_bytes=data
   return encode_base64_vectorized(data_bytes)
