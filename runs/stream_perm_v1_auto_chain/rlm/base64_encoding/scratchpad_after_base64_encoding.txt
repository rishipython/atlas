    def generate_problem(self, n, random_seed=1):
        """
        Generate random binary data for the Base64 encoding task.
        """
        size = n * self.DEFAULT_PLAINTEXT_MULTIPLIER
        if size <= 0:
            return {"plaintext": b""}

        rng = np.random.default_rng(random_seed)

        """
        Encode the plaintext using the standard Base64 algorithm.
