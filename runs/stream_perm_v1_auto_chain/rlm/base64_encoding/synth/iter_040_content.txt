def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    # Extract data array and key
    arr_data = problem.get('data', [])
    key = problem.get('key')
    # Convert arr_data to big-endian uint32 numpy array
    arr_np = np.array(arr_data, dtype='>u4')
    # View as uint8
    byte_buf = arr_np.view(np.uint8)
    # Base64 encode entire buffer
    b64_bytes = base64.b64encode(byte_buf.tobytes())
    # Compute HMAC over raw bytes (not the ascii b64)
    mac_hex = hmac.new(key=key,
                       msg=b64_bytes,
                       digestmod='sha256').hexdigest()
    # Return dictionary
    return {'encoded': b64_bytes.decode(),
            'hmac': mac_hex}
```
