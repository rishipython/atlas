class Base64Encoding:
    def solve(self, problem):
        data = problem.get('data') or problem.get('input') or ''
        if isinstance(data, str):
            b = data.encode()
        else:
            b = bytes(data)
        # vectorized encoding
        import numpy as np
        tbl = np.frombuffer(b'AABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/', dtype=np.uint8)
        # Actually create mapping array
        tbl = np.array([ord(c) for c in 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'], dtype=np.uint8)
        # Convert bytes to uint8 array
        arr = np.frombuffer(b, dtype=np.uint8)
        n = len(arr)
        pad_len = (-n) % 3
        if pad_len:
            arr = np.concatenate([arr, np.zeros(pad_len, dtype=np.uint8)])
        arr = arr.reshape(-1,3)
        ints = (arr[:,0]<<16)|(arr[:,1]<<8)|arr[:,2]
        idx0 = (ints>>18)&0x3F
        idx1 = (ints>>12)&0x3F
        idx2 = (ints>>6)&0x3F
        idx3 = ints&0x3F
        out_bytes = np.concatenate([tbl[idx0], tbl[idx1], tbl[idx2], tbl[idx3]])
        # add padding '='
        if pad_len:
            out_bytes[-pad_len:] = ord('=')
        return out_bytes.tobytes().decode()

    def solve(self, problem):
         raw_input_lines = problem.get('input')
         if raw_input_lines is None:
             raw_input_lines = []
         # If list convert join else str
         if isinstance(raw_input_lines, list):
             data_str=''.join(raw_input_lines)
         else:
             data_str=str(raw_input_lines)
         # encode
         return self.encode_vectorized(data_str)

    def encode_vectorized(self,data_str):
