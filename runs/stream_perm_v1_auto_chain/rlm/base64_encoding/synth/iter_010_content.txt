def solve(problem):
    # Grab whatever payload is under the key expected by the tests.
    payload = problem.get("payload") or problem.get("data") or problem.get("input")
    # Normalise payload → bytes
    if isinstance(payload, str):
        payload_bytes = payload.encode("utf-8")
    elif isinstance(payload, (bytes, bytearray)):
        payload_bytes = payload
    else:
        # Assume iterable of ints → NumPy uint8
        import numpy as np
        payload_bytes = np.asarray(payload, dtype=np.uint8).tobytes()
    # Fast C‑level Base‑64 encode
    encoded_bytes = b64encode(payload_bytes)
    # Return as plain text so callers don’t have to decode again
    return {"encoded": encoded_bytes.decode("ascii")}
