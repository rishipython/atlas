def solve(problem: Dict[str, Any]) -> str:
    # Problem expects 'data' key containing str
    raw = problem.get('data', '')
    # Ensure it's bytes
    if isinstance(raw, str):
        raw_bytes = raw.encode('utf-8')

    # Encode
    encoded_bytes = base64.b64encode(raw_bytes)

```
