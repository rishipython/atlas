```
def solve(problem):
    data = problem.get('data', b'')
    if not data:
        return ''
    arr = np.frombuffer(data, dtype=np.uint8)
    n = arr.size
    pad_len = (3 - n % 3) % 3
    if pad_len:
        arr = np.pad(arr, (0,pad_len), mode='constant')
    # reshape
    blocks = arr.reshape(-1,3).astype(np.uint32)
    vals = (blocks[:,0]<<16)|(blocks[:,1]<<8)|blocks[:,2]
    g0 = (vals>>18)&63
    g1 = (vals>>12)&63
    g2 = (vals>>6)&63
    g3 = vals&63
    indices = np.stack([g0,g1,g2,g3], axis=1).reshape(-1)
    table = np.frombuffer(b'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/', dtype=np.uint8)
    out_bytes = table[indices]
    # add padding
    if pad_len == 1:
        out_bytes[-2:] = ord('=')
    elif pad_len ==2:
        out_bytes[-1:] = ord('=')
    return out_bytes.tobytes().decode('ascii')
```
