import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage
import jax
import jax.numpy as jnp

class Shift2D:
    def __init__(self):
        pass

    @staticmethod
    @jax.jit
    def _compute_shift(a_jnp, b_jnp):
        # determine pad shape = next power of two >= sum dims -1?
        # easier: pad each dimension individually:
        shape_a = a_jnp.shape
        shape_b = b_jnp.shape
        max_shape = tuple(max(sa,sb) for sa,sb in zip(shape_a,shape_b))
        pad_shape = tuple(1 << (ms - 1).bit_length() for ms in max_shape)
        # pad arrays:
        a_pad = jnp.pad(a_jnp,(0,pad_shape[0]-shape_a[0]),mode='constant')
        b_pad = jnp.pad(b_jnp,(0,pad_shape[1]-shape_b[1]),mode='constant')
        # compute fft:
        fa = jnp.fft.fftn(a_pad)
        fb = jnp.fft.fftn(b_pad)
        corr = jnp.real(jnp.ifftn(jnp.conj(fa)*fb))
        # flatten & find argmax:
        idx = jnp.argmax(corr)
        y,x = jnp.unravel_index(idx,corr.shape)
        # adjust shifts relative to original size:
        dy = y - (pad_shape[0]//2)
        dx = x - (pad_shape[1]//2)
        return dy.astype(jnp.int32), dx.astype(jnp.int32)

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        a_np = np.asarray(problem['a'])
        b_np = np.asarray(problem['b'])
        # convert to float64 if not already:
        if a_np.dtype != np.float64:
            a_np = a_np.astype(np.float64)
        if b_np.dtype != np.float64:
            b_np = b_np.astype(np.float64)
        # compute shift via jit function:

        # optional smoothing using scipy:
