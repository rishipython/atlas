class Shift2D:
    def solve(self, problem):
        src = np.asarray(problem['source'])
        tgt = np.asarray(problem['target'])
        # ensure float32 for speed
        src_f = src.astype(np.float32)
        tgt_f = tgt.astype(np.float32)
        # compute cross-correlation via FFT
        corr = sps.correlate(src_f, tgt_f, mode='valid')
        y,x = np.unravel_index(np.argmax(corr), corr.shape)
        # compute actual shift relative positions
        # Since mode='valid', the top-left corner of tgt aligns at (0,0) in src.
        # So shift needed is (y,x) offset.
        # Apply shift using jax for speed
        shifted_tgt = jshift(tgt_f, shift=(y,x), mode='nearest')
        return {'shift': (int(y), int(x)), 'aligned': np.array(shifted_tgt)}
