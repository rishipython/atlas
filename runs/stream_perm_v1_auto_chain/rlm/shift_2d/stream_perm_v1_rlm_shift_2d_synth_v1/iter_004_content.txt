```
def solve(self, problem):
    a = problem['a']
    b = problem['b']
    # ensure float64
    a = np.asarray(a, dtype=np.float64)
    b = np.asarray(b, dtype=np.float64)
    # compute padded size
    M,N = a.shape
    P,Q = b.shape
    size = (M+P-1, N+Q-1)
    # next power of two for speed
    size_fft = [int(2**np.ceil(np.log2(s))) for s in size]
    # pad
    A_pad = np.zeros(size_fft)
    B_pad = np.zeros(size_fft)
    A_pad[:M,:N] = a
    B_pad[:P,:Q] = b
    # fft
    A_fft = np.fft.rfftn(A_pad)
    B_fft = np.fft.rfftn(B_pad)
    C_fft = A_fft * np.conj(B_fft)
    corr = np.fft.irfftn(C_fft, s=size_fft).real
    # crop back to full size
    corr_full = corr[:size[0], :size[1]]
    idx_flat = np.argmax(corr_full)
    i,j = np.unravel_index(idx_flat,corr_full.shape)
    dx = i - (P-1)
    dy = j - (Q-1)
    return {"shift":[int(dx),int(dy)]}
```
