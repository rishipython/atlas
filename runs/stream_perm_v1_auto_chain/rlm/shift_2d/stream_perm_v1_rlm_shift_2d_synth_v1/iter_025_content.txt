import numpy as np
import scipy.ndimage
import scipy.signal

def run_solver(problem: Dict[str, Any]) -> Any:
    src = np.asarray(problem['source'])
    tgt = np.asarray(problem['target'])
    # compute cross-correlation
    corr = scipy.signal.fftconvolve(src, tgt[::-1, ::-1], mode='full')
    # find peak
    max_idx = np.unravel_index(np.argmax(corr), corr.shape)
    # compute shift relative to center
    center = tuple(s//2 for s in corr.shape)
    shift = tuple(m - c for m,c in zip(max_idx, center))
    # apply shift
    shifted = scipy.ndimage.shift(src, shift=shift, order=3, mode='constant', cval=0)
    return {'shift': shift, 'aligned': shifted}
