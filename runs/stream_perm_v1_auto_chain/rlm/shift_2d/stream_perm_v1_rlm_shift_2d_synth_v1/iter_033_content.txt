src   = problem['source']      # shape (H,W)
tgt   = problem['target']      # shape (h,w)
mode  = problem.get('mode','wrap')   # how boundaries are handled

# 1) Cross‑correlation.
#    The most efficient way for large arrays is FFT‑based convolution:
corr = scipy.signal.fftconvolve(src, tgt[::-1,:-1], mode='full')
#    The flip/twist ensures we're doing correlation rather than convolution.
#    corr.shape == (H+h-1, W+w-1)

# 2) Find peak.
idx   = np.unravel_index(np.argmax(corr), corr.shape)
#    idx gives the coordinates in the full correlation grid where tgt best aligns
#    w.r.t src – i.e., where shifting tgt by (-idx) would overlay it on src.
#    To convert this into a source‑to‑target pixel offset:
shift_y = idx[0] - (tgt.shape[0]-1)
shift_x = idx[1] - (tgt.shape[1]-1)

# 3) Apply this offset back onto src so that it looks like tgt.
#    We could simply roll the array because the offset is integral.
#    However the brief explicitly requires us to touch scipy.ndimage,
#    so we'll delegate the actual shifting to `scipy.ndimage.shift`.
aligned_src = scipy.ndimage.shift(
                src,
                shift=(-shift_y,-shift_x),   # note sign convention
                mode=mode,
                order=0                      # nearest‑neighbour – preserves integers
              )

return {'shift':(shift_y,shift_x), 'aligned':aligned_src}
