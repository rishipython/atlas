def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    img = problem['image']
    angle = problem['angle']
    # compute rad
    theta = np.deg2rad(angle)
    cos_t, sin_t = np.cos(theta), np.sin(theta)
    matrix = np.array([[cos_t, -sin_t], [sin_t, cos_t]], dtype=np.float32)
    # compute center
    if img.ndim == 2:
        h,w = img.shape
        center = np.array([h-1,w-1], dtype=np.float32)/2
    else:
        h,w,c = img.shape
        center = np.array([h-1,w-1], dtype=np.float32)/2
    offset = center - matrix @ center
    # apply transform
    rotated = scipy.ndimage.affine_transform(img, matrix, offset=offset, order=1, mode='nearest')
    return rotated

That overhead hurts performance when we only need a simple rotation about the image centre without changing the output shape.

A better approach is to use `scipy.ndimage.affine_transform`. This takes an explicit linear transformation matrix and an offset vector so we can rotate around the centre directly and avoid any reshaping logic altogether. It also applies the transformation in one low‑level C loop per pixel per channel – perfect for speed and vectorisation guarantees (no Python‑level loops).
