import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def _rotate_matrix(mat: np.ndarray, angle: float) -> np.ndarray:
    """
    Rotate a 2‑D array around its centre by `angle` degrees.
    Uses SciPy's ndimage.rotate which is heavily optimised in C.
    The `reshape` flag is set to False so the output keeps the same
    dimensions as the input – this matches the behaviour of most
    reference implementations that simply apply an affine transform.
    """
    # SciPy expects angles in degrees counter‑clockwise.
    # Use bilinear interpolation (order=1) for a good trade‑off.
    return scipy.ndimage.rotate(mat, angle=angle, reshape=False, order=1)

def solve(problem: Dict[str, Any]) -> np.ndarray:
    """
    Problem dictionary must contain:
        - 'matrix': a 2‑D iterable (list of lists or ndarray)
        - 'angle' : rotation angle in degrees (float)
    
    Returns the rotated matrix as a NumPy array.
    """
    mat = np.asarray(problem["matrix"], dtype=float)
    angle = float(problem["angle"])
    return _rotate_matrix(mat, angle)

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    """
    Entry point expected by the evaluation harness.
    
    Parameters
    ----------
    problem : dict
        Dictionary describing the 2‑D rotation task.
    
    Returns
    -------
    ndarray
        Rotated matrix.
    
    Notes
    -----
    The function deliberately contains no Python‑level loops;
    all heavy lifting is delegated to SciPy's C implementation,
    ensuring maximum speed while keeping the code concise.
    
    Random is imported only because the design brief requires it,
    though it is not used in this particular implementation.
    
    """
    return solve(problem)
