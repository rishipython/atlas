```
class Rotate2D:
    def solve(self, problem):
        # extract points
        pts = np.array(problem['points'], dtype=np.float64)
        if pts.size == 0:
            return [0.0]
        # center
        pts -= pts.mean(axis=0)
        # covariance
        C = pts.T @ pts
        # eigh
        eigvals, eigvecs = scipy.linalg.eigh(C)
        # pick largest eigenvector
        v = eigvecs[:, -1]
        theta = np.arctan2(v[1], v[0])
        return [float(theta)]
```
