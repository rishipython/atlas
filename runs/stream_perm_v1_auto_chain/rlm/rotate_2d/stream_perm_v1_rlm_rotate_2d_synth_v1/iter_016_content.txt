def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    img = problem['image']  # 2D array
    angle = problem.get('angle', 0.0)
    # convert degrees to radians
    rad = np.deg2rad(angle)
    cos = np.cos(rad)
    sin = np.sin(rad)
    # rotation matrix
    R = np.array([[cos, -sin], [sin, cos]], dtype=np.float64)
    # compute center
    h,w = img.shape
    center = np.array([h/2., w/2.])
    # offset for affine_transform: shift by center, apply rotation, shift back
    offset = center - R @ center
    # use affine_transform
    rotated = scipy.ndimage.affine_transform(img, R, offset=offset, order=1, mode='constant', cval=0.0)
    return rotated
