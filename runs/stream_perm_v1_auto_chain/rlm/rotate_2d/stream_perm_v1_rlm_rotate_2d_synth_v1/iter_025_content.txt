But there’s still a corner case where SciPy’s rotation is slower than a pure NumPy rotation: when the requested angle is an exact multiple of 90 degrees. import random
from typing import Any, Dict

import numpy as np
import scipy.ndimage


class Rotate2D:
    def solve(self, problem: Dict[str, Any]) -> np.ndarray:
        """
        Rotate an image by an arbitrary angle using SciPy's ndimage.rotate,
        falling back to NumPy's rot90 for multiples of ninety degrees.
        """
        img = problem["image"]
        # Default angle zero if not provided.
        angle = float(problem.get("angle", 0))
        
        # Quick exit for zero rotation.
        if abs(angle) < 1e-8:
            return img.copy()
        
        # Normalise angle into [0,360).
        ang_mod = angle % 360
        
        # Check for exact multiples of ninety degrees within tolerance.
        tol = 1e-6
        if abs(ang_mod -   0) < tol or abs(ang_mod -   360) < tol:
            # Already handled above but keep for clarity.
            return img.copy()
        elif abs(ang_mod -   90) < tol:
            out = np.rot90(img)
            return out.astype(img.dtype)
        elif abs(ang_mod -   180) < tol:
            out = np.rot90(img, k=2)
            return out.astype(img.dtype)
        elif abs(ang_mod -   270) < tol:
            out = np.rot90(img, k=3)
            return out.astype(img.dtype)
        
        # General case: use SciPy's rotate.
        out = scipy.ndimage.rotate(
            img,
            ang_mod,
            reshape=False,
            order=1,
            mode="constant",
            cval=0,
            prefilter=False,
        )
        
        # Preserve original dtype exactly.
        if out.dtype != img.dtype:
            out = out.astype(img.dtype)
        
        return out


def run_solver(problem):
    """
    Entry point expected by AlgoTune harness.
    """
