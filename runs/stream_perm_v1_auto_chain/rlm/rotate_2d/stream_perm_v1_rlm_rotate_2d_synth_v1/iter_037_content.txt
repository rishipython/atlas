class Rotate2D:
    def solve(self, problem):
        # extract data & angle
        arr = problem['matrix']
        angle_deg = problem.get('angle', 0)
        # convert degrees -> rad
        theta = np.deg2rad(angle_deg)
        c,s = np.cos(theta), np.sin(theta)
        R = np.array([[c,-s],[s,c]], dtype=float)
        # compute offset
        h,w = arr.shape
        center = np.array([h/2., w/2.], dtype=float)
        offset = center - R @ center
        # perform transform
        out = scipy.ndimage.affine_transform(arr,
                                             matrix=R.T,
                                             offset=offset,
                                             order=0,
                                             mode='constant',
                                             cval=0)
        # cast back type
        out = out.astype(arr.dtype)
        return out
