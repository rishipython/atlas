def _affine_transform_numba(img_in: np.ndarray,
                            H_inv: np.ndarray,
                            n:int):
    out = np.empty_like(img_in)
    # iterate over i,j
    for i in range(n):
        for j in range(n):
            # compute source coord
            y = H_inv[0,0]*j + H_inv[0,1]*i + H_inv[0,2]
            x = H_inv[1,0]*j + H_inv[1,1]*i + H_inv[1,2]
            # Now perform bicubic interpolation on img_in at (x,y)
            # Compute integer floor
            ix = int(np.floor(x))
            iy = int(np.floor(y))
            tx = x - ix
            ty = y - iy
            # compute weights along x
            wx = _cubic_weights(tx)
            wy = _cubic_weights(ty)
            # accumulate weighted sum over k,l from -1..2
            val=0.
            for m in range(-1,3):
                xm = ix + m
                if xm < 0 or xm >= n:
                    continue
                wxm = wx[m+1] # index shift
                for n_ in range(-1,3):
                    yn = iy + n_
                    if yn < 0 or yn >= n:
                        continue
                    wyn = wy[n_+1]
                    val += img_in[yn,xm]*wxm*wyn
            out[i,j]=val

Return out
