```
def solve(self, problem):
    image = np.asarray(problem["image"])
    mat = np.asarray(problem["matrix"])
    A = mat[:, :2]
    offset = mat[:, 2]
    transformed_image = scipy.ndimage.affine_transform(
        image,
        A,
        offset=offset,
        order=self.order,
        mode=self.mode
    )
    return {"transformed_image": transformed_image}
```
