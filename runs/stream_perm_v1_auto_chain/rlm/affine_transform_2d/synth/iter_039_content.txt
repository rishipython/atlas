```python
import logging
from typing import Dict, Any
import numpy as np
import scipy.ndimage

class AffineTransform2D:
    """
    Optimised affine transform solver.
    """

    def __init__(self):
        """Initialise parameters."""
        self.order = 3
        self.mode = "constant"

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Apply a 2‑D affine transformation using SciPy's fast C implementation.
        """
        # Convert inputs once – avoid repeated Python‑level conversions
        img = np.asarray(problem["image"], dtype=np.float64)
        mat = np.asarray(problem["matrix"], dtype=np.float64)

        # Perform the transformation – no try/except overhead
        transformed_image = scipy.ndimage.affine_transform(
            img,
            mat,
            order=self.order,
            mode=self.mode,
            output_shape=img.shape,
        )
        return {"transformed_image": transformed_image}

    def is_solution(self, problem: Dict[str, Any], solution: Dict[str, Any]) -> bool:
        """
        Verify that the solution matches SciPy's reference implementation.
        """
        try:
            if not all(k in problem for k in ["image", "matrix"]):
                logging.error("Missing required keys in problem.")
                return False

            img_ref = np.asarray(problem["image"], dtype=np.float64)
            mat_ref = np.asarray(problem["matrix"], dtype=np.float64)

            if not isinstance(solution, dict) or "transformed_image" not in solution:
                logging.error("Solution missing 'transformed_image' key.")
                return False

            out = solution["transformed_image"]
            if isinstance(out, list):
                out_arr = np.asarray(out, dtype=float)
            else:
                out_arr = out

            # Ensure shape matches
            if out_arr.shape != img_ref.shape:
                logging.error(f"Shape mismatch: {out_arr.shape} vs {img_ref.shape}")
                return False

            # Reference computation
            ref_out = scipy.ndimage.affine_transform(
                img_ref,
                mat_ref,
                order=self.order,
                mode=self.mode,
                output_shape=img_ref.shape,
            )

            # Compare with tight tolerances
            return np.allclose(out_arr, ref_out, rtol=1e-5, atol=1e-7)
        except Exception as exc:
            logging.error(f"Verification error: {exc}")
            return False


def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    """
    Entry point used by the evaluation harness.
    """
    solver = AffineTransform2D()
    return solver.solve(problem)
```
