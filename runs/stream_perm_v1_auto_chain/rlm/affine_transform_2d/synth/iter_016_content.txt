```python
import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

class AffineTransform2D:
    """
    Fast affine transformation solver for 2‑D images.
    Uses SciPy's ndimage.affine_transform internally,
    avoiding any Python‑level loops or heavy error handling
    that would otherwise dominate runtime.
    """

    def __init__(self):
        # Cubic spline interpolation (order 3) keeps numerical fidelity
        # while still being reasonably fast on modern CPUs.
        self.order = 3
        self.mode = "constant"   # zero padding outside the image

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Apply an affine transformation defined by a 2×3 matrix
        to a square image stored in problem["image"].
        Returns a dictionary containing the transformed image
        under the key "transformed_image".
        """
        # Directly fetch inputs – assume they exist and are numeric.
        # Converting via np.asarray keeps the operation vectorised
        # and avoids Python loops over list elements.
        image = np.asarray(problem["image"], dtype=np.float64)
        matrix = np.asarray(problem["matrix"], dtype=np.float64)

        # Perform the transformation once – SciPy handles all heavy lifting.
        transformed = scipy.ndimage.affine_transform(
            image,
            matrix,
            order=self.order,
            mode=self.mode,
            cval=0,
            prefilter=True   # ensures correct cubic interpolation
        )

        # Return the NumPy array directly; is_solution handles conversion.
        return {"transformed_image": transformed}

def run_solver(problem):
    """
    Entry point used by the evaluation harness.
    Instantiates the solver and returns its result.
    """
    solver = AffineTransform2D()
    return solver.solve(problem)
```
