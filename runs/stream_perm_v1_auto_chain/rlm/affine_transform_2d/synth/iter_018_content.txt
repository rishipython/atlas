```
def solve(problem):
    img = np.asarray(problem["image"], dtype=np.float32)
    mat = np.asarray(problem["matrix"], dtype=np.float32)
    # compute inverse affine matrix
    M = mat.copy()
    # create 3x3 homogeneous
    H = np.eye(3)
    H[:2,:] = M
    invH = np.linalg.inv(H)
    # create grid indices
    n = img.shape[0]
    y_coords = jnp.arange(n)
    x_coords = jnp.arange(n)
    # create meshgrid flattened
    yy, xx = jnp.meshgrid(y_coords, x_coords, indexing='ij')
    ones = jnp.ones_like(yy)
    src_coords = invH @ jnp.stack([xx.ravel(), yy.ravel(), ones.ravel()], axis=0)
    src_x = src_coords[0]
    src_y = src_coords[1]
    # interpolate using jax.scipy.ndimage.map_coordinates
    from jax.scipy.ndimage import map_coordinates
    out_flat = map_coordinates(img.astype(jnp.float32), [src_y, src_x], order=3,
                               mode='constant', cval=0.)
    out = out_flat.reshape((n,n))
    return {"transformed_image": np.array(out)}
```
