```
def solve(self, problem):
    try:
        img = np.asarray(problem["image"], dtype=np.float32)
        mat = np.asarray(problem["matrix"], dtype=np.float64)
        transformed = scipy.ndimage.affine_transform(img, mat, order=self.order,
                                                    mode=self.mode)
        return {"transformed_image": transformed}
    except Exception as e:
        logging.error(f"Affine transform failed: {e}")
        return {"transformed_image": []}
```
