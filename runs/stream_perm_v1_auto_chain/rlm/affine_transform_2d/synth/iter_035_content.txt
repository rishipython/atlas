```python
import logging
from typing import Dict, Any
import numpy as np
import scipy.ndimage


class AffineTransform2D:
    """
    Fast implementation of a 2‑D affine transformation.
    Uses SciPy's `affine_transform` under the hood but avoids Python‑level loops
    and unnecessary type conversions.
    """

    def __init__(self):
        # Use cubic spline interpolation (order = 3) and constant padding.
        self.order = 3
        self.mode = "constant"

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Apply an affine transformation defined by a 2×3 matrix.
        The input image may come as a nested Python list – convert it once
        into a NumPy array of type float32 for speed.
        """
        try:
            # Convert inputs once – this costs negligible time compared
            # with the actual transform even for large images.
            image = np.asarray(problem["image"], dtype=np.float32)
            matrix = np.asarray(problem["matrix"], dtype=np.float32)

            # Guard against degenerate inputs – an empty image would raise
            # an exception inside `affine_transform`.  We simply return an
            # empty list in that case so the caller sees a consistent format.
            if image.size == 0:
                return {"transformed_image": []}

            # Perform the transformation.
            # Explicitly set `output_shape` equal to the input shape;
            # this removes any ambiguity about what size the result should be.
            transformed = scipy.ndimage.affine_transform(
                image,
                matrix,
                output_shape=image.shape,
                order=self.order,
                mode=self.mode,
                cval=0.0,
                prefilter=True,
            )

            # Convert back to plain Python lists – the verifier expects a
            # nested list structure rather than a NumPy array.
            return {"transformed_image": transformed.tolist()}

        except Exception as exc:
            logging.error(f"AffineTransform2D.solve failed: {exc}")
            # Return an empty result so that `is_solution` can decide what
            # constitutes failure versus success.
            return {"transformed_image": []}


def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    """
    Entry point used by the evaluation harness.
    Instantiates the solver class and delegates work.
    """
    solver = AffineTransform2D()
    return solver.solve(problem)
```
