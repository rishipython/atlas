```
def solve(self, problem):
    try:
        image = np.asarray(problem["image"], dtype=np.float64)
        mat = np.asarray(problem["matrix"], dtype=np.float64)
        # compute inverse homogeneous
        H = np.vstack([mat,[0,0,1]])
        H_inv = np.linalg.inv(H)
        A = H_inv[:2,:]
        # create grid
        h,w = image.shape
        yy, xx = np.indices((h,w))
        # flatten
        coord = np.vstack([xx.ravel(), yy.ravel(), np.ones_like(xx.ravel())])
        # compute mapped coords
        mapped = A @ coord
        u = mapped[0].reshape(h,w)
        v = mapped[1].reshape(h,w)
        # sample
        transformed_image = scipy.ndimage.map_coordinates(
            image,
            [v,u],  # note order y,x
            order=self.order,
            mode=self.mode,
            cval=0,
            prefilter=True,
            )
        return {"transformed_image": transformed_image}
    except Exception as e:
         logging.error(...)
         return {"transformed_image": []}
```
