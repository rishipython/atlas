import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage


class AffineTransform2D:
    """
    Fast 2D affine transformation using SciPy's ndimage module.
    """

    def __init__(self):
        self.order = 3
        self.mode = "constant"

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Apply an affine transform to a square image.
        The input image and transform matrix are provided as lists.
        Returns a dictionary containing the transformed image.
        """
        try:
            # Convert inputs to NumPy arrays with float32 precision
            image = np.asarray(problem["image"], dtype=np.float32)
            matrix = np.asarray(problem["matrix"], dtype=np.float32)

            # Validate shapes
            if image.ndim != 2:
                raise ValueError("Input image must be 2‑D.")
            if matrix.shape != (2, 3):
                raise ValueError("Affine matrix must be 2×3.")

            # Perform the transformation
            transformed_image = scipy.ndimage.affine_transform(
                image,
                matrix,
                order=self.order,
                mode=self.mode,
                output_shape=image.shape,
            )

            return {"transformed_image": transformed_image}
        except Exception as exc:
            logging.error(f"Affine transform failed: {exc}")
            return {"transformed_image": []}

    def is_solution(self, problem: Dict[str, Any], solution: Dict[str, Any]) -> bool:
        """
        Verify that the solution matches SciPy's reference implementation.
        """
        try:
            # Basic structural checks
            if not isinstance(solution, dict) or "transformed_image" not in solution:
                logging.error("Missing 'transformed_image' key.")
                return False

            # Convert solution entry to NumPy array
            out = solution["transformed_image"]
            out_arr = np.asarray(out, dtype=float)

            # Expected shape matches input
            img_shape = np.asarray(problem["image"]).shape
            if out_arr.shape != img_shape:
                logging.error(f"Shape mismatch: {out_arr.shape} vs {img_shape}")
                return False

            # Recompute reference for comparison
            ref_img = np.asarray(problem["image"], dtype=np.float32)
            ref_mat = np.asarray(problem["matrix"], dtype=np.float32)
            ref_out = scipy.ndimage.affine_transform(
                ref_img,
                ref_mat,
                order=self.order,
                mode=self.mode,
                output_shape=img_shape,
            )

            # Compare with tolerances suitable for image data
            rtol = 1e-5
            atol = 1e-7
            if not np.allclose(out_arr.astype(np.float64), ref_out.astype(np.float64), rtol=rtol, atol=atol):
                diff = np.abs(out_arr - ref_out)
                max_err = diff.max() if diff.size else 0
                logging.error(f"Solution differs from reference (max error {max_err:.6f}).")
                return False

            return True
        except Exception as exc:
            logging.error(f"is_solution error: {exc}")
            return False


def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    """
    Entry point used by the evaluator.
    """
    solver = AffineTransform2D()
    return solver.solve(problem)
