```python
import logging
import numpy as np
import scipy.ndimage
from typing import Any, Dict

class AffineTransform2D:
    """
    Optimised affine transform solver.
    Uses SciPy's ndimage.affine_transform with minimal Python overhead.
    """

    def __init__(self):
        self.order = 3
        self.mode = "constant"
        self.cval = 0

    def _validate_problem(self, problem: Dict[str, Any]) -> None:
        """Validate required keys and basic shapes."""
        if "image" not in problem or "matrix" not in problem:
            raise KeyError("Problem dictionary must contain 'image' and 'matrix'")
        img = np.asarray(problem["image"])
        mat = np.asarray(problem["matrix"])
        if img.ndim != 2 or mat.shape != (2, 3):
            raise ValueError("Invalid shapes for image or matrix")
        # Ensure image is square as expected
        if img.shape[0] != img.shape[1]:
            raise ValueError("Image must be square")

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Apply the affine transformation using SciPy.
        """
        # Validate early
        self._validate_problem(problem)

        # Convert inputs to NumPy arrays with float32 precision for speed
        image = np.asarray(problem["image"], dtype=np.float32)
        matrix = np.asarray(problem["matrix"], dtype=np.float32)

        # Perform the transform
        transformed = scipy.ndimage.affine_transform(
            image,
            matrix,
            order=self.order,
            mode=self.mode,
            cval=self.cval,
            prefilter=True,
            output=image.dtype,
        )

        # Convert back to Python list of lists for the expected output format
        return {"transformed_image": transformed.tolist()}

def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    """
    Entry point used by the evaluation harness.
    """
    solver = AffineTransform2D()
    return solver.solve(problem)
```
