```
import logging
from typing import Dict
import numpy as np
import scipy.ndimage

class AffineTransform2D:
    def __init__(self):
        self.order = 3
        self.mode = 'constant'

    def solve(self, problem):
        try:
            image = np.ascontiguousarray(problem['image'], dtype=np.float64)
            matrix = np.asarray(problem['matrix'], dtype=np.float64)
            transformed_image = scipy.ndimage.affine_transform(
                image,
                matrix,
                order=self.order,
                mode=self.mode,
                cval=0
            )
            return {'transformed_image': transformed_image}
        except Exception as exc:
            logging.error(f"Affine transform failed: {exc}")
            return {'transformed_image': []}
```
