import typing.Any, Dict
import numpy as np
import scipy.ndimage

class Shift2D:
    def solve(self, problem):
        arr = np.asarray(problem['x'])
        shift = tuple(problem['s'])
        # ensure float
        if arr.dtype.kind in 'iu':
            arr_float = arr.astype(np.float32)
            out_float = scipy.ndimage.shift(arr_float, shift=shift, order=1, mode='nearest')
            out = out_float.astype(arr.dtype)
        else:
            out = scipy.ndimage.shift(arr, shift=shift, order=1, mode='nearest')
        return out

def run_solver(problem):
    return Shift2D().solve(problem)
