```
@jax.jit
def compute_cross_corr(a,b):
    # a,b are float32 arrays
    # pad both to size H+H-1 x W+W-1
    h,w = a.shape
    pad_h = h - 1
    pad_w = w - 1
    # zero-pad
    a_pad = jnp.pad(a,(pad_h,pad_w),(pad_h,pad_w))
    b_pad = jnp.pad(b,(pad_h,pad_w),(pad_h,pad_w))
    # flip b
    b_rev = b_pad[::-1,:][:,:][::-1,:]
    # compute fft
    fa = jnp.fft.fftn(a_pad)
    fb = jnp.fft.fftn(b_rev)
    corr = jnp.real(jnp.ifftn(fa * fb))
    return corr
```
