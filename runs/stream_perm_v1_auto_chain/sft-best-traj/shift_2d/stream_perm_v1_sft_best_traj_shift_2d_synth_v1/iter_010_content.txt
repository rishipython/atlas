def run_solver(problem):
    A = problem['A']
    B = problem['B']
    # compute best integer shift
    corr = fftconvolve(A.astype(np.float32), B[::-1,...], mode='full')
    peak = np.unravel_index(np.argmax(corr), corr.shape)
    center = ((A.shape[0]-1)//2, (A.shape[1]-1)//2)
    dy = peak[0] - center[0]
    dx = peak[1] - center[1]
    # apply shift
    aligned_B = ndimage.shift(B, shift=(-dy,-dx), order=1)
    return {'aligned_B': aligned_B}
