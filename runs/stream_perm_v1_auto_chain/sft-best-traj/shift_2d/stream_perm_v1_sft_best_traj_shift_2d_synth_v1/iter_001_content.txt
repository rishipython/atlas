def solve(problem):
    img1 = problem['img1'].astype(np.float64)
    img2 = problem['img2'].astype(np.float64)

    # optionally smooth
    img1 = ndi.gaussian_filter(img1, sigma=1)
    img2 = ndi.gaussian_filter(img2, sigma=1)

    # compute fft
    F1 = np.fft.fftn(img1)
    F2 = np.fft.fftn(img2)

    # cross power spectrum
    eps = 1e-8
    R = F1 * np.conj(F2)
    R /= np.abs(R) + eps

    corr = np.real(np.ifftn(R))

    # find peak
    max_idx_flat = corr.argmax()
    max_idx = np.unravel_index(max_idx_flat, corr.shape)

    # convert index to shifts
    shifts = []

        size = corr.shape[dim]
        idx = max_idx[dim]
        if idx > size // 2:
            shifts.append(idx - size)
        else:
            shifts.append(idx)
    # convert pixel units -> float (subpixel)
