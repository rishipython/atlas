```
def run_solver(problem):
    src = problem['src']
    dst = problem['dst']
    # ensure float64
    src = src.astype(np.float64)
    dst = dst.astype(np.float64)
    # optional smoothing
    src_sm = ndimage.gaussian_filter(src, sigma=1)
    dst_sm = ndimage.gaussian_filter(dst, sigma=1)
    # cross-correlation via FFT
    f_src = np.fft.rfftn(src_sm)
    f_dst = np.fft.rfftn(dst_sm)
    corr = np.real(np.fft.irfftn(f_src * np.conj(f_dst), s=src.shape))
    # find peak
    idx = np.unravel_index(np.argmax(corr), corr.shape)
    # compute shift relative center
    center = (src.shape[0]//2, src.shape[1]//2)
    dy = idx[0] - center[0]
    dx = idx[1] - center[1]
    return (dy, dx)
```
