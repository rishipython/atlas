import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

class Shift2D:
    def solve(self, problem: Dict[str, Any]) -> np.ndarray:
        # extract data
        data = problem.get('data')
        if data is None:
            raise ValueError("Problem missing 'data'")
        arr = np.asarray(data)
        # get shift
        shift = problem.get('shift', (0., 0.))
        # ensure tuple of two floats
        if not isinstance(shift, (tuple,list)) or len(shift)!=2:
            raise ValueError("'shift' must be tuple/list of length 2")
        # get order
        order = problem.get('order', 1)
        # log
        logging.debug(f"Shift2D solving with shift {shift} on shape {arr.shape}")
        # apply shift
        shifted = scipy.ndimage.shift(arr, shift=shift, order=order,
                                      mode='nearest', prefilter=True)
        # cast back dtype if needed
        if np.issubdtype(arr.dtype, np.integer):
            shifted = shifted.astype(arr.dtype)
        return shifted

def run_solver(problem):
    return Shift2D().solve(problem)
