from scipy.signal import correlate2d

def solve(problem):
    src = np.array(problem['source'])
    tgt = np.array(problem['target'])
    corr = correlate2d(tgt, src, mode='full', boundary='fill', fillvalue=0)
    idx = np.unravel_index(np.argmax(corr), corr.shape)
    # compute shift relative to center
    center = (np.array(corr.shape) - 1) // 2
    shift = idx - center
    return {'dx': int(shift[1]), 'dy': int(shift[0])}
```

import logging
import random
from typing import Any, Dict
import numpy as np
from scipy.signal import correlate

def run_solver(problem):
    src = np.asarray(problem['source'], dtype=np.float64)
    tgt = np.asarray(problem['target'], dtype=np.float64)
    corr = correlate(tgt, src, mode='full', method='auto')
    idx_yx = np.unravel_index(np.argmax(corr), corr.shape)
    center_yx = ((np.array(corr.shape)-1)//2).astype(int)
    shift_yx = np.array(idx_yx) - center_yx
    # Convert back to dx (col), dy(row)
    return {'dx': int(shift_yx[1]), 'dy': int(shift_yx[0])}
```
