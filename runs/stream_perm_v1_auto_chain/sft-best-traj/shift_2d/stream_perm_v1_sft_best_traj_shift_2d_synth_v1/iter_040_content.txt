```
import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    src = np.asarray(problem['src'], dtype=np.float64)
    tgt = np.asarray(problem['tgt'], dtype=np.float64)
    # compute linear cross-correlation via FFT
    s_shape = src.shape
    t_shape = tgt.shape
    # pad sizes
    pad_shape = tuple(s + t - 1 for s,t in zip(s_shape,t_shape))
    src_fft = np.fft.rfftn(src, pad_shape)
    tgt_fft = np.fft.rfftn(tgt, pad_shape)
    corr = np.real(np.fft.irfftn(np.conj(src_fft) * tgt_fft, pad_shape))
    # find peak
    max_idx = np.unravel_index(np.argmax(corr), corr.shape)
    # compute offset
    offset_y = max_idx[0] - (t_shape[0]-1)
    offset_x = max_idx[1] - (t_shape[1]-1)
    # apply negative offset
    shifted_tgt = scipy.ndimage.shift(tgt, shift=(-offset_y,-offset_x), order=3, mode='constant', cval=0)
    return shifted_tgt
```
