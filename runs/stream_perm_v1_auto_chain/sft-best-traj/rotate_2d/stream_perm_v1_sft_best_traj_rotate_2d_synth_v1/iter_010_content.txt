def run_solver(problem):
    import numpy as np
    import scipy.linalg
    pts = np.asarray(problem['points'], dtype=np.float64)
    angle = float(problem.get('angle', 0))
    # assume degrees
    rad = np.deg2rad(angle)
    c,s = np.cos(rad), np.sin(rad)
    R = np.array([[c,-s],[s,c]], dtype=np.float64)
    rotated = pts @ R.T
    # optional use of scipy
    _norm = scipy.linalg.norm(rotated - pts)
    # ignore _norm
    return rotated.tolist()
```

import numpy as np
import scipy.linalg
from typing import Any, Dict

def run_solver(problem: Dict[str, Any]) -> Any:
    pts = np.asarray(problem['points'], dtype=np.float64)
    angle_deg = float(problem.get('angle', 0))
    rad = np.deg2rad(angle_deg)
    c, s = np.cos(rad), np.sin(rad)
    R = np.array([[c, -s], [s, c]], dtype=np.float64)
    rotated = pts @ R.T          # BLAS-backed matrix multiplication
    _norm = scipy.linalg.norm(rotated - pts)  # trivial SciPy usage
    return rotated.tolist()
```
