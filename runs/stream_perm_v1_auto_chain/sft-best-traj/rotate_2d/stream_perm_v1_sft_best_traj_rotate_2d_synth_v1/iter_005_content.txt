```
import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

class Rotate2D:
    def solve(self, problem: Dict[str, Any]) -> np.ndarray:
        data = problem['data']
        angle_deg = problem.get('angle', 0)
        if angle_deg % 360 == 0:
            return data.copy()
        # convert degrees to radians
        theta = np.deg2rad(angle_deg)
        cos_t = np.cos(theta)
        sin_t = np.sin(theta)
        # build rotation matrix
        R = np.array([[cos_t, -sin_t],
                      [sin_t, cos_t]], dtype=data.dtype)

        H, W = data.shape[:2]
        # compute centre offsets
        cx = (W - 1) / 2
        cy = (H - 1) / 2
        centre = np.array([cx, cy], dtype=data.dtype)

        # create grid of coordinates
        x_grid = np.arange(W)
        y_grid = np.arange(H)
        X, Y = np.meshgrid(x_grid, y_grid, indexing='xy')
        coords = np.vstack((X.ravel(), Y.ravel()))   # shape (2,N)

        # shift origin to centre
        shifted = coords - centre[:, None]
        # apply rotation via BLAS-backed dot
        rotated_shifted = R @ shifted   # this triggers BLAS

        # shift back
        rotated_coords = rotated_shifted + centre[:, None]

        # sample using map_coordinates
        flat_rotated = scipy.ndimage.map_coordinates(
            data,
            rotated_coords,
            order=1,
            mode='constant',
            cval=0,
            prefilter=False   # skip extra filtering for speed
        )
        return flat_rotated.reshape(H, W)
```
