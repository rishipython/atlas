from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    # extract image and angle
    img = problem.get("image") or problem.get("array")
    angle = problem.get("angle", 0)
    # ensure numpy array
    arr = np.asarray(img)
    # guard zero-size
    if arr.size == 0:
        return {"rotated": arr}
    # perform rotation
    rotated = scipy.ndimage.rotate(arr.astype(np.float64), angle,
                                   reshape=False,
                                   order=1,
                                   mode='constant',
                                   cval=0)
    # cast back to original dtype if integer
    if np.issubdtype(arr.dtype, np.integer):
        rotated = np.clip(rotated, np.iinfo(arr.dtype).min,
                          np.iinfo(arr.dtype).max)
        rotated = rotated.astype(arr.dtype)
    return {"rotated": rotated}
