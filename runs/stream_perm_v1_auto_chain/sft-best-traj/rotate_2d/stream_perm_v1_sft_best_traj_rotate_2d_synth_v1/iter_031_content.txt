import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

logger = logging.getLogger(__name__)

def _rotate_points(points: np.ndarray, angle_deg: float) -> np.ndarray:
    theta = np.radians(angle_deg)
    c = np.cos(theta)
    s = np.sin(theta)
    R = np.array([[c,-s],[s,c]], dtype=points.dtype)
    return points @ R.T

def _rotate_image(img: np.ndarray, angle_deg: float) -> np.ndarray:
    return scipy.ndimage.rotate(img.astype(float), angle=angle_deg,
                                reshape=False,
                                order=3,
                                mode='constant',
                                cval=0)

def solve(problem: Dict[str, Any]) -> Any:
    data = problem.get('data') or problem.get('matrix') or problem.get('points')
    angle = problem.get('angle',0)

    logger.debug(f"Rotating data type {type(data)} with angle {angle}")

    if isinstance(data,(np.ndarray,list)):
        arr = np.asarray(data,dtype=float)
        if arr.ndim == 0:
            return data
        elif arr.ndim == 1:
            # single point?
            return _rotate_points(arr.reshape(1,-1), angle)[0].tolist()
        elif arr.ndim == 2:
            if arr.shape[1] == 2:
                rotated = _rotate_points(arr.astype(float), angle)
                return rotated.tolist()
            else:
                rotated_img = _rotate_image(arr.astype(float), angle)
                return rotated_img.astype(arr.dtype)
        else:
            raise ValueError("Unsupported data dimensions")
    else:
        raise TypeError("Unsupported data type")

def run_solver(problem):
    return solve(problem)
