import numpy as np
import scipy.ndimage
from typing import Any, Dict

class Rotate2D:
    def solve(self, problem: Dict[str, Any]) -> np.ndarray:
        arr = problem['array']
        angle = problem.get('angle', 0)
        # normalize angle
        angle = angle % 360
        # perform rotation
        rotated = scipy.ndimage.rotate(arr, angle, reshape=False, order=1)
        # ensure dtype same
        if rotated.dtype != arr.dtype:
            rotated = rotated.astype(arr.dtype)
        return rotated

def run_solver(problem):
    return Rotate2D().solve(problem)

def solve(problem):

    ang_deg = float(problem.get('angle', 0))
