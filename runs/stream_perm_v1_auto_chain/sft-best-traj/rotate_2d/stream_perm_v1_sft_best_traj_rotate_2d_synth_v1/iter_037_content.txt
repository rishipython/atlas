import logging
import random
from typing import Any, Dict
import numpy as np
import scipy.ndimage

logger = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO)

def run_solver(problem: Dict[str, Any]) -> Any:
    """Rotate a 2D array or point set based on problem dict."""
    logger.debug("run_solver called")
    # extract data
    arr = problem.get("array") or problem.get("data") or problem.get("points")
    angle = problem.get("angle")
    if arr is None or angle is None:
        raise ValueError("Problem dict must contain 'array'/'data'/'points' and 'angle'")
    # ensure numpy array
    arr_np = np.asarray(arr)
    # If arr has shape (N,2) treat as point set
    if arr_np.ndim == 2 and arr_np.shape[1] == 2:
        # treat as points
        theta = np.deg2rad(angle) if isinstance(angle,(int,float)) else angle
        cos_t = np.cos(theta)
        sin_t = np.sin(theta)
        R = np.array([[cos_t,-sin_t],[sin_t,cos_t]], dtype=arr_np.dtype)
        rotated = arr_np @ R.T
        logger.debug("Rotated point set")
        return rotated.tolist()
    else:
        # treat as image / grid
        rotated_img = scipy.ndimage.rotate(arr_np.astype(np.float64), angle=angle,
                                           reshape=False,
                                           order=3,
                                           mode='constant',
                                           cval=0)
        logger.debug("Rotated image")
        return rotated_img
