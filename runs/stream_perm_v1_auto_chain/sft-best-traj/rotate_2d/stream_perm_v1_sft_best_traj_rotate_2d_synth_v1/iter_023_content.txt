def run_solver(problem):
    img = problem.get('image')
    angle = problem.get('angle')
    center = problem.get('center') # optional
    order = problem.get('order',1)
    mode = problem.get('mode','constant')
    cval = problem.get('cval',0)
    
    # Ensure numpy array
    img_arr = np.asarray(img)
    
    # If custom center:
    if center is not None:
        # Compute shift to move center to array centre
        shift = np.array(center)[::-1] - np.array(img_arr.shape)[::-1]/2
        # Apply shift before rotation
        img_arr = scipy.ndimage.shift(img_arr, shift=shift[::-1], order=order,
                                      mode=mode, cval=cval)
        # Rotate around new centre
        rotated = scipy.ndimage.rotate(img_arr, angle,
                                       reshape=False,
                                       order=order,
                                       mode=mode,
                                       cval=cval)
        # Shift back
        rotated = scipy.ndimage.shift(rotated,
                                      shift=-shift[::-1],
                                      order=order,
                                      mode=mode,
                                      cval=cval)
    else:
        rotated = scipy.ndimage.rotate(img_arr,
                                       angle,
                                       reshape=False,
                                       order=order,
                                       mode=mode,
                                       cval=cval)
    
    # Convert back dtype if needed
    if np.issubdtype(img_arr.dtype, np.integer):
        rotated = np.clip(rotated.round(), np.iinfo(img_arr.dtype).min,
                          np.iinfo(img_arr.dtype).max).astype(img_arr.dtype)
