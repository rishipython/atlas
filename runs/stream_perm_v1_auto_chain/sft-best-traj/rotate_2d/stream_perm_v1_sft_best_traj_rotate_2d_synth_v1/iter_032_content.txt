```
import logging
from typing import Any, Dict
import numpy as np
import scipy.ndimage

def run_solver(problem: Dict[str, Any]) -> np.ndarray:
    # extract
    arr = np.asarray(problem.get('array'))
    theta = float(problem.get('angle', 0))
    # validation
    if arr.ndim < 2:
        raise ValueError('Input must be at least 2-D')
    # handle empty
    if arr.size == 0:
        return arr.copy()
    # log
    logging.info(f"Rotating array of shape {arr.shape} by {theta} degrees")
    # perform rotation
    rotated = scipy.ndimage.rotate(arr,
                                   theta,
                                   reshape=False,
                                   order=1,
                                   mode='constant',
                                   cval=0)
    return rotated
```
