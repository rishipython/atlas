```
def run_solver(problem):
    # parse
    img = np.asarray(problem['image'])
    theta = float(problem.get('angle', 0))
    mode = problem.get('mode', 'bilinear')
    reshape = bool(problem.get('reshape', True))
    # map mode string to order
    order_map = {'nearest':0,'bilinear':1,'bicubic':3}
    order = order_map.get(mode.lower(),1)
    # early exit
    if theta % 360 == 0:
        return {'rotated': img}
    # convert to float32
    img_f32 = img.astype(np.float32)
    # define function with jit
    @jax.jit
    def rot(x):
        return jax_rotate(x, theta, reshape=reshape, order=order)
    rotated_f32 = rot(img_f32)
    # convert back
    rotated = np.array(rotated_f32)  # this casts back to numpy
    rotated = rotated.astype(img.dtype)
    return {'rotated': rotated}
```
