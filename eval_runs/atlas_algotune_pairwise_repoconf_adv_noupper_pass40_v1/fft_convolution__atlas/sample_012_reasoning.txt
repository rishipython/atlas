def conv1d(x,h):
    N = x.shape[0]; M=h.shape[0]
    L = N+M-1
    # compute next power of two? Not necessary but may speed. Use nextpow2(L).
    def nextpow2(n):
        return 1<<(n-1).bit_length()
    size = nextpow2(L)
    X = np.fft.rfft(x, n=size)
    H = np.fft.rfft(h, n=size)
    Y = X*H
    y = np.fft.irfft(Y, n=size)
    return y[:L]
