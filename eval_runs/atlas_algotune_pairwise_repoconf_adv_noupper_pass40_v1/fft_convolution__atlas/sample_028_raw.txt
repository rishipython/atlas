```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64, order="C")
    h = np.asarray(h, dtype=np.float64, order="C")

    N = x.shape[0]
    M = h.shape[0]
    out_len = N + M - 1

    # If one of the arrays is empty, return an empty array
    if out_len <= 0:
        return np.array([], dtype=np.float64)

    # Choose FFT size as next power of two for speed
    fft_len = 1 << (out_len - 1).bit_length()

    # Compute FFTs of padded inputs
    X = np.fft.rfft(x, fft_len)
    H = np.fft.rfft(h, fft_len)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result, truncate to required length
    y = np.fft.irfft(Y, fft_len)[:out_len]

    return y
```
