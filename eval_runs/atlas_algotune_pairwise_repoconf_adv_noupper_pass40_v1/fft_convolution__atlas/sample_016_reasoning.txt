def conv1d(x, h):
    N = x.shape[0]
    M = h.shape[0]
    n = N+M-1
    # compute FFT length: next power of two or just n
    size = 1 << (n-1).bit_length()
    X = np.fft.rfft(x, size)
    H = np.fft.rfft(h, size)
    Y = X * H
    y = np.fft.irfft(Y, size)[:n]
    return y
