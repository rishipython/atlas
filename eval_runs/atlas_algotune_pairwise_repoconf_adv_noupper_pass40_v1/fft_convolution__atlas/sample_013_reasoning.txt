def conv1d(x, h):
    N, M = x.shape[0], h.shape[0]
    conv_len = N + M - 1
    L = np.fft.next_fast_len(conv_len)
    X = np.fft.fft(x, L)
    H = np.fft.fft(h, L)
    y = np.fft.ifft(X * H).real
    return y[:conv_len]
