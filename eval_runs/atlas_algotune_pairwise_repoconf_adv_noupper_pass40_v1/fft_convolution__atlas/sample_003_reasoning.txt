```
import numpy as np

def conv1d(x,h):
    n=len(x); m=len(h)
    size=n+m-1
    # choose fft size: next power of two for speed
    fft_size = 1 << (size-1).bit_length()
    X = np.fft.rfft(x, fft_size)
    H = np.fft.rfft(h, fft_size)
    Y = X*H
    y = np.fft.irfft(Y, fft_size)
    return y[:size]
```
