import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` using FFT for speed."""
    # Ensure inputs are 1-D and convert to float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1  # length of the full convolution

    # Zero‑pad both sequences to length L and compute FFTs
    X = np.fft.fft(x, n=L)
    H = np.fft.fft(h, n=L)

    # Element‑wise multiplication in frequency domain and inverse FFT
    y = np.fft.ifft(X * H).real

    # Return as float64 array of length L
    return y.astype(np.float64)
