import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1-D float64 arrays using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64, order="C")
    h = np.asarray(h, dtype=np.float64, order="C")
    if x.ndim != 1 or h.ndim != 1:
        raise ValueError("Both input arrays must be one-dimensional")

    N, M = x.size, h.size
    L = N + M - 1  # length of the full convolution result

    # Zero‑pad both inputs to the same length L and compute the real FFT
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Element‑wise multiplication in the frequency domain
    Y = X * H

    # Inverse real FFT to obtain the convolution result
    y = np.fft.irfft(Y, n=L)

    return y
