def conv1d(x,h):
    N=len(x); M=len(h)
    L = N+M-1
    # rfft length L
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y
