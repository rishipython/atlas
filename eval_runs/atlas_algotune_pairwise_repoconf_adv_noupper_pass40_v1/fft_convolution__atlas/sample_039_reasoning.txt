```
def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    if N==0 or M==0:
        return np.array([], dtype=np.float64)
    L = 1 << ((N+M-1-1).bit_length())   # next power of two
    X = np.fft.fft(x, L)
    H = np.fft.fft(h, L)
    y = np.fft.ifft(X*H).real
    return y[:N+M-1]
```
