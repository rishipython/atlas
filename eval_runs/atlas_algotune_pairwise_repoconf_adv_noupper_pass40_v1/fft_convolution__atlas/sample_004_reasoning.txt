def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N, M = x.shape[0], h.shape[0]
    if N==0 or M==0:
        return np.empty(0, dtype=np.float64)
    L = N + M - 1
    # choose next power of 2 for speed? Not necessary but okay
    # compute FFT
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = np.fft.irfft(X * H, n=L)
    return Y
