import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of 1‑D arrays `x` and `h`."""
    # Ensure input arrays are float64
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N, M = x.size, h.size
    if N == 0 or M == 0:
        # np.convolve([] , a , 'full') returns an empty array
        return np.empty(0, dtype=np.float64)

    # Length of the full convolution
    L = N + M - 1

    # Next power of two for efficient FFT (optional but usually faster)
    nfft = 1 << (L - 1).bit_length()

    # Compute FFTs using real FFT for speed
    X = np.fft.rfft(x, n=nfft)
    H = np.fft.rfft(h, n=nfft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT and truncate to the exact convolution length
    y = np.fft.irfft(Y, n=nfft)[:L]

    return y
