    N=len(x); M=len(h); n_out=N+M-1
    L=1<<((n_out-1).bit_length())  # next power of two
    X=np.fft.rfft(x, L)
    H=np.fft.rfft(h, L)
    Y=X*H
    y=np.fft.irfft(Y, L)[:n_out]
    return y
