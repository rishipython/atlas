import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure input arrays are 1‑D float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    n = x.size
    m = h.size
    size = n + m - 1

    # Use a power‑of‑two FFT size for speed
    fft_size = 1 << (size - 1).bit_length()

    # Compute FFTs (real‑to‑complex) and multiply pointwise
    X = np.fft.rfft(x, fft_size)
    H = np.fft.rfft(h, fft_size)
    Y = X * H

    # Inverse FFT back to real domain
    y = np.fft.irfft(Y, fft_size)

    # Truncate to the exact convolution length
    return y[:size]
