def conv1d(x, h):
    N = x.shape[0]; M = h.shape[0]; L = N+M-1
    # compute next power of two? but not necessary.
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y
