def conv1d(x,h):

    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.size
    M = h.size
    if N==0 or M==0:
        return np.array([], dtype=np.float64)
    conv_len = N+M-1
    L = 1 << (conv_len-1).bit_length()  # next power of 2
    X = np.fft.fft(x, L)
    H = np.fft.fft(h, L)
    y = np.fft.ifft(X*H).real
    return y[:conv_len]
