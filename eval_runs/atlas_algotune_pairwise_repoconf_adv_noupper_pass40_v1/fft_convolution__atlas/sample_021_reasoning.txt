def conv1d(x, h):
    N, M = len(x), len(h)
    n = N+M-1
    size = 1 << (n-1).bit_length()
    X = np.fft.rfft(x, size)
    H = np.fft.rfft(h, size)
    Y = X * H
    y = np.fft.irfft(Y, size)
    return y[:n]
