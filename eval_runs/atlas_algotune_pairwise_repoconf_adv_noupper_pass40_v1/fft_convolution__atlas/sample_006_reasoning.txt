def conv1d(x,h):
    N = len(x); M=len(h)
    size = N+M-1
    nfft = 1 << (size-1).bit_length()
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    y = np.fft.irfft(X*H, nfft)[:size]
    return y
