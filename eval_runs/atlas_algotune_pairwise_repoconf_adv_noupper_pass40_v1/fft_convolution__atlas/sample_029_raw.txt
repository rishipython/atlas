import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays using FFT."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.shape[0]
    M = h.shape[0]
    n = N + M - 1

    # Handle trivial cases
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    # Use the next fast FFT length for efficiency
    nfft = np.fft.next_fast_len(n)

    # Compute the FFTs
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Pointwise multiplication in frequency domain
    Y = X * H

    # Inverse FFT and truncate to the full convolution length
    y = np.fft.irfft(Y, nfft)[:n]

    return y
