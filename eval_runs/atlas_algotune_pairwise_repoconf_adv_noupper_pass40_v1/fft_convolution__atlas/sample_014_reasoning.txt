def conv1d(x,h):
    N, M = x.shape[0], h.shape[0]
    L = N + M - 1
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    y = np.fft.irfft(X*H, n=L)
    return y
