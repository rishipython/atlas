import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure float64 for consistency with the reference implementation
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row (shape: (N,))
    sq_norm = np.sum(X * X, axis=1)

    # Compute Gram matrix (shape: (N, N))
    gram = X @ X.T

    # Apply the formula: ||x - y||^2 = ||x||^2 + ||y||^2 - 2 x·y
    dist = sq_norm[:, None] + sq_norm[None, :] - 2.0 * gram

    # Numerical errors might lead to tiny negative values; clip them to zero.
    np.maximum(dist, 0.0, out=dist)

    return dist
