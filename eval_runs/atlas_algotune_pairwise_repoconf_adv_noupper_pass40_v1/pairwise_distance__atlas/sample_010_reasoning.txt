def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X*X, axis=1)
    gram = X @ X.T
    D = sq_norms[:,None] + sq_norms[None,:] - 2*gram
    # numerical errors might produce tiny negative values; clip to zero
    np.maximum(D, 0, out=D)
    return D
