import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Uses a vectorized computation based on the identity
    ||x - y||^2 = ||x||^2 + ||y||^2 - 2 x·y.
    """
    # Compute squared norms of each row
    sq_norms = np.sum(X ** 2, axis=1, dtype=np.float64)
    # Compute Gram matrix X X^T
    gram = np.dot(X, X.T)
    # Broadcast to obtain pairwise squared distances
    dist_sq = sq_norms[:, None] + sq_norms[None, :] - 2.0 * gram
    # Numerical errors can make very small negative numbers; clamp to zero
    np.maximum(dist_sq, 0.0, out=dist_sq)
    return dist_sq
