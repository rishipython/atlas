import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure input is float64 for consistency with the reference
    X = np.asarray(X, dtype=np.float64)
    # Compute squared norms of each row
    sq_norms = np.sum(X * X, axis=1)
    # Compute pairwise squared distances using the identity:
    # ||x - y||^2 = ||x||^2 + ||y||^2 - 2 * x·y
    # Broadcasting adds the norms to each other, then subtract twice the dot product
    dist_sq = (
        sq_norms[:, None] + sq_norms[None, :] - 2.0 * X @ X.T
    )
    # Numerical errors might produce tiny negative values; clip to zero
    dist_sq = np.maximum(dist_sq, 0.0)
    return dist_sq
