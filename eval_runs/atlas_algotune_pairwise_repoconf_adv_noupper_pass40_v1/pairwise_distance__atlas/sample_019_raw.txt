import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D), dtype float64.

    Returns
    -------
    np.ndarray
        Array of shape (N, N) containing squared distances, dtype float64.
    """
    # Ensure input is float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row
    sq_norms = np.sum(X * X, axis=1)

    # Use broadcasting to compute pairwise squared distances:
    # D[i, j] = ||x_i||^2 + ||x_j||^2 - 2 * x_i · x_j
    # The dot product matrix is X @ X.T
    prod = X @ X.T
    # Outer sum of squared norms
    outer = np.add.outer(sq_norms, sq_norms)
    # Compute distances
    out = outer - 2.0 * prod

    # Numerical errors can cause tiny negative values; clip to zero
    out[out < 0] = 0.0
    return out
