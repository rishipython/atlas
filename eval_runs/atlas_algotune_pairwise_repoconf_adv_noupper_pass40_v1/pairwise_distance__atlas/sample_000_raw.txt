import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D) with dtype float64.

    Returns
    -------
    np.ndarray
        Array of shape (N, N) where element [i, j] is the squared Euclidean
        distance between X[i] and X[j].
    """
    # Compute squared norms of each point
    sq_norms = np.sum(X ** 2, axis=1, keepdims=True)  # shape (N, 1)
    # Use broadcasting to compute pairwise squared distances
    # D[i, j] = ||x_i||^2 + ||x_j||^2 - 2 * x_i dot x_j
    dist_sq = sq_norms + sq_norms.T - 2 * X @ X.T
    # Numerical errors may cause tiny negative values; clip to zero
    np.maximum(dist_sq, 0.0, out=dist_sq)
    return dist_sq
