import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure the array is float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row: shape (N,)
    sq_norms = np.sum(X * X, axis=1)

    # Use the identity: ||a - b||^2 = ||a||^2 + ||b||^2 - 2 a·b
    # Outer addition of squared norms gives ||a||^2 + ||b||^2
    # Subtract twice the dot product matrix
    dist = sq_norms[:, None] + sq_norms[None, :] - 2.0 * X @ X.T

    # Numerical errors can produce tiny negative values; clip them to zero
    return np.maximum(dist, 0.0)
