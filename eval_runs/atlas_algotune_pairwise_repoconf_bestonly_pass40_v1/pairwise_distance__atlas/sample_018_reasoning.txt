def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1, keepdims=True)
    dist2 = sq_norms + sq_norms.T - 2 * X @ X.T
    # due to numerical errors, small negative values may appear; clip to zero
    np.maximum(dist2, 0, out=dist2)
    return dist2
