def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sum_sq = np.sum(X**2, axis=1)
    # compute outer sum
    out = sum_sq[:, None] + sum_sq[None, :] - 2 * X @ X.T
    # due to floating point errors, small negative values may appear; set them to zero
    out = np.maximum(out, 0.0)
    return out
