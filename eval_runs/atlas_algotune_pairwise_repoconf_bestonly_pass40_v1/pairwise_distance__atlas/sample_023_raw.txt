import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Array of shape (N, D) with dtype float64.

    Returns
    -------
    np.ndarray
        Array of shape (N, N) with dtype float64 where
        out[i, j] = ||X[i] - X[j]||_2^2.
    """
    # Compute squared norms of each row
    norms = np.sum(X * X, axis=1, keepdims=True)  # shape (N, 1)

    # Compute the Gram matrix (dot products)
    gram = X @ X.T  # shape (N, N)

    # Use the identity: ||a - b||^2 = ||a||^2 + ||b||^2 - 2 a·b
    dist = norms + norms.T - 2.0 * gram

    # Numerical errors can lead to tiny negative values; clip them to zero
    np.maximum(dist, 0.0, out=dist)
    return dist
