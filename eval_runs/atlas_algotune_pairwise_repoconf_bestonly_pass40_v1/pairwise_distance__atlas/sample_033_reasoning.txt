So implement:

def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    G = X @ X.T
    sq_norm = np.diagonal(G)
    D = sq_norm[:,None] + sq_norm[None,:] - 2*G
    D = np.maximum(D, 0.0)
    return D
