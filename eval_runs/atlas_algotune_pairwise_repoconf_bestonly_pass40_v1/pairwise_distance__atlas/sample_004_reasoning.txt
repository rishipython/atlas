def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norm = np.sum(X**2, axis=1)
    D = sq_norm[:,None] + sq_norm[None,:] - 2 * X @ X.T
    # numerical errors: set small negatives to zero
    D = np.maximum(D, 0.0)
    return D
