def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    dist = sq_norms[:, None] + sq_norms[None, :] - 2 * X.dot(X.T)
    # due to floating errors, small negative numbers may appear, set to zero
    dist = np.maximum(dist, 0.0)
    return dist
