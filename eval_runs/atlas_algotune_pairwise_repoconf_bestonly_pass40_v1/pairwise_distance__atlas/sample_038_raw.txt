import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    This implementation uses vectorized operations for speed:
    D[i, j] = ||X[i] - X[j]||^2 = ||X[i]||^2 + ||X[j]||^2 - 2 * X[i]·X[j].

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure dtype is float64 for consistency with the reference
    X = X.astype(np.float64, copy=False)

    # Compute squared norms of each row
    sq_norms = np.sum(X * X, axis=1)  # shape (N,)

    # Compute the Gram matrix (dot products)
    gram = X @ X.T  # shape (N, N)

    # Use broadcasting to form the distance matrix
    D = sq_norms[:, None] + sq_norms[None, :] - 2.0 * gram

    # Numerical errors may introduce tiny negative values; clip them to zero
    np.maximum(D, 0.0, out=D)

    return D
