def pairwise_sq_dist(X):
    # check shape, dtype
    X = np.asarray(X, dtype=np.float64)
    # compute
    sq = np.sum(X**2, axis=1)
    D = sq[:,None] + sq[None,:] - 2*np.dot(X, X.T)
    # ensure numeric stability
    D = np.maximum(D, 0.0)
    return D
