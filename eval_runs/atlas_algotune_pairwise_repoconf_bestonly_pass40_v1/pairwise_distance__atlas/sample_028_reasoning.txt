def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sum_X = np.sum(X**2, axis=1)
    D = sum_X[:, None] + sum_X[None, :] - 2 * X @ X.T
    # numerical errors might produce negative small values
    D = np.maximum(D, 0.0)
    return D
