def conv1d(x, h):
    N = x.shape[0]
    M = h.shape[0]
    conv_len = N + M - 1
    # choose fft length as next power of 2? We can just use conv_len.
    # But to be safe for speed, we can use next power of two.
    fftlen = 1 << (conv_len - 1).bit_length()
    X = np.fft.rfft(x, n=fftlen)
    H = np.fft.rfft(h, n=fftlen)
    y = np.fft.irfft(X * H, n=fftlen)
    return y[:conv_len]
