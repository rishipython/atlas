def conv1d(x, h):
    N, M = x.shape[0], h.shape[0]
    n = N+M-1
    if n==0:
        return np.array([], dtype=np.float64)
    nfft = np.fft.next_fast_len(n)
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    Y = X * H
    y = np.fft.irfft(Y, nfft)
    return y[:n]
