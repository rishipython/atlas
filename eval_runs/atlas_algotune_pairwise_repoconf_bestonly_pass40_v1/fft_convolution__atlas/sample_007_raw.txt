import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays.

    Equivalent to ``np.convolve(x, h, mode='full')`` but computed using
    the FFT for speed. The result is numerically close to the reference
    implementation (within the tolerances used by the test harness).
    """
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64, order='C')
    h = np.asarray(h, dtype=np.float64, order='C')

    N = x.shape[0]
    M = h.shape[0]

    # Handle empty inputs
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the full convolution
    out_len = N + M - 1

    # Next power of two for efficient FFT
    L = 1 << (out_len - 1).bit_length()

    # Compute FFTs of zero‑padded inputs
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get time‑domain convolution
    y = np.fft.irfft(Y, n=L)

    # Truncate to the exact output length
    return y[:out_len]
