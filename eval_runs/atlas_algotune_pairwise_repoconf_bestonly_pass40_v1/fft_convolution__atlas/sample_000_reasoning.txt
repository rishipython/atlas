def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.size
    M = h.size
    if N==0 or M==0:
        return np.array([], dtype=np.float64)
    L = 1 << (N+M-1-1).bit_length()
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = np.fft.irfft(X * H, n=L)
    return Y[:N+M-1]
