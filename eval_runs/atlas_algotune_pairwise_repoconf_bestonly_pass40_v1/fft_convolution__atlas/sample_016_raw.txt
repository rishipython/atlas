```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return np.convolve(x, h, mode='full') but MUCH faster using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    if x.size == 0 or h.size == 0:
        # Empty input yields empty output
        return np.array([], dtype=np.float64)

    N, M = x.shape[0], h.shape[0]
    L = N + M - 1  # desired output length

    # Use next power of two for efficient FFT
    L_fft = 1 << (L - 1).bit_length()

    # Compute forward FFTs (real-to-complex)
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get convolution result
    y = np.fft.irfft(Y, n=L_fft)

    # Truncate to the exact output length
    return y[:L]
```
