def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N, M = x.shape[0], h.shape[0]
    L = N + M - 1
    L_fft = 1 << (L-1).bit_length()  # next power of two
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)
    y = np.fft.irfft(X * H, n=L_fft)
    return y[:L]
