import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure input arrays are 1‑D float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size
    # Handle empty inputs
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    out_len = N + M - 1

    # Choose FFT length as next power of two for speed
    L = 1 << (out_len - 1).bit_length()

    # Compute FFTs using real FFT for efficiency
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Pointwise multiplication and inverse real FFT
    Y = np.fft.irfft(X * H, n=L)

    # Truncate to the correct output length
    return Y[:out_len]
