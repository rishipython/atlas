import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT."""
    # Ensure inputs are float64 arrays
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1

    # For very small inputs, fall back to the straightforward implementation
    if L <= 64:
        out = np.zeros(L, dtype=np.float64)
        for n in range(L):
            kmin = max(0, n - M + 1)
            kmax = min(N - 1, n)
            s = 0.0
            for k in range(kmin, kmax + 1):
                s += x[k] * h[n - k]
            out[n] = s
        return out

    # Use real FFT for speed and numerical stability
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)

    # Ensure the output is of type float64
    return y.astype(np.float64)
