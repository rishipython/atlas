import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1‑D float64 arrays using FFT.
    This is equivalent to np.convolve(x, h, mode='full') but much faster for
    large inputs.
    """
    # Ensure the inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1  # length of full convolution

    # Choose FFT size as next power of two for efficiency
    nfft = 1 << (L - 1).bit_length()

    # Compute FFTs of the zero‑padded signals
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Element‑wise multiplication in frequency domain and inverse FFT
    y = np.fft.irfft(X * H, nfft)

    # Truncate to the exact convolution length
    return y[:L]
