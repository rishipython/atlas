def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    n = x.shape[0]
    m = h.shape[0]
    size = n + m - 1
    # next power of two
    Nfft = 1 << (size-1).bit_length()
    X = np.fft.rfft(x, Nfft)
    H = np.fft.rfft(h, Nfft)
    y = np.fft.irfft(X * H, Nfft)
    return y[:size]
