```
def conv1d(x,h):
    N = x.shape[0]
    M = h.shape[0]
    L = N+M-1
    # compute next power of two >= L for speed
    nfft = 1 << (L-1).bit_length()
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    y = np.fft.irfft(X*H, nfft)
    return y[:L]
```
