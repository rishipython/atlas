import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64, order='C')
    h = np.asarray(h, dtype=np.float64, order='C')
    if x.ndim != 1 or h.ndim != 1:
        raise ValueError("Inputs must be 1-D arrays.")
    
    N = x.size
    M = h.size
    out_len = N + M - 1

    if out_len == 0:
        return np.empty(0, dtype=np.float64)

    # Use real FFT for speed: zero‑pad to the required output length
    X = np.fft.rfft(x, n=out_len)
    H = np.fft.rfft(h, n=out_len)
    y = np.fft.irfft(X * H, n=out_len)
    # Ensure dtype is float64 (irfft returns float64)
    return y.astype(np.float64, copy=False)
