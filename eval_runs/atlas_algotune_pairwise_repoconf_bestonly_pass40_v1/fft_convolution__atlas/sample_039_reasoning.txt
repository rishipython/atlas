def conv1d(x,h):
    N = x.size
    M = h.size
    L = N+M-1
    nfft = 1 << (L-1).bit_length()  # next power of two >= L
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    Y = X * H
    y = np.fft.irfft(Y, nfft)[:L]
    return y
