def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1
    # compute next power of two
    L2 = 1 << (L - 1).bit_length()
    # FFT
    X = np.fft.fft(x, L2)
    H = np.fft.fft(h, L2)
    y = np.fft.ifft(X * H).real[:L]
    return y
