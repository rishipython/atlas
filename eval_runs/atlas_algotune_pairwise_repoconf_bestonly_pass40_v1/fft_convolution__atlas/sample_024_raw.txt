import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of 1‑D float64 arrays using FFT."""
    # Ensure input is 1‑D float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    out_len = N + M - 1

    # Choose FFT length: next power of two for speed
    nfft = 1 << (out_len - 1).bit_length()

    # Compute FFTs (real‑to‑complex)
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Point‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT back to time domain (real‑to‑real)
    y_full = np.fft.irfft(Y, nfft)

    # Return only the full convolution length
    return y_full[:out_len]
