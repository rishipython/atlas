def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    n = x.size + h.size - 1
    size = 1 << (n-1).bit_length()  # next power of two >= n
    # compute FFT
    X = np.fft.rfft(x, size)
    H = np.fft.rfft(h, size)
    Y = X * H
    y = np.fft.irfft(Y, size)[:n]
    return y
