def conv1d(x, h):

    N = x.size; M = h.size; L = N+M-1

    nfft = 1 << (L-1).bit_length()  # next power of two >= L

    X = np.fft.rfft(x, n=nfft)

    H = np.fft.rfft(h, n=nfft)

    y = np.fft.irfft(X * H, n=nfft)

    return y[:L]  # slice to full length
