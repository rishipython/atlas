def conv1d(x,h):

    x = np.asarray(x, dtype=np.float64)

    h = np.asarray(h, dtype=np.float64)

    N = x.shape[0]; M = h.shape[0]; L = N+M-1

    n = 1 << (L-1).bit_length()  # next power of two

    X = np.fft.fft(x, n)

    H = np.fft.fft(h, n)

    Y = np.fft.ifft(X*H).real

    return Y[:L]
