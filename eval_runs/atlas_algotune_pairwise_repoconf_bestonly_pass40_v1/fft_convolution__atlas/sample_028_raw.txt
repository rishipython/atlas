import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64, copy=False)
    h = np.asarray(h, dtype=np.float64, copy=False)
    N, M = x.shape[0], h.shape[0]
    L = N + M - 1  # length of the full convolution

    # Perform FFT‑based convolution
    X = np.fft.fft(x, n=L)
    H = np.fft.fft(h, n=L)
    y = np.fft.ifft(X * H)

    # Return the real part (imaginary part should be negligible)
    return np.real(y)
