```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.size
    M = h.size
    L = N + M - 1  # length of the full convolution

    if N == 0 or M == 0:
        # One of the inputs is empty: result is an empty array of length L
        return np.zeros(L, dtype=np.float64)

    # Choose an FFT size that is a power of two and at least L
    nfft = 1 << (L - 1).bit_length()

    # Compute forward FFTs using real FFT for efficiency
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT and truncate to the desired length
    y = np.fft.irfft(Y, nfft)[:L]

    return y
```
