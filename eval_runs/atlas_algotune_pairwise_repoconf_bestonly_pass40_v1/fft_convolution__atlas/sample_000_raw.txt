import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()
    N, M = x.size, h.size

    # Handle empty inputs
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the linear convolution
    out_len = N + M - 1

    # Next power of two for efficient FFT
    fft_len = 1 << (out_len - 1).bit_length()

    # Compute FFTs of zero‑padded inputs
    X = np.fft.rfft(x, n=fft_len)
    H = np.fft.rfft(h, n=fft_len)

    # Element‑wise multiplication in frequency domain
    Y = np.fft.irfft(X * H, n=fft_len)

    # Truncate to the linear convolution length
    return Y[:out_len]
