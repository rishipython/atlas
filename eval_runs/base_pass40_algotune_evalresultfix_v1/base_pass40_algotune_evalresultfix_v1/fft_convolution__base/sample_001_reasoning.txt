def fft_convolve(x,y,mode):
    len_x,len_y=len(x),len(y)
    if len_x==0 or len_y==0: return np.array([],dtype=float)
    n = len_x+len_y-1
    size = 1 << (n-1).bit_length()
    X = np.fft.rfft(x, size)
    Y = np.fft.rfft(y, size)
    Z = X*Y
    conv = np.fft.irfft(Z, size)[:n]
    if mode=="full": return conv
    elif mode=="same":
        start = (n - max(len_x,len_y))//2
        return conv[start:start+max(len_x,len_y)]
    elif mode=="valid":
        k = max(len_x,len_y) - min(len_x,len_y) + 1
        if k<=0: return np.array([],dtype=float)
        start = max(len_x,len_y) - min(len_x,len_y)
        return conv[start:start+k]
    else: raise ValueError
