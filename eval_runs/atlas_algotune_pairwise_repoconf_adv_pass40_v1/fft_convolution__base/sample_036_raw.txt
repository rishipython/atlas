import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays using FFT."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    if N == 0 or M == 0:
        # According to NumPy semantics, convolving with an empty array
        # yields an empty array of length N+M-1 (which is 0 if either is 0).
        return np.empty(N + M - 1, dtype=np.float64)

    L = N + M - 1  # length of the full convolution result

    # Compute FFTs of the two sequences padded to length L
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain the convolution result
    y = np.fft.irfft(Y, n=L)

    return y
