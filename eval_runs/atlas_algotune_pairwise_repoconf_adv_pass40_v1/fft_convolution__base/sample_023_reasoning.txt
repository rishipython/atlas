def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    outlen = N + M - 1
    if outlen == 0:
        return np.array([], dtype=np.float64)
    L = np.fft.next_fast_len(outlen)
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)[:outlen]
    return y
