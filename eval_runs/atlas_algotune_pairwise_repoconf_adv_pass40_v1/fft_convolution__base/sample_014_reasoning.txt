def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    L = x.size + h.size - 1
    # If small, use direct
    if L <= 64:
        out = np.zeros(L, dtype=np.float64)
        for n in range(L):
            kmin = max(0, n - h.size + 1)
            kmax = min(x.size - 1, n)
            out[n] = np.sum(x[kmin:kmax+1] * h[n - np.arange(kmin, kmax+1)])
        return out
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    y = np.fft.irfft(X * H, n=L)
    return y
