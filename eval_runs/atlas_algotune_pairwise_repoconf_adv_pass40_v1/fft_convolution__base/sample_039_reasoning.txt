import numpy as np

def conv1d(x, h):

    N = x.shape[0]; M = h.shape[0]; L = N+M-1

    nfft = 1 << (L-1).bit_length()   # next power of two

    X = np.fft.rfft(x, nfft)

    H = np.fft.rfft(h, nfft)

    Y = np.fft.irfft(X*H, nfft)

    return Y[:L]
