```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT."""
    # Ensure input is 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    out_len = N + M - 1

    # Next power of two for efficient FFT (optional but usually faster)
    L = 1 << (out_len - 1).bit_length()

    # Compute FFTs
    X = np.fft.fft(x, n=L)
    H = np.fft.fft(h, n=L)

    # Elementwise multiplication and inverse FFT
    y = np.fft.ifft(X * H)

    # Truncate to the correct length and take real part
    y = np.real(y[:out_len])

    return y
```
