```
def conv1d(x, h):
    N = x.shape[0]
    M = h.shape[0]
    L = 1 << (N + M - 1 - 1).bit_length()
    X = np.fft.fft(x, L)
    H = np.fft.fft(h, L)
    Y = X * H
    y = np.fft.ifft(Y).real
    return y[:N+M-1]
```
