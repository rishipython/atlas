def conv1d(x, h):
    N, M = x.shape[0], h.shape[0]
    n = N + M - 1
    # next power of 2 for speed
    nfft = 1 << (n-1).bit_length()
    X = np.fft.fft(x, nfft)
    H = np.fft.fft(h, nfft)
    Y = np.fft.ifft(X * H)
    return np.real(Y[:n])
