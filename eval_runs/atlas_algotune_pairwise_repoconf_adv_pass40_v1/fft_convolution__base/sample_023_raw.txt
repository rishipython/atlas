import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.shape[0]
    M = h.shape[0]
    out_len = N + M - 1

    if out_len == 0:
        return np.array([], dtype=np.float64)

    # Choose an FFT length that is fast and at least out_len
    L = np.fft.next_fast_len(out_len)

    # Compute FFTs of the zero‑padded inputs
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Element‑wise product in frequency domain
    Y = X * H

    # Inverse FFT and truncate to the true convolution length
    y = np.fft.irfft(Y, n=L)[:out_len]

    return y
