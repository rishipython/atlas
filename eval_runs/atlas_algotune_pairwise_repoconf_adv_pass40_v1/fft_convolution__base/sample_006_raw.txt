```python
import numpy as np
from scipy.signal import fftconvolve

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1‑D float64 arrays using FFT.
    This is equivalent to np.convolve(x, h, mode='full') but much faster for
    large inputs.

    Parameters
    ----------
    x : np.ndarray
        First input array, 1‑D.
    h : np.ndarray
        Second input array, 1‑D.

    Returns
    -------
    y : np.ndarray
        The full convolution result, dtype float64.
    """
    # Ensure inputs are 1‑D and cast to float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    # Use FFT‑based convolution from scipy.signal
    return fftconvolve(x, h, mode='full')
```
