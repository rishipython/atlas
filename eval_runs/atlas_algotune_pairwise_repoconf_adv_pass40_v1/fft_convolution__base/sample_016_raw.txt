import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT.
    Parameters
    ----------
    x : np.ndarray
        1‑D array of floats, length N.
    h : np.ndarray
        1‑D array of floats, length M.
    Returns
    -------
    y : np.ndarray
        Full linear convolution of x and h, length N + M - 1.
    """
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    # Handle empty inputs: convolution is empty
    if x.size == 0 or h.size == 0:
        return np.empty(0, dtype=np.float64)

    N = x.shape[0]
    M = h.shape[0]
    out_len = N + M - 1

    # Next power of two for efficient FFT
    n_fft = 1 << (out_len - 1).bit_length()

    # FFT of both signals padded to n_fft
    X = np.fft.rfft(x, n=n_fft)
    H = np.fft.rfft(h, n=n_fft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result
    y = np.fft.irfft(Y, n=n_fft)

    # Trim to the exact output length
    return y[:out_len]
