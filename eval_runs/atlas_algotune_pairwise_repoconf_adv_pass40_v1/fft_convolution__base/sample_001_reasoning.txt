def conv1d(x, h):
    N = x.shape[0]
    M = h.shape[0]
    L = N+M-1
    # Use next power of two? Not necessary but faster for some lengths. Use np.fft.fft with n=L
    # Actually np.fft.fft uses length L directly. Good.
    X = np.fft.fft(x, n=L)
    H = np.fft.fft(h, n=L)
    Y = X*H
    y = np.fft.ifft(Y)
    return np.real(y)
