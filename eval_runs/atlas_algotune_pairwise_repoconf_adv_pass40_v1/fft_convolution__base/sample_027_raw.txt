import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays
    using an FFT‑based method for speed.
    """
    # Ensure input arrays are 1‑D and of type float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    L = N + M - 1  # length of the full convolution

    # Zero‑pad inputs to the same length L and compute FFTs
    X = np.fft.fft(x, n=L)
    H = np.fft.fft(h, n=L)

    # Element‑wise product in frequency domain and inverse FFT
    y = np.fft.ifft(X * H).real

    # Return the real part with the correct length
    return y.astype(np.float64)
