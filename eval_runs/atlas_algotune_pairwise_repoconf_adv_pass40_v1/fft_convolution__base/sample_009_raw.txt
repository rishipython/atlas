import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return np.convolve(x, h, mode='full') but faster using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    L = N + M - 1

    if L == 0:
        return np.array([], dtype=np.float64)

    # Use next power of two for efficient FFT
    nfft = 1 << (L - 1).bit_length()

    # Compute FFTs of zero-padded inputs
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Pointwise multiply and inverse FFT
    y = np.fft.irfft(X * H, nfft)

    # Truncate to the exact convolution length
    return y[:L]
