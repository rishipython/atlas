def conv1d(x, h):
    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1
    # Next power of two for speed? Use np.fft.next_fast_len
    L_fft = np.fft.next_fast_len(L)
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)
    Y = X * H
    y = np.fft.irfft(Y, n=L_fft)
    return y[:L]
