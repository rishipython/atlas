def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.size; M = h.size
    if N==0 or M==0:
        return np.array([], dtype=np.float64)
    out_len = N+M-1
    L = 1 << (out_len-1).bit_length()
    X = np.fft.fft(x, L)
    H = np.fft.fft(h, L)
    Y = X*H
    y = np.fft.ifft(Y).real
    return y[:out_len]
