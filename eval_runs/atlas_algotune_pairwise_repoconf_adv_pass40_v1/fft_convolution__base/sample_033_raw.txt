import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays using FFT."""
    # Ensure input arrays are 1‑D float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size
    L = N + M - 1  # length of the full convolution

    # Choose FFT length: next power of two >= L for speed
    fft_len = 1 << (L - 1).bit_length()

    # Compute FFTs of zero‑padded inputs
    X = np.fft.rfft(x, fft_len)
    H = np.fft.rfft(h, fft_len)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT and truncate to the true convolution length
    y = np.fft.irfft(Y, fft_len)[:L]

    return y.astype(np.float64)
