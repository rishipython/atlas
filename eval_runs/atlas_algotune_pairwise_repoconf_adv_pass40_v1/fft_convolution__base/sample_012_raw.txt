```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64, copy=False)
    h = np.asarray(h, dtype=np.float64, copy=False)

    # Handle empty inputs: np.convolve returns an empty array
    if x.size == 0 or h.size == 0:
        return np.array([], dtype=np.float64)

    # Length of the full convolution
    n = x.size + h.size - 1

    # Next power of two for efficient FFT (optional, but often faster)
    nfft = 1 << (n - 1).bit_length()

    # Compute FFTs, multiply, and inverse FFT
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    y = np.fft.irfft(X * H, nfft)[:n]

    return y
```
