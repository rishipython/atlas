import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D arrays using FFT."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64, order='C')
    h = np.asarray(h, dtype=np.float64, order='C')
    N = x.shape[0]
    M = h.shape[0]
    out_len = N + M - 1

    # Choose FFT length as next power of two >= out_len
    L_fft = 1 << (out_len - 1).bit_length()

    # Compute FFTs using real FFT for efficiency
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result
    y_full = np.fft.irfft(Y, n=L_fft)

    # Truncate to the exact output length
    return y_full[:out_len]
