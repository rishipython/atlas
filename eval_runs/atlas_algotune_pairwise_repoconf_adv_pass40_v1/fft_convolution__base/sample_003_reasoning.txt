def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N=len(x); M=len(h)
    L=N+M-1
    n_fft=1
    while n_fft < L:
        n_fft <<=1
    # compute FFTs
    X = np.fft.rfft(x, n_fft)
    H = np.fft.rfft(h, n_fft)
    Y = X*H
    y = np.fft.irfft(Y, n_fft)
    return y[:L]
