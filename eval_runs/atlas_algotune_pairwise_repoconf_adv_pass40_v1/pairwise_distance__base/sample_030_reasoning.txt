def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X * X, axis=1)
    G = X @ X.T
    D = sq_norms[:, None] + sq_norms[None, :] - 2 * G
    # ensure symmetry and non-negative
    # D = np.maximum(D, 0.0)
    return D
