import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure input is a NumPy array with float64 dtype
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row vector
    sq_norms = np.sum(X * X, axis=1)  # shape (N,)

    # Compute the pairwise squared Euclidean distances
    # Using the identity: ||x - y||^2 = ||x||^2 + ||y||^2 - 2 x·y
    distances = np.add.outer(sq_norms, sq_norms) - 2.0 * X.dot(X.T)

    # Numerical errors can produce very small negative values; clip to zero
    np.maximum(distances, 0.0, out=distances)

    return distances
