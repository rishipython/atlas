```python
import numpy as np
import scipy.linalg as la

def expm_fast(A: np.ndarray) -> np.ndarray:
    """
    Return the matrix exponential of a square float64 matrix ``A``.
    
    This implementation delegates to scipy.linalg.expm, which uses
    a highly optimized scaling-and-squaring algorithm with Padé
    approximants for numerical stability and speed.
    
    Parameters
    ----------
    A : np.ndarray
        Square matrix of shape (n, n) with dtype float64 (or castable to it).
    
    Returns
    -------
    np.ndarray
        The matrix exponential of ``A``.
    """
    A = np.asarray(A, dtype=np.float64, copy=False)
    return la.expm(A)
```
