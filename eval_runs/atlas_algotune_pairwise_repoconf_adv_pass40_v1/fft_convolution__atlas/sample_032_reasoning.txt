def conv1d(x, h):
    N, M = len(x), len(h)
    size = N + M - 1
    fft_size = 1 << (size-1).bit_length()
    X = np.fft.rfft(x, fft_size)
    H = np.fft.rfft(h, fft_size)
    Y = X * H
    y = np.fft.irfft(Y, fft_size)[:size]
    return y
