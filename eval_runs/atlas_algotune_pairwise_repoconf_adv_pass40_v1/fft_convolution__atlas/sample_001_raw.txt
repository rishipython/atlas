import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 arrays.
    This implementation uses an FFT-based approach for speed.
    """
    # Ensure inputs are 1-D numpy arrays of float64
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.shape[0]
    M = h.shape[0]
    nfft = N + M - 1

    # Compute the convolution using the FFT (real FFT for speed)
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    Y = X * H
    y = np.fft.irfft(Y, nfft)

    return y
