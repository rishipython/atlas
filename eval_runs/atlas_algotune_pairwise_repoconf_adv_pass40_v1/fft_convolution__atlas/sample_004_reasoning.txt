def conv1d(x, h):
    N, M = x.shape[0], h.shape[0]
    L = N + M - 1
    # next pow of 2 maybe
    L_fft = 1 << (L-1).bit_length()
    X = np.fft.fft(x, n=L_fft)
    H = np.fft.fft(h, n=L_fft)
    Y = X * H
    y = np.fft.ifft(Y).real
    return y[:L]
