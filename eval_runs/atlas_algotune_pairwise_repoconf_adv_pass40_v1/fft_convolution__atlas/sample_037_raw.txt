```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays using FFT.

    Parameters
    ----------
    x : np.ndarray
        Input array of length N.
    h : np.ndarray
        Input array of length M.

    Returns
    -------
    y : np.ndarray
        Convolution of x and h, length N + M - 1.
    """
    # Convert inputs to float64 for consistency
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.shape[0]
    M = h.shape[0]
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the output
    L_out = N + M - 1

    # Next power of two for efficient FFT
    L_fft = 1 << (L_out - 1).bit_length()

    # FFT of both inputs padded to L_fft
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)

    # Elementwise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result
    y = np.fft.irfft(Y, n=L_fft)

    # Truncate to the exact output length
    return y[:L_out].copy()
```
