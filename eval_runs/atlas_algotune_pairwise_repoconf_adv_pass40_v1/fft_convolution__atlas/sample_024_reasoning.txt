def conv1d(x, h):
    N = len(x); M = len(h)
    size = N + M - 1
    fft_size = 1 << (size-1).bit_length()  # next power of two
    X = np.fft.rfft(x, fft_size)
    H = np.fft.rfft(h, fft_size)
    y = np.fft.irfft(X*H, fft_size)[:size]
    return y
