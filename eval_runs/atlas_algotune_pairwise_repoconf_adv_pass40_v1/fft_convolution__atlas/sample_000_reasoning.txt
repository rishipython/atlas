import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    # ensure float64
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)
    L = np.fft.next_fast_len(N + M - 1)
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    y = np.fft.irfft(X * H, n=L)
    return y[:N + M - 1]
