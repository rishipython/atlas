import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays
    using FFT for speed.  The result is equivalent to
    ``np.convolve(x, h, mode='full')`` within numerical tolerances.
    """
    # Ensure the inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    # Desired output length
    out_len = N + M - 1

    # FFT length: next power of two >= out_len for speed
    fft_len = 1 << (out_len - 1).bit_length()

    # Compute FFTs (real‑to‑complex for efficiency)
    X = np.fft.rfft(x, fft_len)
    H = np.fft.rfft(h, fft_len)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get convolution, truncate to exact length
    y = np.fft.irfft(Y, fft_len)[:out_len]

    return y
