def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N, M = x.shape[0], h.shape[0]
    L = 1 << (N + M - 1 - 1).bit_length()  # next power of 2
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y[:N+M-1]
