import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays using FFT.

    The result is equivalent to ``np.convolve(x, h, mode='full')``.
    """
    # Ensure input is 1‑D and of type float64
    x = np.asarray(x, dtype=np.float64, order='C')
    h = np.asarray(h, dtype=np.float64, order='C')
    N, M = x.shape[0], h.shape[0]
    size = N + M - 1

    # Next power of two for efficient FFT (optional but speeds up large arrays)
    nfft = 1 << (size - 1).bit_length()

    # Compute FFTs
    fft_x = np.fft.rfft(x, n=nfft)
    fft_h = np.fft.rfft(h, n=nfft)

    # Element‑wise multiplication in frequency domain
    fft_prod = fft_x * fft_h

    # Inverse FFT to obtain convolution result
    y_full = np.fft.irfft(fft_prod, n=nfft)

    # Truncate to the true convolution length
    return y_full[:size]
