```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 numpy arrays
    using an FFT-based method for speed.
    """
    # Ensure inputs are 1-D arrays of type float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size

    # The length of the linear convolution result
    L = N + M - 1

    # Pad inputs to length L and compute FFTs
    X = np.fft.fft(x, n=L)
    H = np.fft.fft(h, n=L)

    # Element-wise multiplication in frequency domain and inverse FFT
    y = np.fft.ifft(X * H)

    # Due to numerical errors the result may have a tiny imaginary part;
    # take the real part and cast to float64
    return np.real(y).astype(np.float64)
```
