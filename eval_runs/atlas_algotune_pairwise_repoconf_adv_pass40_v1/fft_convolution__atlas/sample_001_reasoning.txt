def conv1d(x,h):
    N=x.shape[0]; M=h.shape[0]
    nfft = N+M-1
    # choose rfft
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    Y = X*H
    y = np.fft.irfft(Y, nfft)
    return y
