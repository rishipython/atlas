def conv1d(x,h):
    N=len(x); M=len(h); L = N+M-1; L_fft = 1<< (L-1).bit_length()  # next power of 2
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)
    Y = X*H
    y = np.fft.irfft(Y, n=L_fft)
    return y[:L].copy()
