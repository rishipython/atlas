```
def conv1d(x,h):
    N, M = x.shape[0], h.shape[0]
    if N==0 or M==0:
        return np.array([], dtype=np.float64)
    L = 1 << ((N+M-1-1).bit_length())
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y[:N+M-1]
```
