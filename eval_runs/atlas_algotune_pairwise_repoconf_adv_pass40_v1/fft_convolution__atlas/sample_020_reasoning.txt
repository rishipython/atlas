```
import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1
    # next power of two
    K = 1 << (L - 1).bit_length()
    # FFT
    X = np.fft.fft(x, K)
    H = np.fft.fft(h, K)
    Y = X * H
    y = np.fft.ifft(Y).real
    return y[:L]
```
