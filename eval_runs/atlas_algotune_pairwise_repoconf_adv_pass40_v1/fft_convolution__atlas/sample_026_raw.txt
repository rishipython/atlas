```python
import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return np.convolve(x, h, mode='full') but MUCH faster using FFT."""
    # Ensure inputs are 1-D float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size

    # Handle empty inputs: numpy returns an empty array
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    L = N + M - 1  # length of the full convolution

    # Use real FFT for efficiency
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    out = np.fft.irfft(Y, n=L)

    # Numerical errors can leave tiny imaginary parts; take real part
    return out.astype(np.float64)
```
