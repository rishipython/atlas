import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return np.convolve(x, h, mode='full') but MUCH faster using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64, order='C')
    h = np.asarray(h, dtype=np.float64, order='C')

    N = x.shape[0]
    M = h.shape[0]
    if N == 0 or M == 0:
        # Convolution with an empty array yields an empty array
        return np.empty(0, dtype=np.float64)

    # Length of the full convolution result
    L = N + M - 1

    # Use the next fast length for FFT for efficiency
    nfft = np.fft.next_fast_len(L)

    # Compute FFTs of zero-padded signals
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result, truncate to length L
    y_full = np.fft.irfft(Y, nfft, dtype=np.float64)[:L]

    return y_full
