import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure input is a float64 array
    X = np.asarray(X, dtype=np.float64)
    # Compute squared norms of each row
    sq_norm = np.sum(X ** 2, axis=1)
    # Use the identity: ||x - y||^2 = ||x||^2 + ||y||^2 - 2 x·y
    # X @ X.T gives the Gram matrix (dot products)
    gram = np.dot(X, X.T)
    # Broadcast to form the distance matrix
    D = sq_norm[:, None] + sq_norm[None, :] - 2.0 * gram
    # Numerical errors might produce tiny negative values; clamp to zero
    D[D < 0] = 0.0
    return D
