import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Uses the identity:
        ||x_i - x_j||^2 = ||x_i||^2 + ||x_j||^2 - 2 * x_i·x_j
    which can be computed efficiently with matrix operations.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure the input is float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row (vector)
    norms = np.sum(X * X, axis=1, keepdims=True)  # shape (N, 1)

    # Compute the pairwise dot product matrix
    dot_prod = X @ X.T  # shape (N, N)

    # Apply the squared distance formula
    dist = norms + norms.T - 2.0 * dot_prod

    # Numerical errors may produce tiny negative values; clip them to zero
    np.maximum(dist, 0.0, out=dist)

    return dist
