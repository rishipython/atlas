import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D) with dtype float64.

    Returns
    -------
    out : np.ndarray
        Array of shape (N, N) containing squared distances, dtype float64.
    """
    # Ensure input is float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row: (N,)
    sq_norms = np.sum(X * X, axis=1)

    # Compute the Gram matrix: (N, N)
    gram = X @ X.T

    # Apply the squared distance formula: ||x_i||^2 + ||x_j||^2 - 2 * x_i·x_j
    out = sq_norms[:, None] + sq_norms[None, :] - 2.0 * gram

    # Numerical errors can produce tiny negative values; clip them to zero
    out = np.maximum(out, 0.0)

    return out
