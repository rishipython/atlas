import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    X = np.asarray(X, dtype=np.float64)
    # compute dot product
    sq_norms = np.sum(X**2, axis=1)
    D = sq_norms[:, None] + sq_norms[None, :] - 2 * X @ X.T
    # due to floating errors, make small negatives zero
    D[D < 0] = 0.0
    return D
