import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D), dtype float64.

    Returns
    -------
    np.ndarray
        Distance matrix of shape (N, N), dtype float64.
    """
    X = np.asarray(X, dtype=np.float64)
    # Compute squared norms of each row
    sq_norm = np.sum(X * X, axis=1)
    # Compute Gram matrix
    gram = X @ X.T
    # Use broadcasting to compute pairwise squared distances
    D = sq_norm[:, None] + sq_norm[None, :] - 2 * gram
    # Numerical errors may introduce tiny negative values
    np.maximum(D, 0.0, out=D)
    return D
