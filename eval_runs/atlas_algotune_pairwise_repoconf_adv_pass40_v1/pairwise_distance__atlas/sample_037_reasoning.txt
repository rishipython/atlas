import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    X = np.asarray(X, dtype=np.float64)
    norms = np.sum(X * X, axis=1)
    D = norms[:, None] + norms[None, :] - 2 * X @ X.T
    # numerical errors might give small negative values
    np.maximum(D, 0, out=D)
    return D
