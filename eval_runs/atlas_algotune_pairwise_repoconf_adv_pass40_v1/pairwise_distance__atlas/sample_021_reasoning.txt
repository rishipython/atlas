def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X * X, axis=1)
    dist_sq = sq_norms[:, None] + sq_norms[None, :] - 2 * X.dot(X.T)
    # Ensure non-negative
    dist_sq = np.maximum(dist_sq, 0.0)
    return dist_sq
