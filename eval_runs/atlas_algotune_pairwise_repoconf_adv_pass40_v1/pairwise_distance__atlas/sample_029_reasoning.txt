def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    norms = np.sum(X**2, axis=1)
    dist = norms[:,None] + norms[None,:] - 2*np.dot(X, X.T)
    # Numerical errors
    dist[dist < 0] = 0.0
    return dist
