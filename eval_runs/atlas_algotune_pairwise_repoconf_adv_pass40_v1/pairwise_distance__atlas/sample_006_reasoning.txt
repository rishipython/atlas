def pairwise_sq_dist(X):
    sq_norm = np.sum(X**2, axis=1)
    dist = sq_norm[:, None] + sq_norm[None, :] - 2 * X.dot(X.T)
    # Due to floating errors, may be slightly negative; set to zero
    dist = np.maximum(dist, 0.0)
    return dist
