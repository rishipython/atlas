import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D) with dtype float64.

    Returns
    -------
    np.ndarray
        Distance matrix of shape (N, N) with dtype float64.
    """
    # Ensure input is a 2-D array of float64
    X = np.asarray(X, dtype=np.float64)
    N, D = X.shape

    # Compute squared norms of each row
    norms = np.sum(X * X, axis=1)  # shape (N,)

    # Compute pairwise squared distances using the identity:
    # ||x - y||^2 = ||x||^2 + ||y||^2 - 2 * x·y
    # X @ X.T gives the Gram matrix of dot products
    gram = X @ X.T  # shape (N, N)

    # Broadcast norms to form the outer sum and subtract twice the Gram matrix
    dist2 = norms[:, None] + norms[None, :] - 2.0 * gram

    # Numerical errors can lead to tiny negative values; clip to zero
    np.maximum(dist2, 0.0, out=dist2)

    return dist2
