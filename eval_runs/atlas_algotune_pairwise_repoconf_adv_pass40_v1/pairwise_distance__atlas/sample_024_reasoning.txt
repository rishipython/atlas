def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    norms = np.sum(X**2, axis=1).reshape(-1,1)
    prod = X @ X.T
    out = norms + norms.T - 2*prod
    # numeric errors may produce small negative values
    out[out<0] = 0
    return out
