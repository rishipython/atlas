def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    norms = np.sum(X**2, axis=1, keepdims=True)
    G = X @ X.T
    dist = norms + norms.T - 2*G
    # due to numerical errors, small negative -> zero
    dist = np.maximum(dist, 0.0)
    return dist
