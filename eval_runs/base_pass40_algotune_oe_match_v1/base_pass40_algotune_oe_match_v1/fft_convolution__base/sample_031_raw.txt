import numpy as np

class FFTConvolution:
    """
    Fast FFT-based convolution solver for real signals.
    """

    def __init__(self):
        pass

    def solve(self, problem):
        """
        Compute the convolution of two real-valued signals using FFT.

        Parameters
        ----------
        problem : dict
            Dictionary containing:
                - "signal_x": list of floats
                - "signal_y": list of floats
                - "mode": "full", "same" or "valid" (default "full")

        Returns
        -------
        dict
            Dictionary with key "convolution" containing a list of floats
            representing the convolution result.
        """
        # Convert inputs to NumPy arrays of type float64
        x = np.asarray(problem.get("signal_x", []), dtype=np.float64)
        y = np.asarray(problem.get("signal_y", []), dtype=np.float64)
        mode = problem.get("mode", "full")

        # Handle empty inputs early
        if x.size == 0 or y.size == 0:
            return {"convolution": []}

        # Full convolution length
        n_full = x.size + y.size - 1

        # Perform real FFTs with zero-padding to the full length
        X = np.fft.rfft(x, n=n_full)
        Y = np.fft.rfft(y, n=n_full)
        Z = X * Y
        conv_full = np.fft.irfft(Z, n=n_full)

        # Choose slice according to mode
        if mode == "full":
            result = conv_full

        elif mode == "same":
            # Length of the central part equals the larger input length
            out_len = max(x.size, y.size)
            start = (n_full - out_len) // 2
            result = conv_full[start : start + out_len]

        elif mode == "valid":
            # Length of overlap
            out_len = max(x.size, y.size) - min(x.size, y.size) + 1
            if out_len <= 0:
                result = np.empty(0, dtype=np.float64)
            else:
                start = min(x.size, y.size) - 1
                result = conv_full[start : start + out_len]

        else:
            raise ValueError(f"Unsupported mode: {mode!r}")

        # Convert to list for the expected output format
        return {"convolution": result.tolist()}

def run_solver(problem):
    """
    Entry point for the evaluator.

    Parameters
    ----------
    problem : dict
        Problem definition for the FFT convolution task.

    Returns
    -------
    dict
        Solution dictionary with key "convolution".
    """
    solver = FFTConvolution()
    return solver.solve(problem)
