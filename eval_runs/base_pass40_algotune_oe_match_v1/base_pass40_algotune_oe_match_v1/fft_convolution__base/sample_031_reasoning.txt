def solve(problem):
    x = np.array(problem["signal_x"], dtype=np.float64)
    y = np.array(problem["signal_y"], dtype=np.float64)
    mode = problem.get("mode","full")
    if x.size==0 or y.size==0:
        return {"convolution": []}
    n = len(x)+len(y)-1
    # compute next power of two for speed? np.fft.fft2 automatically uses optimal.
    # Use rfft: length n
    # But rfft expects real input, which is fine.
    X = np.fft.rfft(x, n=n)
    Y = np.fft.rfft(y, n=n)
    Z = X*Y
    conv_full = np.fft.irfft(Z, n=n)
    # conv_full is float64
    if mode=="full":
        result = conv_full
    elif mode=="same":
        # same length as max(len_x,len_y)
        # compute start index
        if len_x>=len_y:
            start = (len_y-1)//2
        else:
            start = (len_x-1)//2
        L = max(len_x,len_y)
        result = conv_full[start:start+L]
    elif mode=="valid":
        if len_x>=len_y:
            L = len_x - len_y + 1
            start = len_y-1
        else:
            L = len_y - len_x + 1
            start = len_x-1
        if L<=0:
            result = np.array([],dtype=np.float64)
        else:
            result = conv_full[start:start+L]
    else:
        raise ValueError("Invalid mode")
    return {"convolution": result.tolist()}
