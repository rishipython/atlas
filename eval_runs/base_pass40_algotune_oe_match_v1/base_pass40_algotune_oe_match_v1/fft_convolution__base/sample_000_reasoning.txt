```
def compute_full(x,y):
    if len(x)==0 or len(y)==0: return np.array([],dtype=float)
    size = len(x)+len(y)-1
    n = 1<< (size-1).bit_length()
    X = np.fft.rfft(x, n)
    Y = np.fft.rfft(y, n)
    Z = np.fft.irfft(X*Y, n)[:size]
    return Z
```
