```python
import logging
import numpy as np
from typing import Any, Dict

# --------------------------------------------------------------------------- #
# FFT Convolution solver – optimized implementation using real FFT
# --------------------------------------------------------------------------- #

class FFTConvolution:
    """
    Optimized FFT convolution solver.
    """

    def __init__(self):
        """No initialization required."""
        pass

    @staticmethod
    def _fast_fft_convolve(signal_x: np.ndarray, signal_y: np.ndarray, mode: str) -> np.ndarray:
        """
        Compute the linear convolution of two real signals using FFT.
        The result is sliced according to the requested mode.

        Parameters
        ----------
        signal_x : np.ndarray
            First input signal.
        signal_y : np.ndarray
            Second input signal.
        mode : str
            One of 'full', 'same', 'valid'.

        Returns
        -------
        np.ndarray
            The convolution result sliced according to mode.
        """
        len_x, len_y = signal_x.shape[0], signal_y.shape[0]

        # Handle empty signals
        if len_x == 0 or len_y == 0:
            return np.empty(0, dtype=float)

        # Full convolution length
        len_full = len_x + len_y - 1

        # For very small signals, direct convolution is cheaper
        if len_x * len_y < 64:
            full = np.convolve(signal_x, signal_y)
        else:
            # Next power of two for zero‑padding
            N = 1 << (len_full - 1).bit_length()
            # Real FFT of both signals
            X = np.fft.rfft(signal_x, n=N)
            Y = np.fft.rfft(signal_y, n=N)
            # Pointwise multiplication in frequency domain
            Z = X * Y
            # Inverse real FFT and truncate to true linear length
            full = np.fft.irfft(Z, n=N)[:len_full]

        # Slice according to mode
        if mode == "full":
            return full
        elif mode == "same":
            out_len = max(len_x, len_y)
            start = (len_full - out_len) // 2
            return full[start:start + out_len]
        elif mode == "valid":
            out_len = max(0, max(len_x, len_y) - min(len_x, len_y) + 1)
            if out_len == 0:
                return np.empty(0, dtype=float)
            start = len_y - 1 if len_x >= len_y else len_x - 1
            return full[start:start + out_len]
        else:
            raise ValueError(f"Unsupported mode '{mode}'. Expected 'full', 'same', or 'valid'.")

    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Solve the FFT convolution problem.

        Parameters
        ----------
        problem : dict
            Dictionary containing 'signal_x', 'signal_y', and optional 'mode'.

        Returns
        -------
        dict
            Dictionary with key 'convolution' containing the result as a list.
        """
        try:
            # Extract data
            signal_x = np.array(problem.get("signal_x", []), dtype=float)
            signal_y = np.array(problem.get("signal_y", []), dtype=float)
            mode = problem.get("mode", "full")

            # Compute convolution
            conv = self._fast_fft_convolve(signal_x, signal_y, mode)

            # Return as list
            return {"convolution": conv.tolist()}

        except Exception as exc:
            logging.error(f"Error in solve: {exc}")
            raise

    @staticmethod
    def is_solution(problem: Dict[str, Any], solution: Dict[str, Any]) -> bool:
        """
        Validate the solution using the reference implementation.
        """
        try:
            if "convolution" not in solution:
                logging.error("Solution missing 'convolution' key.")
                return False

            student_result = solution["convolution"]

            if not isinstance(student_result, list):
                logging.error("Convolution result must be a list.")
                return False

            student_np = np.array(student_result, dtype=float)
            if not np.all(np.isfinite(student_np)):
                logging.error("Convolution result contains non-finite values.")
                return False

            # Reference using scipy for validation
            import scipy.signal

            x = np.array(problem["signal_x"])
            y = np.array(problem["signal_y"])
            mode = problem.get("mode", "full")
            reference = scipy.signal.fftconvolve(x, y, mode=mode)

            if not np.allclose(student_np, reference, rtol=1e-6, atol=1e-6):
                diff = np.abs(student_np - reference)
                logging.error(
                    f"Numerical mismatch: max diff {np.max(diff):.2e}, avg diff {np.mean(diff):.2e}"
                )
                return False

            return True

        except Exception as exc:
            logging.error(f"Error in is_solution: {exc}")
            return False


def run_solver(problem: Dict[str, Any]) -> Dict[str, Any]:
    """
    Entry point for the evaluator.

    Parameters
    ----------
    problem : dict
        Problem definition.

    Returns
    -------
    dict
        Solution dictionary containing the convolution result.
    """
    solver = FFTConvolution()
    return solver.solve(problem)
```
