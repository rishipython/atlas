def fft_convolve(x, y, mode):

    if len(x)==0 or len(y)==0: return np.array([], dtype=float)

    n = len(x)+len(y)-1
    fft_len = 1 << (n-1).bit_length()
    X = np.fft.rfft(x, fft_len)
    Y = np.fft.rfft(y, fft_len)
    Z = X*Y
    z_full = np.fft.irfft(Z, fft_len)[:n]

    if mode=='full': return z_full
    elif mode=='same':
        L = max(len(x), len(y))
        start = (n - L)//2
        return z_full[start:start+L]
    elif mode=='valid':
        if len(x)>=len(y):
            L = len(x)-len(y)+1
        else:
            L = len(y)-len(x)+1
        start = n - L
        return z_full[start:]
    else: raise ValueError
