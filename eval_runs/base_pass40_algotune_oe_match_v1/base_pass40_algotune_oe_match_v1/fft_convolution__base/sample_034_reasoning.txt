def solve(problem):
   x = np.array(problem["signal_x"], dtype=float)
   y = np.array(problem["signal_y"], dtype=float)
   mode = problem.get("mode","full")
   if x.size==0 or y.size==0:
       return {"convolution":[]}
   if x.size*y.size <= 256:
       full = np.convolve(x,y)
   else:
       n = x.size + y.size -1
       m = 1 << (n-1).bit_length()
       # rfft
       fx = np.fft.rfft(x, n=m)
       fy = np.fft.rfft(y, n=m)
       f = fx*fy
       conv = np.fft.irfft(f, n=m)[:n]
       full = conv
   # slice
   if mode=="full":
        res=full
   elif mode=="same":
        if x.size>=y.size:
            start = (y.size-1)//2
            end = start + x.size
        else:
            start = (x.size-1)//2
            end = start + y.size
        res=full[start:end]
   elif mode=="valid":
        if x.size>=y.size:
            start = y.size-1
            length = x.size - y.size +1
        else:
            start = x.size-1
            length = y.size - x.size +1
        res=full[start:start+length]
   else:
        raise ValueError
   return {"convolution": res.tolist()}
