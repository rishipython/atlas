import numpy as np
from scipy import signal
import logging

class FFTConvolution:
    """
    Optimised FFT convolution implementation.
    """

    def __init__(self):
        pass

    @staticmethod
    def _next_fast_len(n: int) -> int:
        """
        Find the next fast length for FFT. Uses scipy's implementation if available,
        otherwise falls back to the next power of two.
        """
        try:
            from scipy.fft import next_fast_len
            return next_fast_len(n)
        except Exception:
            # fallback: next power of two
            return 1 << (n - 1).bit_length()

    def solve(self, problem):
        """
        Compute convolution using FFT with minimal padding.
        Handles 'full', 'same', and 'valid' modes.
        """
        try:
            # Convert inputs to numpy arrays
            x = np.asarray(problem["signal_x"], dtype=np.float64)
            y = np.asarray(problem["signal_y"], dtype=np.float64)
            mode = problem.get("mode", "full")

            # Handle empty inputs
            if x.size == 0 or y.size == 0:
                return {"convolution": []}

            len_x, len_y = x.size, y.size

            # Full convolution length
            L_full = len_x + len_y - 1
            # Determine FFT length (next fast length for efficiency)
            nfft = self._next_fast_len(L_full)

            # Use real FFT for real inputs
            X = np.fft.rfft(x, n=nfft)
            Y = np.fft.rfft(y, n=nfft)
            conv_full = np.fft.irfft(X * Y, n=nfft)[:L_full]

            # Slice according to mode
            if mode == "full":
                result = conv_full
            elif mode == "same":
                # Output length is max(len_x, len_y)
                out_len = max(len_x, len_y)
                start = (L_full - out_len) // 2
                result = conv_full[start:start + out_len]
            elif mode == "valid":
                if len_x >= len_y:
                    out_len = len_x - len_y + 1
                    start = len_y - 1
                else:
                    out_len = len_y - len_x + 1
                    start = len_x - 1
                result = conv_full[start:start + out_len]
            else:
                raise ValueError(f"Unsupported mode: {mode}")

            # Return as list to match expected format
            return {"convolution": result.tolist()}

        except Exception as e:
            logging.error(f"Error in solve method: {e}")
            raise

    def is_solution(self, problem, solution):
        """
        Validate the convolution result against scipy's reference.
        """
        try:
            if "convolution" not in solution:
                logging.error("Solution missing 'convolution' key.")
                return False

            student_result = solution["convolution"]

            if not isinstance(student_result, list):
                logging.error("Convolution result must be a list.")
                return False

            try:
                student_arr = np.array(student_result, dtype=float)
                if not np.all(np.isfinite(student_arr)):
                    logging.error("Convolution result contains non-finite values.")
                    return False
            except ValueError:
                logging.error("Could not convert convolution result to numeric array.")
                return False

            x = np.asarray(problem["signal_x"], dtype=float)
            y = np.asarray(problem["signal_y"], dtype=float)
            mode = problem.get("mode", "full")

            # Expected length calculation
            len_x, len_y = x.size, y.size
            if mode == "full":
                expected_len = len_x + len_y - 1
            elif mode == "same":
                expected_len = max(len_x, len_y)
            elif mode == "valid":
                if len_x >= len_y:
                    expected_len = len_x - len_y + 1
                else:
                    expected_len = len_y - len_x + 1
            else:
                logging.error(f"Invalid mode: {mode}")
                return False

            if len_x == 0 or len_y == 0:
                expected_len = 0

            if student_arr.size != expected_len:
                logging.error(f"Incorrect result length for mode '{mode}'. "
                              f"Expected {expected_len}, got {student_arr.size}.")
                return False

            # Reference using scipy
            try:
                ref = signal.fftconvolve(x, y, mode=mode)
            except Exception as e:
                logging.error(f"Reference convolution failed: {e}")
                return False

            if not np.allclose(student_arr, ref, rtol=1e-6, atol=1e-6):
                diff = np.abs(student_arr - ref)
                logging.error(f"Numerical mismatch: max diff {diff.max():.2e}, "
                              f"mean diff {diff.mean():.2e}")
                return False

            return True

        except Exception as e:
            logging.error(f"Error in is_solution: {e}")
            return False

def run_solver(problem):
    """
    Entry point used by the evaluator.
    """
    solver = FFTConvolution()
    return solver.solve(problem)
