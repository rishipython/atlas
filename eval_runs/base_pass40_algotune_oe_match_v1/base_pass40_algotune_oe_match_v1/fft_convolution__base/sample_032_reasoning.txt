def solve(self, problem):
    x = np.asarray(problem["signal_x"], dtype=float)
    y = np.asarray(problem["signal_y"], dtype=float)
    mode = problem.get("mode","full")
    if x.size==0 or y.size==0:
        return {"convolution": []}
    # small threshold
    if x.size + y.size < 500:
        res = np.convolve(x,y)
    else:
        n = x.size + y.size -1
        N = 1 << (n-1).bit_length()
        X = np.fft.rfft(x, n=N)
        Y = np.fft.rfft(y, n=N)
        Z = X*Y
        full = np.fft.irfft(Z, n=N)
        full = full[:n]
        res = full
    # slice based on mode
    if mode=="full":
        out = res
    elif mode=="same":
        len_out = max(x.size, y.size)
        start = (res.size - len_out)//2
        out = res[start:start+len_out]
    elif mode=="valid":
        len_out = max(0, max(x.size, y.size) - min(x.size, y.size) +1)
        if len_out==0:
            out = np.array([])
        else:
            start = (res.size - len_out)//2
            out = res[start:start+len_out]
    else:
        raise ValueError
    return {"convolution": out.tolist()}
