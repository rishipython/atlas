import numpy as np
from typing import Dict, Any, List

# ------------------------------------------------------------
# FFT Convolution implementation – fast, pure NumPy version
# ------------------------------------------------------------

class FFTConvolution:
    """
    A fast implementation of linear convolution using the FFT.
    Handles all three standard modes: 'full', 'same', 'valid'.
    Works for real-valued input signals.
    """

    @staticmethod
    def _next_pow_two(n: int) -> int:
        """Return the next power of two greater than or equal to n."""
        return 1 << (n - 1).bit_length()

    def solve(self, problem: Dict[str, Any]) -> Dict[str, List[float]]:
        """
        Compute the convolution of two signals using FFT.

        Parameters
        ----------
        problem : dict
            Dictionary containing:
                - 'signal_x': list[float]
                - 'signal_y': list[float]
                - 'mode': str, one of 'full', 'same', 'valid'

        Returns
        -------
        dict
            {'convolution': [float, ...]}
        """
        try:
            # Convert inputs to NumPy arrays (float64)
            x = np.asarray(problem.get("signal_x", []), dtype=np.float64)
            y = np.asarray(problem.get("signal_y", []), dtype=np.float64)
            mode = problem.get("mode", "full").lower()

            # Handle empty signals early
            if x.size == 0 or y.size == 0:
                return {"convolution": []}

            # Length of the full linear convolution
            n = x.size + y.size - 1

            # Use the smallest power‑of‑two length >= n for FFT efficiency
            fft_len = self._next_pow_two(n)

            # Compute real‑valued FFTs (half‑complex representation)
            X = np.fft.rfft(x, n=fft_len)
            Y = np.fft.rfft(y, n=fft_len)

            # Element‑wise multiplication in frequency domain
            Z = X * Y

            # Inverse FFT back to time domain
            conv_full = np.fft.irfft(Z, n=fft_len)[:n]  # truncate to linear size

            # Slice according to requested mode
            if mode == "full":
                result = conv_full
            elif mode == "same":
                # centre part of length max(len(x), len(y))
                len_out = max(x.size, y.size)
                start = (n - len_out) // 2
                result = conv_full[start:start + len_out]
            elif mode == "valid":
                len_out = max(0, min(x.size, y.size) - abs(x.size - y.size) + 1)
                if len_out <= 0:
                    result = np.array([], dtype=np.float64)
                else:
                    # For valid, the overlap length is min(x.size, y.size)
                    start = n - len_out
                    result = conv_full[start:]
            else:
                raise ValueError(f"Unsupported mode: {mode!r}")

            # Convert to Python list for JSON serialisation
            return {"convolution": result.tolist()}

        except Exception as exc:
            # In the unlikely event of an error, re‑raise for visibility
            raise RuntimeError(f"FFTConvolution.solve failed: {exc}") from exc


# ------------------------------------------------------------
# Entry point for the evaluator
# ------------------------------------------------------------
def run_solver(problem: Dict[str, Any]) -> Dict[str, List[float]]:
    """
    Main function used by the evaluation harness.

    Parameters
    ----------
    problem : dict
        Problem definition as described in the task.

    Returns
    -------
    dict
        Solution dictionary with key 'convolution'.
    """
    solver = FFTConvolution()
    return solver.solve(problem)
