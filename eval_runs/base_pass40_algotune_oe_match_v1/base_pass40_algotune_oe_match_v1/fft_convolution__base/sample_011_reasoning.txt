def fast_fft_convolve(x, y, mode):

    if len(x)==0 or len(y)==0: return np.array([], dtype=float)

    len_full = len(x)+len(y)-1

    if len(x)*len(y) < 64:  # small
        full = np.convolve(x, y)
    else:
        N = 1 << (len_full-1).bit_length()
        # rfft length N
        X = np.fft.rfft(x, n=N)
        Y = np.fft.rfft(y, n=N)
        Z = X * Y
        full = np.fft.irfft(Z, n=N)[:len_full]
    # now slice based on mode
    if mode=="full":
        return full
    elif mode=="same":
        out_len = max(len(x), len(y))
        start = (len_full - out_len)//2
        return full[start:start+out_len]
    elif mode=="valid":
        out_len = max(0, max(len(x), len(y)) - min(len(x), len(y)) + 1)
        if out_len==0:
            return np.array([], dtype=float)
        if len(x)>=len(y):
            start = len(y)-1
        else:
            start = len(x)-1
        return full[start:start+out_len]
    else:
        raise ValueError("Invalid mode")
