import torch
import triton
import triton.language as tl


@triton.jit
def _matmul_kernel(
    A_ptr, B_ptr, C_ptr,
    M: tl.constexpr, N: tl.constexpr, K: tl.constexpr,
    stride_a: tl.constexpr, stride_b: tl.constexpr, stride_c: tl.constexpr,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr
):
    """
    Triton kernel for matrix multiplication C = A @ B
    A: (M, K), B: (K, N), C: (M, N)
    """
    # 2‑D grid: (row tile, col tile)
    row_tile = tl.program_id(0) * BLOCK_M
    col_tile = tl.program_id(1) * BLOCK_N

    # Indices within the tile
    rows = tl.arange(0, BLOCK_M) + row_tile
    cols = tl.arange(0, BLOCK_N) + col_tile

    # Masks for the boundaries of M and N
    mask_rows = rows < M
    mask_cols = cols < N
    mask = mask_rows[:, None] & mask_cols[None, :]

    # Accumulator
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    # Loop over K in blocks
    for k in range(0, K, BLOCK_K):
        k_indices = tl.arange(0, BLOCK_K) + k
        mask_k = k_indices < K

        # Load tile from A: shape (BLOCK_M, BLOCK_K)
        A_tile = tl.load(
            A_ptr + rows[:, None] * stride_a + k_indices[None, :],
            mask=mask_rows[:, None] & mask_k[None, :],
            other=0.0,
        )

        # Load tile from B: shape (BLOCK_K, BLOCK_N)
        B_tile = tl.load(
            B_ptr + k_indices[:, None] * stride_b + cols[None, :],
            mask=mask_k[:, None] & mask_cols[None, :],
            other=0.0,
        )

        # Accumulate
        acc += tl.dot(A_tile, B_tile)

    # Store the result
    tl.store(
        C_ptr + rows[:, None] * stride_c + cols[None, :],
        acc,
        mask=mask,
    )


def optimized_matmul(A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
    """
    Compute C = A @ B using a custom Triton kernel.

    Args:
        A (torch.Tensor): (M, K) matrix of dtype float32.
        B (torch.Tensor): (K, N) matrix of dtype float32.

    Returns:
        torch.Tensor: (M, N) matrix of dtype float32.
    """
    assert A.dtype == torch.float32 and B.dtype == torch.float32
    M, K = A.shape
    K2, N = B.shape
    assert K == K2, "Inner dimensions must agree."

    C = torch.empty((M, N), dtype=A.dtype, device=A.device)

    # Block sizes (tuned for typical GPUs)
    BLOCK_M = 128
    BLOCK_N = 128
    BLOCK_K = 128

    # Strides for contiguous tensors
    stride_a = A.stride(0)
    stride_b = B.stride(0)
    stride_c = C.stride(0)

    # Launch grid
    grid = (
        triton.cdiv(M, BLOCK_M),
        triton.cdiv(N, BLOCK_N),
    )

    _matmul_kernel[grid](
        A_ptr=A,
        B_ptr=B,
        C_ptr=C,
        M=M,
        N=N,
        K=K,
        stride_a=stride_a,
        stride_b=stride_b,
        stride_c=stride_c,
        BLOCK_M=BLOCK_M,
        BLOCK_N=BLOCK_N,
        BLOCK_K=BLOCK_K,
    )

    return C


# Example usage and sanity check
if __name__ == "__main__":
    torch.manual_seed(0)
    for shape in [{'M': 512, 'K': 512, 'N': 512}, {'M': 1024, 'K': 1024, 'N': 1024}]:
        M, K, N = shape['M'], shape['K'], shape['N']
        A = torch.randn(M, K, dtype=torch.float32, device="cuda")
        B = torch.randn(K, N, dtype=torch.float32, device="cuda")
        C_ref = torch.matmul(A, B)
        C_opt = optimized_matmul(A, B)
        print(
            f"Shape {shape}:",
            torch.allclose(C_ref, C_opt, atol=0.1, rtol=0.1),
        )
