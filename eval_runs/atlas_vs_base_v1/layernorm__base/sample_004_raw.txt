import torch
import triton
import triton.language as tl

@triton.jit
def _layernorm_kernel(
    x_ptr, w_ptr, b_ptr, y_ptr,
    M, N,
    stride_x0, stride_x1,
    stride_w, stride_b,
    stride_y0, stride_y1,
    eps,
    BLOCK_N: tl.constexpr
):
    row = tl.program_id(0)

    if row >= M:
        return

    # pointers to the start of the row
    x_row_ptr = x_ptr + row * stride_x0
    y_row_ptr = y_ptr + row * stride_y0

    # ---------- compute mean and variance ----------
    sum_ = 0.0
    sum_sq = 0.0

    for offset in range(0, N, BLOCK_N):
        base = x_row_ptr + offset * stride_x1
        mask = (offset + tl.arange(0, BLOCK_N) < N)
        x_block = tl.load(base, mask=mask, other=0.0)
        sum_ += tl.sum(x_block)
        sum_sq += tl.sum(x_block * x_block)

    mean = sum_ / N
    var = sum_sq / N - mean * mean
    std = tl.sqrt(var + eps)

    # ---------- compute output ----------
    for offset in range(0, N, BLOCK_N):
        base_x = x_row_ptr + offset * stride_x1
        base_y = y_row_ptr + offset * stride_y1
        base_w = w_ptr + offset * stride_w
        base_b = b_ptr + offset * stride_b

        mask = (offset + tl.arange(0, BLOCK_N) < N)

        x_block = tl.load(base_x, mask=mask, other=0.0)
        w_block = tl.load(base_w, mask=mask, other=0.0)
        b_block = tl.load(base_b, mask=mask, other=0.0)

        y_block = (x_block - mean) / std * w_block + b_block
        tl.store(base_y, y_block, mask=mask)

def optimized_layernorm(x: torch.Tensor,
                        weight: torch.Tensor,
                        bias: torch.Tensor,
                        eps: float = 1e-5) -> torch.Tensor:
    """
    Layer normalization over the last dimension of a 2-D tensor using a Triton kernel.

    Args:
        x (torch.Tensor): Input tensor of shape (M, N).
        weight (torch.Tensor): Weight vector of shape (N,).
        bias (torch.Tensor): Bias vector of shape (N,).
        eps (float): Small value added to variance for numerical stability.

    Returns:
        torch.Tensor: Normalized output of shape (M, N).
    """
    assert x.ndim == 2, "Input must be 2-D"
    assert weight.shape == (x.shape[1],)
    assert bias.shape == (x.shape[1],)

    # Ensure tensors are contiguous and on the same device
    device = x.device
    dtype = x.dtype
    x = x.contiguous()
    weight = weight.contiguous()
    bias = bias.contiguous()

    M, N = x.shape
    y = torch.empty_like(x)

    # Strides (in elements)
    stride_x0, stride_x1 = x.stride()
    stride_y0, stride_y1 = y.stride()
    stride_w, stride_b = weight.stride(0), bias.stride(0)

    # Kernel launch parameters
    BLOCK_N = 256
    grid = (triton.cdiv(M, 1),)

    _layernorm_kernel[grid](
        x_ptr=x.data_ptr(),
        w_ptr=weight.data_ptr(),
        b_ptr=bias.data_ptr(),
        y_ptr=y.data_ptr(),
        M=M,
        N=N,
        stride_x0=stride_x0,
        stride_x1=stride_x1,
        stride_w=stride_w,
        stride_b=stride_b,
        stride_y0=stride_y0,
        stride_y1=stride_y1,
        eps=float(eps),
        BLOCK_N=BLOCK_N,
    )

    return y
