def pairwise_sq_dist(X):

    X = np.asarray(X, dtype=np.float64)

    G = X @ X.T

    sq_norm = np.einsum('ij,ij->i', X, X)  # or X**2 sum

    D = sq_norm[:,None] + sq_norm[None,:] - 2*G

    # due to floating errors, negative small values; clip
    np.maximum(D, 0, out=D)

    return D
