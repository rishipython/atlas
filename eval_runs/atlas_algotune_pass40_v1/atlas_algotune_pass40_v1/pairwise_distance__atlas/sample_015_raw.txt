import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Compute the full pairwise squared Euclidean distance matrix for a
    2‑D array X of shape (N, D).

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D) with dtype float64.

    Returns
    -------
    np.ndarray
        Distance matrix of shape (N, N) with dtype float64.  The entry
        (i, j) equals sum_k (X[i, k] - X[j, k])**2.
    """
    # Ensure input is float64 for consistent numeric results
    X = np.asarray(X, dtype=np.float64, order="C")
    # Compute squared norms of each row
    sq_norm = np.einsum("ij,ij->i", X, X)  # faster than sum of squares
    # Compute Gram matrix
    G = X @ X.T
    # Use the identity: ||a-b||^2 = ||a||^2 + ||b||^2 - 2 a·b
    D = sq_norm[:, None] + sq_norm[None, :] - 2.0 * G
    # Numerical errors may lead to tiny negative values; clip to zero
    np.maximum(D, 0.0, out=D)
    return D
