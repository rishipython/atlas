def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    dist2 = sq_norms[:, None] + sq_norms[None, :] - 2*np.dot(X, X.T)
    dist2[dist2 < 0] = 0  # due to numerical errors
    return dist2
