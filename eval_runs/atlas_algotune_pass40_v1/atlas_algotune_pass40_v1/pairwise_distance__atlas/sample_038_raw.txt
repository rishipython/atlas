import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray of shape (N, D), dtype float64

    Returns
    -------
    np.ndarray of shape (N, N), dtype float64
    """
    # Ensure we have a float64 array for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row
    sq_norms = np.sum(X * X, axis=1)

    # Use the identity: ||x - y||^2 = ||x||^2 + ||y||^2 - 2 x·y
    # Outer sum of squared norms plus the negative double dot product
    dist_sq = sq_norms[:, None] + sq_norms[None, :] - 2.0 * np.dot(X, X.T)

    # Numerical errors might produce tiny negative values; clip to zero
    np.maximum(dist_sq, 0.0, out=dist_sq)

    return dist_sq
