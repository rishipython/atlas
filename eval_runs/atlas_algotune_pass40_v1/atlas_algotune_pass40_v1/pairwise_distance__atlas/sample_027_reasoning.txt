def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq = np.sum(X**2, axis=1)
    dist = sq[:,None] + sq[None,:] - 2 * X @ X.T
    # due to floating errors, negative values small set to 0
    np.maximum(dist, 0, out=dist)
    return dist
