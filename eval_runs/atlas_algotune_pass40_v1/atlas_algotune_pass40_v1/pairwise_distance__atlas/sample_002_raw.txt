import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray
        Input array of shape (N, D), dtype float64.

    Returns
    -------
    np.ndarray
        Distance matrix of shape (N, N), dtype float64.
    """
    # Ensure input is float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row
    sq_norms = np.sum(X * X, axis=1)

    # Use the identity: ||x - y||^2 = ||x||^2 + ||y||^2 - 2 x·y
    # Outer sum of squared norms gives ||x||^2 + ||y||^2
    # The dot product matrix gives x·y
    distances = sq_norms[:, None] + sq_norms[None, :] - 2 * X.dot(X.T)

    # Numerical errors may produce tiny negative values; clamp to zero
    np.maximum(distances, 0, out=distances)

    return distances
