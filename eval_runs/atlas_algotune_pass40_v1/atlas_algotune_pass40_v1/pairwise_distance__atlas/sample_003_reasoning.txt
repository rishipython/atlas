def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    G = X @ X.T
    D = sq_norms[:, None] + sq_norms[None, :] - 2*G
    # numeric safety
    D[D < 0] = 0
    return D
