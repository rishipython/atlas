def pairwise_sq_dist(X):

    X = np.asarray(X, dtype=np.float64, order='C')
    N = X.shape[0]
    # compute dot product
    G = X @ X.T
    sq_norms = np.diag(G)
    D = sq_norms[:,None] + sq_norms[None,:] - 2*G
    # due to numerical errors, negative small values
    D = np.maximum(D, 0)
    return D
