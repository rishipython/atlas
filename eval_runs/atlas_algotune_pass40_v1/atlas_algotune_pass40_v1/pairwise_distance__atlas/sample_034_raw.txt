import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    This implementation uses vectorized matrix operations for speed:
    D[i, j] = ||X[i] - X[j]||^2 = ||X[i]||^2 + ||X[j]||^2 - 2 X[i]·X[j].

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row: shape (N,)
    sq_norms = np.sum(X**2, axis=1)

    # Broadcast to compute pairwise squared distances
    # D = sq_norms[:, None] + sq_norms[None, :] - 2 * X @ X.T
    # Using @ for matrix multiplication
    D = sq_norms[:, None] + sq_norms[None, :] - 2 * X @ X.T

    # Numerical errors can lead to tiny negative values; clip them to zero
    np.maximum(D, 0.0, out=D)

    return D
