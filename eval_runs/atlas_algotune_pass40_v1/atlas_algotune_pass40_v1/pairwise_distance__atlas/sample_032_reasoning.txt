def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    gram = X @ X.T
    dist = sq_norms[:, None] + sq_norms[None, :] - 2 * gram
    np.maximum(dist, 0, out=dist)  # clip negative to zero
    return dist
