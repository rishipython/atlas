def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.einsum('ij,ij->i', X, X)
    # compute pairwise
    dist = sq_norms[:,None] + sq_norms[None,:] - 2*np.dot(X, X.T)
    # due to numerical errors might be negative small values; clip.
    dist = np.maximum(dist, 0.0)
    return dist
