def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norm = np.sum(X**2, axis=1)
    D = sq_norm[:,None] + sq_norm[None,:] - 2*np.dot(X, X.T)
    # due to floating errors, set small negative to zero
    D[D<0] = 0
    return D
