import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Parameters
    ----------
    X : np.ndarray of shape (N, D), dtype float64.

    Returns
    -------
    out : np.ndarray of shape (N, N), dtype float64.
        out[i, j] = sum_k (X[i, k] - X[j, k])**2
    """
    # Ensure input is float64 for consistency
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row vector
    sq_norms = np.einsum('ij,ij->i', X, X)  # shape (N,)

    # Compute pairwise squared distances using the identity:
    # ||a - b||^2 = ||a||^2 + ||b||^2 - 2 a·b
    # Outer addition of norms and matrix multiplication for dot products
    dist = sq_norms[:, None] + sq_norms[None, :] - 2.0 * np.dot(X, X.T)

    # Numerical errors can produce tiny negative values; clip them to zero
    np.maximum(dist, 0.0, out=dist)

    return dist
