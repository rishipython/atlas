def pairwise_sq_dist(X): 
    sq_norms = np.sum(X**2, axis=1)
    # compute pairwise distances: sq_norms[:,None] + sq_norms[None,:] - 2*X.dot(X.T)
    # ensure no negative due to numerical errors: clip to zero.
    D = sq_norms[:,None] + sq_norms[None,:] - 2 * X.dot(X.T)
    np.maximum(D, 0, out=D)  # clip negative small values
    return D
