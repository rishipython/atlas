import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure input is float64 for consistent precision
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms for each row
    sq_norms = np.sum(X * X, axis=1)

    # Compute the distance matrix using the identity:
    # ||x_i - x_j||^2 = ||x_i||^2 + ||x_j||^2 - 2 * x_i·x_j
    dist_matrix = (
        sq_norms[:, np.newaxis] + sq_norms[np.newaxis, :] - 2.0 * X @ X.T
    )

    # Numerical errors may produce small negative values; clip to zero
    np.maximum(dist_matrix, 0.0, out=dist_matrix)

    return dist_matrix
