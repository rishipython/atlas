import numpy as np

def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norm = np.sum(X**2, axis=1)
    dist = sq_norm[:, None] + sq_norm[None, :] - 2 * X @ X.T
    # Numerical errors may produce small negative values
    np.maximum(dist, 0, out=dist)
    return dist
