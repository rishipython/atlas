import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) matrix of pairwise squared Euclidean distances
    for the rows of X.

    Parameters
    ----------
    X : np.ndarray
        Array of shape (N, D) containing N points in D-dimensional space.
        The array is cast to float64 if it is not already.

    Returns
    -------
    dist2 : np.ndarray
        Array of shape (N, N) where dist2[i, j] = ||X[i] - X[j]||^2.
        The result is guaranteed to be non-negative (up to machine epsilon).
    """
    # Ensure float64 precision
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row: shape (N,)
    norms = np.sum(X * X, axis=1)

    # Use the identity: ||a - b||^2 = ||a||^2 + ||b||^2 - 2 a·b
    # Compute pairwise dot product matrix: shape (N, N)
    dot_prod = X @ X.T

    # Broadcast norms to form the outer sum and subtract 2 * dot product
    dist2 = norms[:, None] + norms[None, :] - 2.0 * dot_prod

    # Numerical errors might produce tiny negative values; clip them to zero
    dist2 = np.maximum(dist2, 0.0)

    return dist2
