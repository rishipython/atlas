import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    X = np.asarray(X, dtype=np.float64)
    sum_X_sq = np.sum(X**2, axis=1).reshape(-1,1)
    dist = sum_X_sq + sum_X_sq.T - 2*np.dot(X, X.T)
    # Numerical errors: set small negative to zero
    dist = np.maximum(dist, 0.0)
    return dist
