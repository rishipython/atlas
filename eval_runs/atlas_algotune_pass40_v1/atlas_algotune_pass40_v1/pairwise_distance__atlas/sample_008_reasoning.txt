import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    X = np.asarray(X, dtype=np.float64)
    sum_sq = np.sum(X**2, axis=1)
    D = sum_sq[:,None] + sum_sq[None,:] - 2 * X.dot(X.T)
    # due to floating point errors, some negatives may appear
    D = np.maximum(D, 0.0)
    return D
