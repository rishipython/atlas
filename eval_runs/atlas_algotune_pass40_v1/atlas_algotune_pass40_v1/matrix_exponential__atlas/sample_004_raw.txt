```python
import numpy as np
from scipy.linalg import expm


def expm_fast(A: np.ndarray) -> np.ndarray:
    """
    Return the matrix exponential of a square float64 matrix ``A``.

    Parameters
    ----------
    A : np.ndarray
        A square matrix of type float64.

    Returns
    -------
    np.ndarray
        The matrix exponential of ``A``.
    """
    # SciPy's expm already implements a robust scaling-and-squaring
    # algorithm with a Padé approximant, which is far faster and more
    # accurate than a naive Taylor series.
    return expm(A.astype(np.float64))
```
