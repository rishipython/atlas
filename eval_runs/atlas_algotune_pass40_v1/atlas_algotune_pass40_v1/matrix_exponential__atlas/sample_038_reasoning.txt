import numpy as np
from scipy.linalg import expm

def expm_fast(A: np.ndarray) -> np.ndarray:
    """Return the matrix exponential of a square float64 matrix A."""
    A = np.asarray(A, dtype=np.float64)
    if A.ndim != 2 or A.shape[0] != A.shape[1]:
        raise ValueError("A must be a square matrix")
    return expm(A)

def pade13(A):
    n = A.shape[0]
    A2 = A @ A
    A4 = A2 @ A2
    A6 = A4 @ A2
    # compute powers up to A^13
    # Use A2, A4, A6 to build A^8, A^10, A^12 etc.
    A8 = A4 @ A4
    A10 = A8 @ A2
    A12 = A8 @ A4
    # For odd powers: A^1, A^3, A^5, A^7, A^9, A^11, A^13
    A3 = A2 @ A
    A5 = A4 @ A
    A7 = A6 @ A
    A9 = A8 @ A
    A11 = A10 @ A
    A13 = A12 @ A
    # compute U and V
    c = np.array([64764752532480000, 32382376266240000, 7771770303897600,
                  1187353796428800, 129060195264000, 10559470521600,
                  670442572800, 33522128640, 1323241920, 40840800,
                  960960, 16380, 182, 1], dtype=np.float64)
    # Even indices: 0,2,4,6,8,10,12
    # Odd indices: 1,3,5,7,9,11,13
    V = c[0]*np.eye(n) + c[2]*A2 + c[4]*A4 + c[6]*A6 + c[8]*A8 + c[10]*A10 + c[12]*A12
    U = c[1]*A + c[3]*A3 + c[5]*A5 + c[7]*A7 + c[9]*A9 + c[11]*A11 + c[13]*A13
    # Compute R = (V+U) @ inv(V-U)
    # Solve (V-U) X = V+U
    B = V - U
    C = V + U
    # Solve linear system B X = C for X
    R = np.linalg.solve(B, C)
    return R
