import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1  # length of the full convolution

    # If one of the arrays is empty, return an empty array
    if L == 0:
        return np.empty(0, dtype=np.float64)

    # Use the next power of two for efficient FFT (optional but faster on large data)
    size = 1 << (L - 1).bit_length()

    # FFT of the zero‑padded inputs
    X = np.fft.rfft(x, size)
    H = np.fft.rfft(h, size)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get the convolution result
    y = np.fft.irfft(Y, size)[:L]

    return y
