import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT."""
    # Ensure inputs are 1‑D and float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    # Length of full convolution
    out_len = N + M - 1

    # Perform convolution via FFT using real‑to‑complex transforms
    X = np.fft.rfft(x, n=out_len)
    H = np.fft.rfft(h, n=out_len)
    Y = X * H
    y = np.fft.irfft(Y, n=out_len)

    return y.astype(np.float64)
