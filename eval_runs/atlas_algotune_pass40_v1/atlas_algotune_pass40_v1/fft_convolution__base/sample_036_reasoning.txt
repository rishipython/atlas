def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    N = x.shape[0]
    M = h.shape[0]
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)
    size = N + M - 1
    # Compute FFTs
    X = np.fft.rfft(x, n=size)
    H = np.fft.rfft(h, n=size)
    Y = X * H
    y = np.fft.irfft(Y, n=size)
    return y
