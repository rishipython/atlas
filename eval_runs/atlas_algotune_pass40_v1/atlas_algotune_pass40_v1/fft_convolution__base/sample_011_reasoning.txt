def conv1d(x, h):
    N = x.shape[0]; M = h.shape[0]
    L = N + M - 1
    # next power of 2? Not needed but maybe faster
    nfft = 1 << (L-1).bit_length()
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    Y = X * H
    y = np.fft.irfft(Y, nfft)[:L]
    return y
