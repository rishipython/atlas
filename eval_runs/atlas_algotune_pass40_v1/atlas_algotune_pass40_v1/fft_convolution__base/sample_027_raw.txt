```python
import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 arrays using FFT.
    This is equivalent to np.convolve(x, h, mode='full') but is much faster
    for large inputs.
    """
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the linear convolution
    L = N + M - 1

    # Use the next power of two for efficient FFT
    L_fft = 1 << (L - 1).bit_length()

    # Compute FFTs of zero-padded signals
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get time-domain convolution
    y = np.fft.irfft(Y, n=L_fft)

    # Truncate to the true convolution length
    return y[:L]
```
