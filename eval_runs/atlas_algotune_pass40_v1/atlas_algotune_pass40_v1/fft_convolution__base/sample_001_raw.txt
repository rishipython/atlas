import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster."""
    # Ensure input arrays are 1-D float64
    x = np.asarray(x, dtype=np.float64, order='C').ravel()
    h = np.asarray(h, dtype=np.float64, order='C').ravel()

    N, M = x.size, h.size
    L = N + M - 1  # length of full convolution

    # Next power of two for efficient FFT (optional, but usually faster)
    nfft = 1 << (L - 1).bit_length()

    # Compute FFTs using real FFT for speed
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT and truncate to the exact convolution length
    y = np.fft.irfft(Y, nfft)[:L]

    return y
