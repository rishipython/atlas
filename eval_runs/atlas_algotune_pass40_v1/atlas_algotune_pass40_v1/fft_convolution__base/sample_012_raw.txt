import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return ``np.convolve(x, h, mode='full')`` but MUCH faster using FFT."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    out_len = N + M - 1

    # Handle empty input
    if out_len <= 0:
        return np.array([], dtype=np.float64)

    # Compute next power of two for efficient FFT
    L = 1 << (out_len - 1).bit_length()

    # FFT of the zero‑padded inputs (using real FFT for speed)
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT to get the convolution result
    y = np.fft.irfft(Y, n=L)

    # Truncate to the true convolution length
    return y[:out_len].astype(np.float64)
