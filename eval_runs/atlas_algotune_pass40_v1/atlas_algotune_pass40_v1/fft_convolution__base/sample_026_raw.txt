import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return np.convolve(x, h, mode='full') but MUCH faster."""
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    out_len = N + M - 1

    # Use the real FFT for speed (works for real input signals)
    # Pad to the required output length
    X = np.fft.rfft(x, n=out_len)
    H = np.fft.rfft(h, n=out_len)
    Y = X * H
    y = np.fft.irfft(Y, n=out_len)

    # The result may contain tiny imaginary parts due to numerical errors
    # but irfft returns a real array. Ensure dtype consistency.
    return y.astype(np.float64)
