import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays using FFT."""
    # Ensure inputs are 1‑D and float64
    x = np.asarray(x, dtype=np.float64, copy=False).ravel()
    h = np.asarray(h, dtype=np.float64, copy=False).ravel()

    N, M = x.size, h.size
    L = N + M - 1  # desired output length

    # Choose FFT length as next power of two for speed (optional)
    nfft = 1 << (L - 1).bit_length()

    # Compute FFTs of zero‑padded inputs
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result
    y = np.fft.irfft(Y, nfft)

    # Truncate to the expected length and return a copy
    return y[:L].copy()
