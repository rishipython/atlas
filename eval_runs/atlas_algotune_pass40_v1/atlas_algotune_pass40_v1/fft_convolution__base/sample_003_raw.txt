```python
import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 arrays using FFT.
    Equivalent to np.convolve(x, h, mode='full') but much faster for large arrays.
    """
    # Ensure inputs are numpy arrays of dtype float64
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.shape[0]
    M = h.shape[0]

    # Handle empty inputs: np.convolve returns an empty array when either input is empty
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    # Length of the full convolution result
    L = N + M - 1

    # Choose an FFT length that is a fast size (next power of two or similar)
    L_fft = np.fft.next_fast_len(L)

    # Perform real FFTs, multiply, and inverse FFT
    X = np.fft.rfft(x, n=L_fft)
    H = np.fft.rfft(h, n=L_fft)
    Y = X * H
    y = np.fft.irfft(Y, n=L_fft)

    # Truncate to the exact output length
    return y[:L]
```
