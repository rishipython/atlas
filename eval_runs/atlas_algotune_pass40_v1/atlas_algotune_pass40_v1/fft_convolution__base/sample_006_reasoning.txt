import numpy as np
try:
    from scipy.signal import fftconvolve
except Exception:
    def fftconvolve(x,h,mode='full'):
        # simple fallback using numpy.fft
        n = x.shape[0]
        m = h.shape[0]
        size = n+m-1
        size_fft = 1 << (size-1).bit_length()
        X = np.fft.rfft(x, size_fft)
        H = np.fft.rfft(h, size_fft)
        y = np.fft.irfft(X*H, size_fft)
        return y[:size]

def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    return fftconvolve(x,h,mode='full')
