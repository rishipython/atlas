```python
import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 arrays using an
    FFT-based approach for speed.  The result is numerically close to
    `np.convolve(x, h, mode='full')`.

    Parameters
    ----------
    x : np.ndarray
        Input array of shape (N,).
    h : np.ndarray
        Input array of shape (M,).

    Returns
    -------
    y : np.ndarray
        Convolution of `x` and `h` with shape (N + M - 1,).
    """
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size

    # Handle empty inputs: np.convolve([], a) or np.convolve(a, []) returns empty array
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the full convolution
    outlen = N + M - 1

    # Choose FFT size: next power of two for speed (optional)
    fft_size = 1 << (outlen - 1).bit_length()

    # Compute FFTs of zero-padded inputs
    X = np.fft.rfft(x, fft_size)
    H = np.fft.rfft(h, fft_size)

    # Pointwise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain convolution result
    y = np.fft.irfft(Y, fft_size)[:outlen]

    return y
```
