```python
import numpy as np
from scipy.linalg import expm

def expm_fast(A: np.ndarray) -> np.ndarray:
    """Return the matrix exponential of a square float64 matrix ``A``.

    This implementation delegates to :func:`scipy.linalg.expm`, which uses a
    robust scaling-and-squaring algorithm with Padé approximants.  It is
    considerably faster and more numerically stable than a naive Taylor
    series truncation.

    Parameters
    ----------
    A : np.ndarray
        Square matrix of dtype ``float64``.

    Returns
    -------
    np.ndarray
        The matrix exponential of ``A``.
    """
    return expm(A)
```
