```python
import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Compute squared norms of each row
    sq_norm = np.sum(X ** 2, axis=1)
    # Compute pairwise squared distances using the identity:
    # ||x - y||^2 = ||x||^2 + ||y||^2 - 2 * x·y
    # Broadcast sq_norm to form the outer sum
    D = sq_norm[:, None] + sq_norm[None, :] - 2.0 * np.dot(X, X.T)
    # Numerical errors might produce tiny negative values; clip them to zero
    np.maximum(D, 0.0, out=D)
    return D
```
