def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    D = sq_norms[:, None] + sq_norms[None, :] - 2 * X @ X.T
    # due to numerical errors, small negative values may appear
    np.maximum(D, 0, out=D)
    return D
