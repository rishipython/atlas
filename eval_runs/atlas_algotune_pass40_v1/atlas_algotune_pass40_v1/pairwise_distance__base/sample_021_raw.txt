import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    This implementation uses the algebraic identity
        ||x_i - x_j||^2 = ||x_i||^2 + ||x_j||^2 - 2 x_i·x_j
    to compute the matrix in O(N^2) time with efficient NumPy operations.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure the input is a float64 array
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row
    sq_norm = np.sum(X * X, axis=1)  # shape (N,)

    # Use broadcasting to compute pairwise squared distances
    D = sq_norm[:, None] + sq_norm[None, :] - 2.0 * X @ X.T

    # Numerical errors may lead to tiny negative values; clip them to zero
    np.maximum(D, 0.0, out=D)

    return D
