def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    N, D = X.shape
    if N == 0:
        return np.empty((0,0), dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    Dmat = sq_norms[:, None] + sq_norms[None, :] - 2.0 * X.dot(X.T)
    # Numerical errors may produce tiny negative values
    Dmat = np.maximum(Dmat, 0.0)
    return Dmat
