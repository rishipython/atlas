```python
import numpy as np

def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """
    Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Compute squared norms of each row
    norms = np.sum(X ** 2, axis=1, keepdims=True)  # shape (N, 1)

    # Compute distance matrix using the identity:
    # ||x - y||^2 = ||x||^2 + ||y||^2 - 2 * x·y
    # The dot product matrix has shape (N, N)
    dot_product = X @ X.T

    # Broadcast norms to create the sum of norms matrix
    out = norms + norms.T - 2.0 * dot_product

    # Numerical errors can produce tiny negative values; clip them to zero
    out = np.maximum(out, 0.0)

    return out
```
