import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    This implementation uses vectorised linear algebra operations
    to achieve a significant speedup over a naive triple-loop.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure the input is a float64 array
    X = np.asarray(X, dtype=np.float64)
    # Compute squared norms of each row
    sq_norms = np.sum(X * X, axis=1)
    # Use the identity: ||a - b||^2 = ||a||^2 + ||b||^2 - 2 a·b
    # Broadcast to obtain the full distance matrix
    D = sq_norms[:, None] + sq_norms[None, :] - 2.0 * X @ X.T
    # Numerical errors can lead to tiny negative values; clip them to zero
    D = np.maximum(D, 0.0)
    return D
