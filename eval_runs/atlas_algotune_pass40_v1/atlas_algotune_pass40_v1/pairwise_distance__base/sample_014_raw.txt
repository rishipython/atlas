import numpy as np


def pairwise_sq_dist(X: np.ndarray) -> np.ndarray:
    """Return the (N, N) pairwise squared Euclidean distance matrix.

    Args:
        X: np.ndarray of shape (N, D), dtype float64.

    Returns:
        np.ndarray of shape (N, N), dtype float64.
    """
    # Ensure float64 for numerical stability
    X = np.asarray(X, dtype=np.float64)

    # Compute squared norms of each row
    norms = np.sum(X * X, axis=1, keepdims=True)  # shape (N, 1)

    # Compute Gram matrix
    gram = X @ X.T  # shape (N, N)

    # Using the identity: ||x - y||^2 = ||x||^2 + ||y||^2 - 2*x·y
    # Broadcast norms to add pairwise sums
    dists = norms + norms.T - 2.0 * gram

    # Numerical errors may cause tiny negative values; clip to zero
    np.maximum(dists, 0.0, out=dists)

    return dists
