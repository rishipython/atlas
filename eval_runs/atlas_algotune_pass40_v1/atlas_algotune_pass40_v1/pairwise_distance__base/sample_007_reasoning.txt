def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    norms = np.sum(X**2, axis=1)
    D = norms[:, None] + norms[None, :] - 2 * X @ X.T

    np.maximum(D, 0, out=D)
    return D
