def pairwise_sq_dist(X):
    X = np.asarray(X, dtype=np.float64)
    sq_norms = np.sum(X**2, axis=1)
    dot = X @ X.T
    out = sq_norms[:, None] + sq_norms[None, :] - 2 * dot
    # Numerical errors might cause small negative values
    out = np.maximum(out, 0.0)
    return out
