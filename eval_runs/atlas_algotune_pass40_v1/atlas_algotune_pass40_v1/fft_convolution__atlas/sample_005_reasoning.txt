def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.size
    M = h.size
    size = N + M - 1
    # Use next power of two for speed
    nfft = 1 << (size-1).bit_length()
    X = np.fft.fft(x, nfft)
    H = np.fft.fft(h, nfft)
    Y = X*H
    y = np.fft.ifft(Y).real
    return y[:size]
