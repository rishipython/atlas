def conv1d(x,h):
    N = x.shape[0]; M = h.shape[0]
    size = N+M-1
    nfft = 1 << (size-1).bit_length()  # next power of 2
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    y = np.fft.irfft(X*H, nfft)[:size]
    return y
