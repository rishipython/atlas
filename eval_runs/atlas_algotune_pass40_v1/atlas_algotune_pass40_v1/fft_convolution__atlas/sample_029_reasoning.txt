```
def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.size
    M = h.size
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)
    n = N + M - 1
    # next power of 2? Not necessary, but faster maybe. Use n
    X = np.fft.fft(x, n=n)
    H = np.fft.fft(h, n=n)
    y = np.fft.ifft(X * H).real
    # rounding small imaginary due to numerical errors
    return y
```
