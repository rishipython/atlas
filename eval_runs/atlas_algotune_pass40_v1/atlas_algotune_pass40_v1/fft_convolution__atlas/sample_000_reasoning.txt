def conv1d(x, h):
    N, M = x.shape[0], h.shape[0]
    nconv = N + M - 1
    L = 1 << (nconv - 1).bit_length()
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y[:nconv]
