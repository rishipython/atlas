import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D real-valued arrays
    using an FFT-based algorithm for speed.

    Parameters
    ----------
    x : np.ndarray
        First input array, shape (N,).
    h : np.ndarray
        Second input array, shape (M,).

    Returns
    -------
    y : np.ndarray
        Convolution result, shape (N + M - 1,).
    """
    # Ensure inputs are 1-D and of float64 dtype
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()
    N, M = x.size, h.size
    out_len = N + M - 1

    # Zero padding to next power of two for efficient FFT
    fft_len = 1 << (out_len - 1).bit_length()

    # Compute forward FFTs
    X = np.fft.rfft(x, n=fft_len)
    H = np.fft.rfft(h, n=fft_len)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get convolution result
    y = np.fft.irfft(Y, n=fft_len)

    # Truncate to the exact output length
    return y[:out_len]
