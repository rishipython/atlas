def conv1d(x,h):
    N = len(x)
    M = len(h)
    n = N+M-1
    NFFT = 1 << (n-1).bit_length()  # next power of two >= n
    X = np.fft.rfft(x, NFFT)
    H = np.fft.rfft(h, NFFT)
    y = np.fft.irfft(X*H, NFFT)[:n]
    return y
