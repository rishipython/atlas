def conv1d(x,h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    n_out = N + M -1
    L = 1 << (n_out-1).bit_length()  # next power of two
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y[:n_out]
