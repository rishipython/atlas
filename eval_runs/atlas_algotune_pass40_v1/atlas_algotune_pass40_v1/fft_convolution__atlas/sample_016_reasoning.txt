```
import numpy as np
from numpy.fft import rfft, irfft

def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.shape[0]
    M = h.shape[0]
    out_len = N + M - 1
    L = np.fft.next_fast_len(out_len)
    X = rfft(x, n=L)
    H = rfft(h, n=L)
    Y = irfft(X * H, n=L)
    return Y[:out_len]
```
