def conv1d(x, h):
    N = x.shape[0]
    M = h.shape[0]
    L = 1 << (N+M-1-1).bit_length()
    X = np.fft.rfft(x, L)
    H = np.fft.rfft(h, L)
    y = np.fft.irfft(X * H, L)
    return y[:N+M-1].copy()
