import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1‑D float64 arrays using an FFT‑based algorithm.

    Parameters
    ----------
    x : np.ndarray
        Input array of length N.
    h : np.ndarray
        Input array of length M.

    Returns
    -------
    y : np.ndarray
        Convolution result of length N + M - 1.
    """
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    n, m = x.size, h.size
    # Handle empty inputs
    if n == 0 or m == 0:
        return np.array([], dtype=np.float64)

    # Length of the full convolution
    conv_len = n + m - 1

    # Next power of two for efficient FFT
    L = 1 << (conv_len - 1).bit_length()

    # Compute real FFTs
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Pointwise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT and truncate to the desired length
    y = np.fft.irfft(Y, n=L)[:conv_len]

    return y
