def conv1d(x,h):
    N=len(x); M=len(h)
    size = N+M-1
    # next power of two
    nfft = 1 << (size-1).bit_length()
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)
    Y = X * H
    y = np.fft.irfft(Y, nfft)
    return y[:size]
