def conv1d(x,h):
    N=len(x); M=len(h)
    L=N+M-1
    nfft=np.fft.next_fast_len(L)
    X=np.fft.rfft(x, nfft)
    H=np.fft.rfft(h, nfft)
    y=np.fft.irfft(X*H, nfft)
    return y[:L]
