def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    N = x.size
    M = h.size
    n = N + M - 1
    n_fft = 1 << (n - 1).bit_length()  # next power of 2 >= n
    X = np.fft.fft(x, n_fft)
    H = np.fft.fft(h, n_fft)
    y = np.fft.ifft(X * H).real
    return y[:n]
