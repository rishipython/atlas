import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1‑D float64 arrays.
    This implementation uses an FFT‑based approach for speed.
    """
    # Ensure inputs are float64 arrays
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)

    N = x.size
    M = h.size

    # Handle empty inputs
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the full convolution
    L = N + M - 1

    # Next power of two for efficient FFT (zero‑pad to this length)
    n_fft = 1 << (L - 1).bit_length()

    # Compute FFTs of zero‑padded signals
    X = np.fft.rfft(x, n=n_fft)
    H = np.fft.rfft(h, n=n_fft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to get convolution result (real‑valued)
    y = np.fft.irfft(Y, n=n_fft)

    # Truncate to the exact convolution length
    return y[:L]
