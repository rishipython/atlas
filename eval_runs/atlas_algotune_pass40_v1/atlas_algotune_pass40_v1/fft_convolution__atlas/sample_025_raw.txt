```python
import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 arrays
    using an FFT-based algorithm for speed.

    Parameters
    ----------
    x : np.ndarray
        First input array of shape (N,).
    h : np.ndarray
        Second input array of shape (M,).

    Returns
    -------
    y : np.ndarray
        Convolution result of shape (N + M - 1,).
    """
    # Ensure inputs are 1-D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    # Length of the linear convolution result
    out_len = N + M - 1

    # Next power of two for efficient FFT
    n_fft = 1 << (out_len - 1).bit_length()

    # FFT of both signals padded to n_fft
    X = np.fft.rfft(x, n_fft)
    H = np.fft.rfft(h, n_fft)

    # Elementwise multiplication in frequency domain
    Y = X * H

    # Inverse FFT to obtain the time-domain convolution
    y = np.fft.irfft(Y, n_fft)

    # Truncate to the exact linear convolution length
    return y[:out_len]
```
