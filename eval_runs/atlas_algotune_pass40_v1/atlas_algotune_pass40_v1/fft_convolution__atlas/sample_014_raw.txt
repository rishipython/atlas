import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1‑D float64 arrays using FFT.

    Parameters
    ----------
    x : np.ndarray
        Input array of length N.
    h : np.ndarray
        Input array of length M.

    Returns
    -------
    y : np.ndarray
        The full convolution of shape (N + M - 1,).
    """
    # Ensure inputs are 1‑D float64 arrays
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.shape[0]
    M = h.shape[0]
    L = N + M - 1

    # If either array is empty, return an empty array
    if L <= 0:
        return np.array([], dtype=np.float64)

    # Choose FFT length: next power of two for speed
    nfft = 1 << (L - 1).bit_length()  # next power of two >= L

    # Compute real FFTs of zero‑padded inputs
    X = np.fft.rfft(x, n=nfft)
    H = np.fft.rfft(h, n=nfft)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT and truncate to the required length
    y = np.fft.irfft(Y, n=nfft)[:L]

    return y
