```
import numpy as np

def conv1d(x, h):
    x = np.asarray(x, dtype=np.float64)
    h = np.asarray(h, dtype=np.float64)
    n = x.size
    m = h.size
    if n == 0 or m == 0:
        return np.array([], dtype=np.float64)
    size = n + m - 1
    L = 1 << (size-1).bit_length()  # next power of two >= size
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)
    Y = X * H
    y = np.fft.irfft(Y, n=L)
    return y[:size]
```
