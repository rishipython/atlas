```python
import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of x and h, computed via FFT.

    Parameters
    ----------
    x, h : np.ndarray
        1-D arrays of type float64 (or castable to it).

    Returns
    -------
    y : np.ndarray
        The convolution of x and h, length ``len(x)+len(h)-1``.
    """
    # Ensure input is 1‑D and float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size
    out_len = N + M - 1

    if out_len == 0:
        return np.array([], dtype=np.float64)

    # Choose FFT length: next power of two for speed
    nfft = 1 << (out_len - 1).bit_length()

    # Compute FFTs
    X = np.fft.fft(x, nfft)
    H = np.fft.fft(h, nfft)

    # Point‑wise multiplication
    Y = X * H

    # Inverse FFT and take real part
    y = np.fft.ifft(Y).real[:out_len]
    return y
```
