import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1-D float64 arrays.
    Equivalent to np.convolve(x, h, mode='full') but uses FFT for speed.

    Parameters
    ----------
    x : np.ndarray
        Input array of length N.
    h : np.ndarray
        Input array of length M.

    Returns
    -------
    y : np.ndarray
        Convolution result of length N + M - 1.
    """
    # Ensure inputs are 1-D float64
    x = np.asarray(x, dtype=np.float64, order='C')
    h = np.asarray(h, dtype=np.float64, order='C')

    N = x.shape[0]
    M = h.shape[0]
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    out_len = N + M - 1

    # Next power of two for efficient FFT
    nfft = 1 << (out_len - 1).bit_length()

    # Compute real FFTs
    X = np.fft.rfft(x, nfft)
    H = np.fft.rfft(h, nfft)

    # Element-wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT and truncate to the desired length
    y = np.fft.irfft(Y, nfft, norm=None)[:out_len]

    return y.astype(np.float64, copy=False)
