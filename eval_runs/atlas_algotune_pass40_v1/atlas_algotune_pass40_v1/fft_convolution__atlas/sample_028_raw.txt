import numpy as np


def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """Return the full linear convolution of two 1‑D float64 arrays.

    The implementation uses FFT‑based convolution for speed while
    maintaining numerical accuracy close to that of
    ``np.convolve(x, h, mode='full')``.
    """
    # Ensure inputs are 1‑D and cast to float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N, M = x.size, h.size
    if N == 0 or M == 0:
        return np.empty(0, dtype=np.float64)

    # Length of the linear convolution result
    conv_len = N + M - 1

    # Use next power of two for efficient FFT
    L = 1 << (conv_len - 1).bit_length()

    # Compute real FFTs
    X = np.fft.rfft(x, n=L)
    H = np.fft.rfft(h, n=L)

    # Element‑wise multiplication in frequency domain
    Y = X * H

    # Inverse real FFT to get back to time domain
    y = np.fft.irfft(Y, n=L)

    # Truncate to the exact convolution length
    return y[:conv_len]
