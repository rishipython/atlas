import numpy as np

def conv1d(x: np.ndarray, h: np.ndarray) -> np.ndarray:
    """
    Return the full linear convolution of two 1‑D float64 arrays using FFT.
    Equivalent to np.convolve(x, h, mode='full') but much faster for large inputs.
    """
    # Ensure inputs are 1-D and cast to float64
    x = np.asarray(x, dtype=np.float64).ravel()
    h = np.asarray(h, dtype=np.float64).ravel()

    N = x.size
    M = h.size
    # Handle empty inputs (numpy convolve returns an empty array)
    if N == 0 or M == 0:
        return np.array([], dtype=np.float64)

    n = N + M - 1  # length of the full convolution

    # Next power of two for efficient FFT (optional, can also use n directly)
    NFFT = 1 << (n - 1).bit_length()

    # Compute FFTs of zero‑padded inputs
    X = np.fft.rfft(x, NFFT)
    H = np.fft.rfft(h, NFFT)

    # Element‑wise multiplication in frequency domain and inverse FFT
    y = np.fft.irfft(X * H, NFFT)[:n]

    return y
