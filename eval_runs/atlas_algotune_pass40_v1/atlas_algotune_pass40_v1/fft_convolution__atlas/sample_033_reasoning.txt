def conv1d(x,h):
    N=x.size; M=h.size; L=N+M-1; nfft=1<< (L-1).bit_length()
    X=np.fft.rfft(x, nfft)
    H=np.fft.rfft(h, nfft)
    y=np.fft.irfft(X*H, nfft)
    return y[:L]
